104# FK4-NO-E to/from FK4 -----------------------------> 

105# Unlike other frames, this module include *two* frame classes for FK4 

106# coordinates - one including the E-terms of aberration (FK4), and 

107# one not including them (FK4NoETerms). The following functions 

108# implement the transformation between these two. 

109def fk4_e_terms(equinox): 

110 """ 

111 Return the e-terms of aberration vector. 

112 

113 Parameters 

114 ---------- 

115 equinox : Time object 

116 The equinox for which to compute the e-terms 

117 """ 

118 # Constant of aberration at J2000; from Explanatory Supplement to the 

119 # Astronomical Almanac (Seidelmann, 2005). 

120 k = 0.0056932 # in degrees (v_earth/c ~ 1e-4 rad ~ 0.0057 deg) 

121 k = np.radians(k) 

122 

123 # Eccentricity of the Earth's orbit 

124 e = earth.eccentricity(equinox.jd) 

125 

126 # Mean longitude of perigee of the solar orbit 

127 g = earth.mean_lon_of_perigee(equinox.jd) 

128 g = np.radians(g) 

129 

130 # Obliquity of the ecliptic 

131 o = earth.obliquity(equinox.jd, algorithm=1980) 

132 o = np.radians(o) 

133 

134 return ( 

135 e * k * np.sin(g), 

136 -e * k * np.cos(g) * np.cos(o), 

137 -e * k * np.cos(g) * np.sin(o), 

138 )

141@frame_transform_graph.transform( 

142 FunctionTransformWithFiniteDifference, FK4, FK4NoETerms 

143) 

144def fk4_to_fk4_no_e(fk4coord, fk4noeframe): 

145 # Extract cartesian vector 

146 rep = fk4coord.cartesian 

147 

148 # Find distance (for re-normalization) 

149 d_orig = rep.norm() 

150 rep /= d_orig 

151 

152 # Apply E-terms of aberration. Note that this depends on the equinox (not 

153 # the observing time/epoch) of the coordinates. See issue #1496 for a 

154 # discussion of this. 

155 eterms_a = CartesianRepresentation( 

156 u.Quantity(fk4_e_terms(fk4coord.equinox), u.dimensionless_unscaled, copy=False), 

157 copy=False, 

158 ) 

159 rep = rep - eterms_a + eterms_a.dot(rep) * rep 

160 

161 # Find new distance (for re-normalization) 

162 d_new = rep.norm() 

163 

164 # Renormalize 

165 rep *= d_orig / d_new 

166 

167 # now re-cast into an appropriate Representation, and precess if need be 

168 if isinstance(fk4coord.data, UnitSphericalRepresentation): 

169 rep = rep.represent_as(UnitSphericalRepresentation) 

170 

171 # if no obstime was given in the new frame, use the old one for consistency 

172 newobstime = ( 

173 fk4coord._obstime if fk4noeframe._obstime is None else fk4noeframe._obstime 

174 ) 

175 

176 fk4noe = FK4NoETerms(rep, equinox=fk4coord.equinox, obstime=newobstime) 

177 if fk4coord.equinox != fk4noeframe.equinox: 

178 # precession 

179 fk4noe = fk4noe.transform_to(fk4noeframe) 

180 return fk4noe
