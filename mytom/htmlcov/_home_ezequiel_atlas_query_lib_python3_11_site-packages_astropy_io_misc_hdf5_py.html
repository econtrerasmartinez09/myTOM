63def read_table_hdf5(input, path=None, character_as_bytes=True): 

64 """ 

65 Read a Table object from an HDF5 file. 

66 

67 This requires `h5py <http://www.h5py.org/>`_ to be installed. If more than one 

68 table is present in the HDF5 file or group, the first table is read in and 

69 a warning is displayed. 

70 

71 Parameters 

72 ---------- 

73 input : str or :class:`h5py.File` or :class:`h5py.Group` or 

74 :class:`h5py.Dataset` If a string, the filename to read the table from. 

75 If an h5py object, either the file or the group object to read the 

76 table from. 

77 path : str 

78 The path from which to read the table inside the HDF5 file. 

79 This should be relative to the input file or group. 

80 character_as_bytes : bool 

81 If `True` then Table columns are left as bytes. 

82 If `False` then Table columns are converted to unicode. 

83 """ 

84 try: 

85 import h5py 

86 except ImportError: 

87 raise Exception("h5py is required to read and write HDF5 files") 

88 

89 # This function is iterative, and only gets to writing the file when 

90 # the input is an hdf5 Group. Moreover, the input variable is changed in 

91 # place. 

92 # Here, we save its value to be used at the end when the conditions are 

93 # right. 

94 input_save = input 

95 if isinstance(input, (h5py.File, h5py.Group)): 

96 # If a path was specified, follow the path 

97 

98 if path is not None: 

99 try: 

100 input = input[path] 

101 except (KeyError, ValueError): 

102 raise OSError(f"Path {path} does not exist") 

103 

104 # `input` is now either a group or a dataset. If it is a group, we 

105 # will search for all structured arrays inside the group, and if there 

106 # is one we can proceed otherwise an error is raised. If it is a 

107 # dataset, we just proceed with the reading. 

108 

109 if isinstance(input, h5py.Group): 

110 # Find all structured arrays in group 

111 arrays = _find_all_structured_arrays(input) 

112 

113 if len(arrays) == 0: 

114 raise ValueError(f"no table found in HDF5 group {path}") 

115 elif len(arrays) > 0: 

116 path = arrays[0] if path is None else path + "/" + arrays[0] 

117 if len(arrays) > 1: 

118 warnings.warn( 

119 "path= was not specified but multiple tables" 

120 " are present, reading in first available" 

121 f" table (path={path})", 

122 AstropyUserWarning, 

123 ) 

124 return read_table_hdf5(input, path=path) 

125 

126 elif not isinstance(input, h5py.Dataset): 

127 # If a file object was passed, then we need to extract the filename 

128 # because h5py cannot properly read in file objects. 

129 

130 if hasattr(input, "read"): 

131 try: 

132 input = input.name 

133 except AttributeError: 

134 raise TypeError("h5py can only open regular files") 

135 

136 # Open the file for reading, and recursively call read_table_hdf5 with 

137 # the file object and the path. 

138 

139 f = h5py.File(input, "r") 

140 

141 try: 

142 return read_table_hdf5(f, path=path, character_as_bytes=character_as_bytes) 

143 finally: 

144 f.close() 

145 

146 # If we are here, `input` should be a Dataset object, which we can now 

147 # convert to a Table. 

148 

149 # Create a Table object 

150 from astropy.table import Table, meta, serialize 

151 

152 table = Table(np.array(input)) 

153 

154 # Read the meta-data from the file. For back-compatibility, we can read 

155 # the old file format where the serialized metadata were saved in the 

156 # attributes of the HDF5 dataset. 

157 # In the new format, instead, metadata are stored in a new dataset in the 

158 # same file. This is introduced in Astropy 3.0 

159 old_version_meta = META_KEY in input.attrs 

160 new_version_meta = path is not None and meta_path(path) in input_save 

161 if old_version_meta or new_version_meta: 

162 if new_version_meta: 

163 header = meta.get_header_from_yaml( 

164 h.decode("utf-8") for h in input_save[meta_path(path)] 

165 ) 

166 else: 

167 # Must be old_version_meta is True. if (A or B) and not A then B is True 

168 header = meta.get_header_from_yaml( 

169 h.decode("utf-8") for h in input.attrs[META_KEY] 

170 ) 

171 if "meta" in list(header.keys()): 

172 table.meta = header["meta"] 

173 

174 header_cols = {x["name"]: x for x in header["datatype"]} 

175 for col in table.columns.values(): 

176 for attr in ("description", "format", "unit", "meta"): 

177 if attr in header_cols[col.name]: 

178 setattr(col, attr, header_cols[col.name][attr]) 

179 

180 # Construct new table with mixins, using tbl.meta['__serialized_columns__'] 

181 # as guidance. 

182 table = serialize._construct_mixins_from_columns(table) 

183 

184 else: 

185 # Read the meta-data from the file 

186 table.meta.update(input.attrs) 

187 

188 if not character_as_bytes: 

189 table.convert_bytestring_to_unicode() 

190 

191 return table

194def _encode_mixins(tbl): 

195 """Encode a Table ``tbl`` that may have mixin columns to a Table with only 

196 astropy Columns + appropriate meta-data to allow subsequent decoding. 

197 """ 

198 from astropy.table import serialize 

199 from astropy.utils.data_info import serialize_context_as 

200 

201 # Convert the table to one with no mixins, only Column objects. This adds 

202 # meta data which is extracted with meta.get_yaml_from_table. 

203 with serialize_context_as("hdf5"): 

204 encode_tbl = serialize.represent_mixins_as_columns(tbl) 

205 

206 return encode_tbl

209def write_table_hdf5( 

210 table, 

211 output, 

212 path=None, 

213 compression=False, 

214 append=False, 

215 overwrite=False, 

216 serialize_meta=False, 

217 **create_dataset_kwargs, 

218): 

219 """ 

220 Write a Table object to an HDF5 file. 

221 

222 This requires `h5py <http://www.h5py.org/>`_ to be installed. 

223 

224 Parameters 

225 ---------- 

226 table : `~astropy.table.Table` 

227 Data table that is to be written to file. 

228 output : str or :class:`h5py.File` or :class:`h5py.Group` 

229 If a string, the filename to write the table to. If an h5py object, 

230 either the file or the group object to write the table to. 

231 path : str 

232 The path to which to write the table inside the HDF5 file. 

233 This should be relative to the input file or group. 

234 If not specified, defaults to ``__astropy_table__``. 

235 compression : bool or str or int 

236 Whether to compress the table inside the HDF5 file. If set to `True`, 

237 ``'gzip'`` compression is used. If a string is specified, it should be 

238 one of ``'gzip'``, ``'szip'``, or ``'lzf'``. If an integer is 

239 specified (in the range 0-9), ``'gzip'`` compression is used, and the 

240 integer denotes the compression level. 

241 append : bool 

242 Whether to append the table to an existing HDF5 file. 

243 overwrite : bool 

244 Whether to overwrite any existing file without warning. 

245 If ``append=True`` and ``overwrite=True`` then only the dataset will be 

246 replaced; the file/group will not be overwritten. 

247 serialize_meta : bool 

248 Whether to serialize rich table meta-data when writing the HDF5 file, in 

249 particular such data required to write and read back mixin columns like 

250 ``Time``, ``SkyCoord``, or ``Quantity`` to the file. 

251 **create_dataset_kwargs 

252 Additional keyword arguments are passed to 

253 ``h5py.File.create_dataset()`` or ``h5py.Group.create_dataset()``. 

254 """ 

255 from astropy.table import meta 

256 

257 try: 

258 import h5py 

259 except ImportError: 

260 raise Exception("h5py is required to read and write HDF5 files") 

261 

262 if path is None: 

263 # table is just an arbitrary, hardcoded string here. 

264 path = "__astropy_table__" 

265 elif path.endswith("/"): 

266 raise ValueError("table path should end with table name, not /") 

267 

268 if "/" in path: 

269 group, name = path.rsplit("/", 1) 

270 else: 

271 group, name = None, path 

272 

273 if isinstance(output, (h5py.File, h5py.Group)): 

274 if len(list(output.keys())) > 0 and name == "__astropy_table__": 

275 raise ValueError( 

276 "table path should always be set via the " 

277 "path= argument when writing to existing " 

278 "files" 

279 ) 

280 elif name == "__astropy_table__": 

281 warnings.warn( 

282 "table path was not set via the path= argument; " 

283 f"using default path {path}" 

284 ) 

285 

286 if group: 

287 try: 

288 output_group = output[group] 

289 except (KeyError, ValueError): 

290 output_group = output.create_group(group) 

291 else: 

292 output_group = output 

293 

294 elif isinstance(output, str): 

295 if os.path.exists(output) and not append: 

296 if overwrite and not append: 

297 os.remove(output) 

298 else: 

299 raise OSError(NOT_OVERWRITING_MSG.format(output)) 

300 

301 # Open the file for appending or writing 

302 f = h5py.File(output, "a" if append else "w") 

303 

304 # Recursively call the write function 

305 try: 

306 return write_table_hdf5( 

307 table, 

308 f, 

309 path=path, 

310 compression=compression, 

311 append=append, 

312 overwrite=overwrite, 

313 serialize_meta=serialize_meta, 

314 ) 

315 finally: 

316 f.close() 

317 

318 else: 

319 raise TypeError("output should be a string or an h5py File or Group object") 

320 

321 # Check whether table already exists 

322 if name in output_group: 

323 if append and overwrite: 

324 # Delete only the dataset itself 

325 del output_group[name] 

326 if serialize_meta and name + ".__table_column_meta__" in output_group: 

327 del output_group[name + ".__table_column_meta__"] 

328 else: 

329 raise OSError(f"Table {path} already exists") 

330 

331 # Encode any mixin columns as plain columns + appropriate metadata 

332 table = _encode_mixins(table) 

333 

334 # Table with numpy unicode strings can't be written in HDF5 so 

335 # to write such a table a copy of table is made containing columns as 

336 # bytestrings. Now this copy of the table can be written in HDF5. 

337 if any(col.info.dtype.kind == "U" for col in table.itercols()): 

338 table = table.copy(copy_data=False) 

339 table.convert_unicode_to_bytestring() 

340 

341 # Warn if information will be lost when serialize_meta=False. This is 

342 # hardcoded to the set difference between column info attributes and what 

343 # HDF5 can store natively (name, dtype) with no meta. 

344 if serialize_meta is False: 

345 for col in table.itercols(): 

346 for attr in ("unit", "format", "description", "meta"): 

347 if getattr(col.info, attr, None) not in (None, {}): 

348 warnings.warn( 

349 "table contains column(s) with defined 'unit', 'format'," 

350 " 'description', or 'meta' info attributes. These will" 

351 " be dropped since serialize_meta=False.", 

352 AstropyUserWarning, 

353 ) 

354 

355 # Write the table to the file 

356 if compression: 

357 if compression is True: 

358 compression = "gzip" 

359 dset = output_group.create_dataset( 

360 name, 

361 data=table.as_array(), 

362 compression=compression, 

363 **create_dataset_kwargs, 

364 ) 

365 else: 

366 dset = output_group.create_dataset( 

367 name, data=table.as_array(), **create_dataset_kwargs 

368 ) 

369 

370 if serialize_meta: 

371 header_yaml = meta.get_yaml_from_table(table) 

372 header_encoded = np.array([h.encode("utf-8") for h in header_yaml]) 

373 output_group.create_dataset(meta_path(name), data=header_encoded) 

374 

375 else: 

376 # Write the Table meta dict key:value pairs to the file as HDF5 

377 # attributes. This works only for a limited set of scalar data types 

378 # like numbers, strings, etc., but not any complex types. This path 

379 # also ignores column meta like unit or format. 

380 for key in table.meta: 

381 val = table.meta[key] 

382 try: 

383 dset.attrs[key] = val 

384 except TypeError: 

385 warnings.warn( 

386 f"Attribute `{key}` of type {type(val)} cannot be written to " 

387 "HDF5 files - skipping. (Consider specifying " 

388 "serialize_meta=True to write all meta data)", 

389 AstropyUserWarning, 

390 )
