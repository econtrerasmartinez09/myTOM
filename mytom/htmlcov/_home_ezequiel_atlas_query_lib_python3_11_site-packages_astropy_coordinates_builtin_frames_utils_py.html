40def get_polar_motion(time): 

41 """ 

42 gets the two polar motion components in radians for use with apio. 

43 """ 

44 # Get the polar motion from the IERS table 

45 iers_table = iers.earth_orientation_table.get() 

46 xp, yp, status = iers_table.pm_xy(time, return_status=True) 

47 

48 wmsg = ( 

49 "Tried to get polar motions for times {} IERS data is " 

50 "valid. Defaulting to polar motion from the 50-yr mean for those. " 

51 "This may affect precision at the arcsec level. Please check your " 

52 "astropy.utils.iers.conf.iers_auto_url and point it to a newer " 

53 "version if necessary." 

54 ) 

55 if np.any(status == iers.TIME_BEFORE_IERS_RANGE): 

56 xp[status == iers.TIME_BEFORE_IERS_RANGE] = _DEFAULT_PM[0] 

57 yp[status == iers.TIME_BEFORE_IERS_RANGE] = _DEFAULT_PM[1] 

58 

59 warnings.warn(wmsg.format("before"), AstropyWarning) 

60 

61 if np.any(status == iers.TIME_BEYOND_IERS_RANGE): 

62 xp[status == iers.TIME_BEYOND_IERS_RANGE] = _DEFAULT_PM[0] 

63 yp[status == iers.TIME_BEYOND_IERS_RANGE] = _DEFAULT_PM[1] 

64 

65 warnings.warn(wmsg.format("after"), AstropyWarning) 

66 

67 return xp.to_value(u.radian), yp.to_value(u.radian)

95def get_jd12(time, scale): 

96 """ 

97 Gets ``jd1`` and ``jd2`` from a time object in a particular scale. 

98 

99 Parameters

101 time : `~astropy.time.Time`

103 scale : str

109 jd2 : float

123def norm(p): 

124 """ 

125 Normalise a p-vector. 

126 """ 

127 return p / np.sqrt(np.einsum("...i,...i", p, p))[..., np.newaxis]

140def get_cip(jd1, jd2): 

141 """ 

142 Find the X, Y coordinates of the CIP and the CIO locator, s. 

143 

144 Parameters 

145 ---------- 

146 jd1 : float or `np.ndarray` 

147 First part of two part Julian date (TDB) 

148 jd2 : float or `np.ndarray` 

149 Second part of two part Julian date (TDB) 

150 

151 Returns 

152 ------- 

153 x : float or `np.ndarray` 

154 x coordinate of the CIP 

155 y : float or `np.ndarray` 

156 y coordinate of the CIP 

157 s : float or `np.ndarray` 

158 CIO locator, s 

159 """ 

160 # classical NPB matrix, IAU 2006/2000A 

161 rpnb = erfa.pnm06a(jd1, jd2) 

162 # CIP X, Y coordinates from array 

163 x, y = erfa.bpn2xy(rpnb) 

164 # CIO locator, s 

165 s = erfa.s06(jd1, jd2, x, y) 

166 return x, y, s

169def aticq(srepr, astrom): 

170 """ 

171 A slightly modified version of the ERFA function ``eraAticq``. 

172 

173 ``eraAticq`` performs the transformations between two coordinate systems, 

174 with the details of the transformation being encoded into the ``astrom`` array. 

175 

176 There are two issues with the version of aticq in ERFA. Both are associated 

177 with the handling of light deflection. 

178 

179 The companion function ``eraAtciqz`` is meant to be its inverse. However, this 

180 is not true for directions close to the Solar centre, since the light deflection 

181 calculations are numerically unstable and therefore not reversible. 

182 

183 This version sidesteps that problem by artificially reducing the light deflection 

184 for directions which are within 90 arcseconds of the Sun's position. This is the 

185 same approach used by the ERFA functions above, except that they use a threshold of 

186 9 arcseconds. 

187 

188 In addition, ERFA's aticq assumes a distant source, so there is no difference between 

189 the object-Sun vector and the observer-Sun vector. This can lead to errors of up to a 

190 few arcseconds in the worst case (e.g a Venus transit). 

191 

192 Parameters 

193 ---------- 

194 srepr : `~astropy.coordinates.SphericalRepresentation` 

195 Astrometric GCRS or CIRS position of object from observer 

196 astrom : eraASTROM array 

197 ERFA astrometry context, as produced by, e.g. ``eraApci13`` or ``eraApcs13`` 

198 

199 Returns 

200 ------- 

201 rc : float or `~numpy.ndarray` 

202 Right Ascension in radians 

203 dc : float or `~numpy.ndarray` 

204 Declination in radians 

205 """ 

206 # ignore parallax effects if no distance, or far away 

207 srepr_distance = srepr.distance 

208 ignore_distance = srepr_distance.unit == u.one 

209 

210 # RA, Dec to cartesian unit vectors 

211 pos = erfa.s2c(srepr.lon.radian, srepr.lat.radian) 

212 

213 # Bias-precession-nutation, giving GCRS proper direction. 

214 ppr = erfa.trxp(astrom["bpn"], pos) 

215 

216 # Aberration, giving GCRS natural direction 

217 d = np.zeros_like(ppr) 

218 for j in range(2): 

219 before = norm(ppr - d) 

220 after = erfa.ab(before, astrom["v"], astrom["em"], astrom["bm1"]) 

221 d = after - before 

222 pnat = norm(ppr - d) 

223 

224 # Light deflection by the Sun, giving BCRS coordinate direction 

225 d = np.zeros_like(pnat) 

226 for j in range(5): 

227 before = norm(pnat - d) 

228 if ignore_distance: 

229 # No distance to object, assume a long way away 

230 q = before 

231 else: 

232 # Find BCRS direction of Sun to object. 

233 # astrom['eh'] and astrom['em'] contain Sun to observer unit vector, 

234 # and distance, respectively. 

235 eh = astrom["em"][..., np.newaxis] * astrom["eh"] 

236 # unit vector from Sun to object 

237 q = eh + srepr_distance[..., np.newaxis].to_value(u.au) * before 

238 sundist, q = erfa.pn(q) 

239 sundist = sundist[..., np.newaxis] 

240 # calculation above is extremely unstable very close to the sun 

241 # in these situations, default back to ldsun-style behaviour, 

242 # since this is reversible and drops to zero within stellar limb 

243 q = np.where(sundist > 1.0e-10, q, before) 

244 

245 after = erfa.ld(1.0, before, q, astrom["eh"], astrom["em"], 1e-6) 

246 d = after - before 

247 pco = norm(pnat - d) 

248 

249 # ICRS astrometric RA, Dec 

250 rc, dc = erfa.c2s(pco) 

251 return erfa.anp(rc), dc

254def atciqz(srepr, astrom): 

255 """ 

256 A slightly modified version of the ERFA function ``eraAtciqz``. 

257 

258 ``eraAtciqz`` performs the transformations between two coordinate systems, 

259 with the details of the transformation being encoded into the ``astrom`` array. 

260 

261 There are two issues with the version of atciqz in ERFA. Both are associated 

262 with the handling of light deflection. 

263 

264 The companion function ``eraAticq`` is meant to be its inverse. However, this 

265 is not true for directions close to the Solar centre, since the light deflection 

266 calculations are numerically unstable and therefore not reversible. 

267 

268 This version sidesteps that problem by artificially reducing the light deflection 

269 for directions which are within 90 arcseconds of the Sun's position. This is the 

270 same approach used by the ERFA functions above, except that they use a threshold of 

271 9 arcseconds. 

272 

273 In addition, ERFA's atciqz assumes a distant source, so there is no difference between 

274 the object-Sun vector and the observer-Sun vector. This can lead to errors of up to a 

275 few arcseconds in the worst case (e.g a Venus transit). 

276 

277 Parameters 

278 ---------- 

279 srepr : `~astropy.coordinates.SphericalRepresentation` 

280 Astrometric ICRS position of object from observer 

281 astrom : eraASTROM array 

282 ERFA astrometry context, as produced by, e.g. ``eraApci13`` or ``eraApcs13`` 

283 

284 Returns 

285 ------- 

286 ri : float or `~numpy.ndarray` 

287 Right Ascension in radians 

288 di : float or `~numpy.ndarray` 

289 Declination in radians 

290 """ 

291 # ignore parallax effects if no distance, or far away 

292 srepr_distance = srepr.distance 

293 ignore_distance = srepr_distance.unit == u.one 

294 

295 # BCRS coordinate direction (unit vector). 

296 pco = erfa.s2c(srepr.lon.radian, srepr.lat.radian) 

297 

298 # Find BCRS direction of Sun to object 

299 if ignore_distance: 

300 # No distance to object, assume a long way away 

301 q = pco 

302 else: 

303 # Find BCRS direction of Sun to object. 

304 # astrom['eh'] and astrom['em'] contain Sun to observer unit vector, 

305 # and distance, respectively. 

306 eh = astrom["em"][..., np.newaxis] * astrom["eh"] 

307 # unit vector from Sun to object 

308 q = eh + srepr_distance[..., np.newaxis].to_value(u.au) * pco 

309 sundist, q = erfa.pn(q) 

310 sundist = sundist[..., np.newaxis] 

311 # calculation above is extremely unstable very close to the sun 

312 # in these situations, default back to ldsun-style behaviour, 

313 # since this is reversible and drops to zero within stellar limb 

314 q = np.where(sundist > 1.0e-10, q, pco) 

315 

316 # Light deflection by the Sun, giving BCRS natural direction. 

317 pnat = erfa.ld(1.0, pco, q, astrom["eh"], astrom["em"], 1e-6) 

318 

319 # Aberration, giving GCRS proper direction. 

320 ppr = erfa.ab(pnat, astrom["v"], astrom["em"], astrom["bm1"]) 

321 

322 # Bias-precession-nutation, giving CIRS proper direction. 

323 # Has no effect if matrix is identity matrix, in which case gives GCRS ppr. 

324 pi = erfa.rxp(astrom["bpn"], ppr) 

325 

326 # CIRS (GCRS) RA, Dec 

327 ri, di = erfa.c2s(pi) 

328 return erfa.anp(ri), di

331def prepare_earth_position_vel(time): 

332 """ 

333 Get barycentric position and velocity, and heliocentric position of Earth. 

334 

335 Parameters 

336 ---------- 

337 time : `~astropy.time.Time` 

338 time at which to calculate position and velocity of Earth 

339 

340 Returns 

341 ------- 

342 earth_pv : `np.ndarray` 

343 Barycentric position and velocity of Earth, in au and au/day 

344 earth_helio : `np.ndarray` 

345 Heliocentric position of Earth in au 

346 """ 

347 # this goes here to avoid circular import errors 

348 from astropy.coordinates.solar_system import ( 

349 get_body_barycentric, 

350 get_body_barycentric_posvel, 

351 solar_system_ephemeris, 

352 ) 

353 

354 # get barycentric position and velocity of earth 

355 

356 ephemeris = solar_system_ephemeris.get() 

357 

358 # if we are using the builtin erfa based ephemeris, 

359 # we can use the fact that epv00 already provides all we need. 

360 # This avoids calling epv00 twice, once 

361 # in get_body_barycentric_posvel('earth') and once in 

362 # get_body_barycentric('sun') 

363 if ephemeris == "builtin": 

364 jd1, jd2 = get_jd12(time, "tdb") 

365 earth_pv_heliocentric, earth_pv = erfa.epv00(jd1, jd2) 

366 earth_heliocentric = earth_pv_heliocentric["p"] 

367 

368 # all other ephemeris providers probably don't have a shortcut like this 

369 else: 

370 earth_p, earth_v = get_body_barycentric_posvel("earth", time) 

371 

372 # get heliocentric position of earth, preparing it for passing to erfa. 

373 sun = get_body_barycentric("sun", time) 

374 earth_heliocentric = (earth_p - sun).get_xyz(xyz_axis=-1).to_value(u.au) 

375 

376 # Also prepare earth_pv for passing to erfa, which wants it as 

377 # a structured dtype. 

378 earth_pv = pav2pv( 

379 earth_p.get_xyz(xyz_axis=-1).to_value(u.au), 

380 earth_v.get_xyz(xyz_axis=-1).to_value(u.au / u.d), 

381 ) 

382 

383 return earth_pv, earth_heliocentric

386def get_offset_sun_from_barycenter(time, include_velocity=False, reverse=False): 

387 """ 

388 Returns the offset of the Sun center from the solar-system barycenter (SSB). 

389 

390 Parameters 

391 ---------- 

392 time : `~astropy.time.Time` 

393 Time at which to calculate the offset 

394 include_velocity : `bool` 

395 If ``True``, attach the velocity as a differential. Defaults to ``False``. 

396 reverse : `bool` 

397 If ``True``, return the offset of the barycenter from the Sun. Defaults to ``False``. 

398 

399 Returns 

400 ------- 

401 `~astropy.coordinates.CartesianRepresentation` 

402 The offset 

403 """ 

404 if include_velocity: 

405 # Import here to avoid a circular import 

406 from astropy.coordinates.solar_system import get_body_barycentric_posvel 

407 

408 offset_pos, offset_vel = get_body_barycentric_posvel("sun", time) 

409 if reverse: 

410 offset_pos, offset_vel = -offset_pos, -offset_vel 

411 offset_vel = offset_vel.represent_as(CartesianDifferential) 

412 offset_pos = offset_pos.with_differentials(offset_vel) 

413 

414 else: 

415 # Import here to avoid a circular import 

416 from astropy.coordinates.solar_system import get_body_barycentric 

417 

418 offset_pos = get_body_barycentric("sun", time) 

419 if reverse: 

420 offset_pos = -offset_pos 

421 

422 return offset_pos
