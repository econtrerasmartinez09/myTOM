34 @abc.abstractmethod 

35 def tag_mapping(self): 

36 """ 

37 A list of 2-tuples or callables mapping YAML tag prefixes to JSON Schema 

38 URL prefixes. 

39 

40 For each entry: 

41 

42 - If a 2-tuple, the first part of the tuple is a YAML tag 

43 prefix to match. The second part is a string, where case 

44 the following are available as Python formatting tokens: 

45 

46 - ``{tag}``: the complete YAML tag. 

47 - ``{tag_suffix}``: the part of the YAML tag after the 

48 matched prefix. 

49 - ``{tag_prefix}``: the matched YAML tag prefix. 

50 

51 - If a callable, it is passed the entire YAML tag must return 

52 the entire JSON schema URL if it matches, otherwise, return `None`. 

53 

54 Note that while JSON Schema URLs uniquely define a JSON 

55 Schema, they do not have to actually exist on an HTTP server 

56 and be fetchable (much like XML namespaces). 

57 

58 For example, to match all YAML tags with the 

59 ``tag:nowhere.org:custom` prefix to the 

60 ``http://nowhere.org/schemas/custom/`` URL prefix:: 

61 

62 return [('tag:nowhere.org:custom/', 

63 'http://nowhere.org/schemas/custom/{tag_suffix}')] 

64 """ 

65 

66 @property 

67 @abc.abstractmethod 

68 def url_mapping(self): 

69 """ 

70 Schema content can be provided using the resource Mapping API. 

71 

72 A list of 2-tuples or callables mapping JSON Schema URLs to 

73 other URLs. This is useful if the JSON Schemas are not 

74 actually fetchable at their corresponding URLs but are on the 

75 local filesystem, or, to save bandwidth, we have a copy of 

76 fetchable schemas on the local filesystem. If neither is 

77 desirable, it may simply be the empty list. 

78 

79 For each entry: 

80 

81 - If a 2-tuple, the first part is a URL prefix to match. The 

82 second part is a string, where the following are available 

83 as Python formatting tokens: 

84 

85 - ``{url}``: The entire JSON schema URL 

86 - ``{url_prefix}``: The matched URL prefix 

87 - ``{url_suffix}``: The part of the URL after the prefix. 

88 

89 - If a callable, it is passed the entire JSON Schema URL and 

90 must return a resolvable URL pointing to the schema content. 

91 If it doesn't match, should return `None`. 

92 

93 For example, to map a remote HTTP URL prefix to files installed 

94 alongside as data alongside Python module:: 

95 

96 return [('http://nowhere.org/schemas/custom/1.0.0/', 

97 asdf.util.filepath_to_url( 

98 os.path.join(SCHEMA_PATH, 'stsci.edu')) + 

99 '/{url_suffix}.yaml' 

100 )] 

101 """

109 def __init__(self, extensions): 

110 from ._extension import ExtensionProxy 

111 

112 extensions = [ExtensionProxy.maybe_wrap(e) for e in extensions] 

113 

114 tag_mapping = [] 

115 url_mapping = [] 

116 validators = {} 

117 self._type_index = AsdfTypeIndex() 

118 for extension in extensions: 

119 tag_mapping.extend(extension.tag_mapping) 

120 url_mapping.extend(extension.url_mapping) 

121 for typ in extension.types: 

122 self._type_index.add_type(typ, extension) 

123 validators.update(typ.validators) 

124 for sibling in typ.versioned_siblings: 

125 self._type_index.add_type(sibling, extension) 

126 validators.update(sibling.validators) 

127 self._extensions = extensions 

128 self._tag_mapping = resolver.Resolver(tag_mapping, "tag") 

129 self._url_mapping = resolver.Resolver(url_mapping, "url") 

130 self._resolver = resolver.ResolverChain(self._tag_mapping, self._url_mapping) 

131 self._validators = validators

167def get_cached_asdf_extension_list(extensions): 

168 """ 

169 Get a previously created AsdfExtensionList for the specified 

170 extensions, or create and cache one if necessary. Building 

171 the type index is expensive, so it helps performance to reuse 

172 the index when possible. 

173 

174 Parameters 

175 ---------- 

176 extensions : list of asdf.extension.AsdfExtension 

177 

178 Returns 

179 ------- 

180 asdf.extension.AsdfExtensionList 

181 """ 

182 from ._extension import ExtensionProxy 

183 

184 # The tuple makes the extensions hashable so that we 

185 # can pass them to the lru_cache method. The ExtensionProxy 

186 # overrides __hash__ to return the hashed object id of the wrapped 

187 # extension, so this will method will only return the same 

188 # AsdfExtensionList if the list contains identical extension 

189 # instances in identical order. 

190 extensions = tuple(ExtensionProxy.maybe_wrap(e) for e in extensions) 

191 

192 return _get_cached_asdf_extension_list(extensions)
