34def calculate_gravatar_hash(email): 

35 # Calculate the email hash 

36 enc_email = email.strip().lower().encode("utf-8") 

37 email_hash = hashlib.md5(enc_email).hexdigest() 

38 return email_hash

41def get_gravatar_url(email, size=GRAVATAR_DEFAULT_SIZE, default=GRAVATAR_DEFAULT_IMAGE, 

42 rating=GRAVATAR_DEFAULT_RATING, secure=GRAVATAR_DEFAULT_SECURE): 

43 """ 

44 Builds a url to a gravatar from an email address. 

45 

46 :param email: The email to fetch the gravatar for 

47 :param size: The size (in pixels) of the gravatar to fetch 

48 :param default: What type of default image to use if the gravatar does not exist 

49 :param rating: Used to filter the allowed gravatar ratings 

50 :param secure: If True use https, otherwise plain http 

51 """ 

52 if secure: 

53 url_base = GRAVATAR_SECURE_URL 

54 else: 

55 url_base = GRAVATAR_URL 

56 

57 # Calculate the email hash 

58 email_hash = calculate_gravatar_hash(email) 

59 

60 # Build querystring 

61 query_string = urlencode({ 

62 's': str(size), 

63 'd': default, 

64 'r': rating, 

65 }) 

66 

67 # Build url 

68 url = '{base}avatar/{hash}.jpg?{qs}'.format(base=url_base, 

69 hash=email_hash, qs=query_string) 

70 

71 return url

74def has_gravatar(email): 

75 """ 

76 Returns True if the user has a gravatar, False if otherwise 

77 """ 

78 # Request a 404 response if the gravatar does not exist 

79 url = get_gravatar_url(email, default=GRAVATAR_DEFAULT_IMAGE_404) 

80 

81 # Verify an OK response was received 

82 try: 

83 request = Request(url) 

84 request.get_method = lambda: 'HEAD' 

85 return 200 == urlopen(request).code 

86 except (HTTPError, URLError): 

87 return False
