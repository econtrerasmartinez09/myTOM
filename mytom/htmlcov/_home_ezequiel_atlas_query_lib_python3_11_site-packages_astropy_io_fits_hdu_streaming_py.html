15class StreamingHDU: 

16 """ 

17 A class that provides the capability to stream data to a FITS file 

18 instead of requiring data to all be written at once. 

19 

20 The following pseudocode illustrates its use:: 

21 

22 header = astropy.io.fits.Header() 

23 

24 for all the cards you need in the header: 

25 header[key] = (value, comment) 

26 

27 shdu = astropy.io.fits.StreamingHDU('filename.fits', header) 

28 

29 for each piece of data: 

30 shdu.write(data) 

31 

32 shdu.close() 

33 """ 

34 

35 def __init__(self, name, header): 

36 """ 

37 Construct a `StreamingHDU` object given a file name and a header. 

38 

39 Parameters 

40 ---------- 

41 name : path-like or file-like 

42 The file to which the header and data will be streamed. If opened, 

43 the file object must be opened in a writeable binary mode such as 

44 'wb' or 'ab+'. 

45 

46 header : `Header` instance 

47 The header object associated with the data to be written 

48 to the file. 

49 

50 Notes 

51 ----- 

52 The file will be opened and the header appended to the end of 

53 the file. If the file does not already exist, it will be 

54 created, and if the header represents a Primary header, it 

55 will be written to the beginning of the file. If the file 

56 does not exist and the provided header is not a Primary 

57 header, a default Primary HDU will be inserted at the 

58 beginning of the file and the provided header will be added as 

59 the first extension. If the file does already exist, but the 

60 provided header represents a Primary header, the header will 

61 be modified to an image extension header and appended to the 

62 end of the file. 

63 """ 

64 if isinstance(name, gzip.GzipFile): 

65 raise TypeError("StreamingHDU not supported for GzipFile objects.") 

66 

67 self._header = header.copy() 

68 

69 # handle a file object instead of a file name 

70 filename = fileobj_name(name) or "" 

71 

72 filename = os.path.expanduser(filename) 

73 

74 # Check if the file already exists. If it does not, check to see 

75 # if we were provided with a Primary Header. If not we will need 

76 # to prepend a default PrimaryHDU to the file before writing the 

77 # given header. 

78 

79 newfile = False 

80 

81 if filename: 

82 if not os.path.exists(filename) or os.path.getsize(filename) == 0: 

83 newfile = True 

84 elif hasattr(name, "len") and name.len == 0: 

85 newfile = True 

86 

87 if newfile: 

88 if "SIMPLE" not in self._header: 

89 hdulist = HDUList([PrimaryHDU()]) 

90 hdulist.writeto(name, "exception") 

91 else: 

92 # This will not be the first extension in the file so we 

93 # must change the Primary header provided into an image 

94 # extension header. 

95 

96 if "SIMPLE" in self._header: 

97 self._header.set("XTENSION", "IMAGE", "Image extension", after="SIMPLE") 

98 del self._header["SIMPLE"] 

99 

100 if "PCOUNT" not in self._header: 

101 dim = self._header["NAXIS"] 

102 

103 if dim == 0: 

104 dim = "" 

105 else: 

106 dim = str(dim) 

107 

108 self._header.set( 

109 "PCOUNT", 0, "number of parameters", after="NAXIS" + dim 

110 ) 

111 

112 if "GCOUNT" not in self._header: 

113 self._header.set("GCOUNT", 1, "number of groups", after="PCOUNT") 

114 

115 self._ffo = _File(name, "append") 

116 

117 # TODO : Fix this once the HDU writing API is cleaned up 

118 tmp_hdu = _BaseHDU() 

119 # Passing self._header as an argument to _BaseHDU() will cause its 

120 # values to be modified in undesired ways...need to have a better way 

121 # of doing this 

122 tmp_hdu._header = self._header 

123 self._header_offset = tmp_hdu._writeheader(self._ffo)[0] 

124 self._data_offset = self._ffo.tell() 

125 self._size = self.size 

126 

127 if self._size != 0: 

128 self.writecomplete = False 

129 else: 

130 self.writecomplete = True 

131

134 return self

137 self.close()

139 def write(self, data): 

140 """ 

141 Write the given data to the stream. 

142 

143 Parameters 

144 ---------- 

145 data : ndarray 

146 Data to stream to the file. 

147 

148 Returns 

149 ------- 

150 writecomplete : int 

151 Flag that when `True` indicates that all of the required 

152 data has been written to the stream. 

153 

154 Notes 

155 ----- 

156 Only the amount of data specified in the header provided to the class 

157 constructor may be written to the stream. If the provided data would 

158 cause the stream to overflow, an `OSError` exception is 

159 raised and the data is not written. Once sufficient data has been 

160 written to the stream to satisfy the amount specified in the header, 

161 the stream is padded to fill a complete FITS block and no more data 

162 will be accepted. An attempt to write more data after the stream has 

163 been filled will raise an `OSError` exception. If the 

164 dtype of the input data does not match what is expected by the header, 

165 a `TypeError` exception is raised. 

166 """ 

167 size = self._ffo.tell() - self._data_offset 

168 

169 if self.writecomplete or size + data.nbytes > self._size: 

170 raise OSError( 

171 "Attempt to write more data to the stream than the header specified." 

172 ) 

173 

174 if BITPIX2DTYPE[self._header["BITPIX"]] != data.dtype.name: 

175 raise TypeError( 

176 "Supplied data does not match the type specified in the header." 

177 ) 

178 

179 if data.dtype.str[0] != ">": 

180 # byteswap little endian arrays before writing 

181 output = data.byteswap() 

182 else: 

183 output = data 

184 

185 self._ffo.writearray(output) 

186 

187 if self._ffo.tell() - self._data_offset == self._size: 

188 # the stream is full so pad the data to the next FITS block 

189 self._ffo.write(_pad_length(self._size) * "\0") 

190 self.writecomplete = True 

191 

192 self._ffo.flush() 

193 

194 return self.writecomplete

197 def size(self): 

198 """ 

199 Return the size (in bytes) of the data portion of the HDU. 

200 """ 

201 size = 0 

202 naxis = self._header.get("NAXIS", 0) 

203 

204 if naxis > 0: 

205 simple = self._header.get("SIMPLE", "F") 

206 random_groups = self._header.get("GROUPS", "F") 

207 

208 if simple == "T" and random_groups == "T": 

209 groups = 1 

210 else: 

211 groups = 0 

212 

213 size = 1 

214 

215 for idx in range(groups, naxis): 

216 size = size * self._header["NAXIS" + str(idx + 1)] 

217 bitpix = self._header["BITPIX"] 

218 gcount = self._header.get("GCOUNT", 1) 

219 pcount = self._header.get("PCOUNT", 0) 

220 size = abs(bitpix) * gcount * (pcount + size) // 8 

221 return size
