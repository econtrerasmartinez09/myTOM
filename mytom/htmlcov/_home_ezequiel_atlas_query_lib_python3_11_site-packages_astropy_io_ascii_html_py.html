44def identify_table(soup, htmldict, numtable): 

45 """ 

46 Checks whether the given BeautifulSoup tag is the table 

47 the user intends to process. 

48 """ 

49 if soup is None or soup.name != "table": 

50 return False # Tag is not a <table> 

51 

52 elif "table_id" not in htmldict: 

53 return numtable == 1 

54 table_id = htmldict["table_id"] 

55 

56 if isinstance(table_id, str): 

57 return "id" in soup.attrs and soup["id"] == table_id 

58 elif isinstance(table_id, int): 

59 return table_id == numtable 

60 

61 # Return False if an invalid parameter is given 

62 return False

73 def process_lines(self, lines): 

74 """ 

75 Convert the given input into a list of SoupString rows 

76 for further processing. 

77 """ 

78 try: 

79 from bs4 import BeautifulSoup 

80 except ImportError: 

81 raise core.OptionalTableImportError( 

82 "BeautifulSoup must be installed to read HTML tables" 

83 ) 

84 

85 if "parser" not in self.html: 

86 with warnings.catch_warnings(): 

87 # Ignore bs4 parser warning #4550. 

88 warnings.filterwarnings( 

89 "ignore", ".*no parser was explicitly specified.*" 

90 ) 

91 soup = BeautifulSoup("\n".join(lines)) 

92 else: # use a custom backend parser 

93 soup = BeautifulSoup("\n".join(lines), self.html["parser"]) 

94 tables = soup.find_all("table") 

95 for i, possible_table in enumerate(tables): 

96 if identify_table(possible_table, self.html, i + 1): 

97 table = possible_table # Find the correct table 

98 break 

99 else: 

100 if isinstance(self.html["table_id"], int): 

101 err_descr = f"number {self.html['table_id']}" 

102 else: 

103 err_descr = f"id '{self.html['table_id']}'" 

104 raise core.InconsistentTableError( 

105 f"ERROR: HTML table {err_descr} not found" 

106 ) 

107 

108 # Get all table rows 

109 soup_list = [SoupString(x) for x in table.find_all("tr")] 

110 

111 return soup_list

114class HTMLSplitter(core.BaseSplitter): 

115 """ 

116 Split HTML table data. 

117 """ 

118 

119 def __call__(self, lines): 

120 """ 

121 Return HTML data from lines as a generator. 

122 """ 

123 for line in lines: 

124 if not isinstance(line, SoupString): 

125 raise TypeError("HTML lines should be of type SoupString") 

126 soup = line.soup 

127 header_elements = soup.find_all("th") 

128 if header_elements: 

129 # Return multicolumns as tuples for HTMLHeader handling 

130 yield [ 

131 (el.text.strip(), el["colspan"]) 

132 if el.has_attr("colspan") 

133 else el.text.strip() 

134 for el in header_elements 

135 ] 

136 data_elements = soup.find_all("td") 

137 if data_elements: 

138 yield [el.text.strip() for el in data_elements] 

139 if len(lines) == 0: 

140 raise core.InconsistentTableError( 

141 "HTML tables must contain data in a <table> tag" 

142 )

145class HTMLOutputter(core.TableOutputter): 

146 """ 

147 Output the HTML data as an ``astropy.table.Table`` object. 

148 

149 This subclass allows for the final table to contain 

150 multidimensional columns (defined using the colspan attribute 

151 of <th>). 

152 """ 

153 

154 default_converters = [ 

155 core.convert_numpy(int), 

156 core.convert_numpy(float), 

157 core.convert_numpy(str), 

158 ] 

159 

160 def __call__(self, cols, meta): 

161 """ 

162 Process the data in multidimensional columns. 

163 """ 

164 new_cols = [] 

165 col_num = 0 

166 

167 while col_num < len(cols): 

168 col = cols[col_num] 

169 if hasattr(col, "colspan"): 

170 # Join elements of spanned columns together into list of tuples 

171 span_cols = cols[col_num : col_num + col.colspan] 

172 new_col = core.Column(col.name) 

173 new_col.str_vals = list(zip(*[x.str_vals for x in span_cols])) 

174 new_cols.append(new_col) 

175 col_num += col.colspan 

176 else: 

177 new_cols.append(col) 

178 col_num += 1 

179 

180 return super().__call__(new_cols, meta)

262class HTML(core.BaseReader): 

263 """HTML format table. 

264 

265 In order to customize input and output, a dict of parameters may 

266 be passed to this class holding specific customizations. 

267 

268 **htmldict** : Dictionary of parameters for HTML input/output. 

269 

270 * css : Customized styling 

271 If present, this parameter will be included in a <style> 

272 tag and will define stylistic attributes of the output. 

273 

274 * table_id : ID for the input table 

275 If a string, this defines the HTML id of the table to be processed. 

276 If an integer, this specifies the index of the input table in the 

277 available tables. Unless this parameter is given, the reader will 

278 use the first table found in the input file. 

279 

280 * multicol : Use multi-dimensional columns for output 

281 The writer will output tuples as elements of multi-dimensional 

282 columns if this parameter is true, and if not then it will 

283 use the syntax 1.36583e-13 .. 1.36583e-13 for output. If not 

284 present, this parameter will be true by default. 

285 

286 * raw_html_cols : column name or list of names with raw HTML content 

287 This allows one to include raw HTML content in the column output, 

288 for instance to include link references in a table. This option 

289 requires that the bleach package be installed. Only whitelisted 

290 tags are allowed through for security reasons (see the 

291 raw_html_clean_kwargs arg). 

292 

293 * raw_html_clean_kwargs : dict of keyword args controlling HTML cleaning 

294 Raw HTML will be cleaned to prevent unsafe HTML from ending up in 

295 the table output. This is done by calling ``bleach.clean(data, 

296 **raw_html_clean_kwargs)``. For details on the available options 

297 (e.g. tag whitelist) see: 

298 https://bleach.readthedocs.io/en/latest/clean.html 

299 

300 * parser : Specific HTML parsing library to use 

301 If specified, this specifies which HTML parsing library 

302 BeautifulSoup should use as a backend. The options to choose 

303 from are 'html.parser' (the standard library parser), 'lxml' 

304 (the recommended parser), 'xml' (lxml's XML parser), and 

305 'html5lib'. html5lib is a highly lenient parser and therefore 

306 might work correctly for unusual input if a different parser 

307 fails. 

308 

309 * jsfiles : list of js files to include when writing table. 

310 

311 * cssfiles : list of css files to include when writing table. 

312 

313 * js : js script to include in the body when writing table. 

314 

315 * table_class : css class for the table 

316 

317 """
