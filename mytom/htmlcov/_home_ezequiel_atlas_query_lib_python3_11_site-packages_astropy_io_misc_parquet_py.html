26def parquet_identify(origin, filepath, fileobj, *args, **kwargs): 

27 """Checks if input is in the Parquet format. 

28 

29 Parameters 

30 ---------- 

31 origin : Any 

32 filepath : str or None 

33 fileobj : `~pyarrow.NativeFile` or None 

34 *args, **kwargs 

35 

36 Returns 

37 ------- 

38 is_parquet : bool 

39 True if 'fileobj' is not None and is a pyarrow file, or if 

40 'filepath' is a string ending with '.parquet' or '.parq'. 

41 False otherwise. 

42 """ 

43 if fileobj is not None: 

44 try: # safely test if pyarrow file 

45 pos = fileobj.tell() # store current stream position 

46 except AttributeError: 

47 return False 

48 

49 signature = fileobj.read(4) # read first 4 bytes 

50 fileobj.seek(pos) # return to original location 

51 

52 return signature == PARQUET_SIGNATURE 

53 elif filepath is not None: 

54 return filepath.endswith((".parquet", ".parq")) 

55 else: 

56 return False

59def read_table_parquet( 

60 input, include_names=None, exclude_names=None, schema_only=False, filters=None 

61): 

62 """ 

63 Read a Table object from a Parquet file. 

64 

65 This requires `pyarrow <https://arrow.apache.org/docs/python/>`_ 

66 to be installed. 

67 

68 The ``filters`` parameter consists of predicates that are expressed 

69 in disjunctive normal form (DNF), like ``[[('x', '=', 0), ...], ...]``. 

70 DNF allows arbitrary boolean logical combinations of single column 

71 predicates. The innermost tuples each describe a single column predicate. 

72 The list of inner predicates is interpreted as a conjunction (AND), 

73 forming a more selective and multiple column predicate. Finally, the most 

74 outer list combines these filters as a disjunction (OR). 

75 

76 Predicates may also be passed as List[Tuple]. This form is interpreted 

77 as a single conjunction. To express OR in predicates, one must 

78 use the (preferred) List[List[Tuple]] notation. 

79 

80 Each tuple has format: (``key``, ``op``, ``value``) and compares the 

81 ``key`` with the ``value``. 

82 The supported ``op`` are: ``=`` or ``==``, ``!=``, ``<``, ``>``, ``<=``, 

83 ``>=``, ``in`` and ``not in``. If the ``op`` is ``in`` or ``not in``, the 

84 ``value`` must be a collection such as a ``list``, a ``set`` or a 

85 ``tuple``. 

86 

87 For example: 

88 

89 .. code-block:: python 

90 

91 ('x', '=', 0) 

92 ('y', 'in', ['a', 'b', 'c']) 

93 ('z', 'not in', {'a','b'}) 

94 

95 Parameters 

96 ---------- 

97 input : str or path-like or file-like object 

98 If a string or path-like object, the filename to read the table from. 

99 If a file-like object, the stream to read data. 

100 include_names : list [str], optional 

101 List of names to include in output. If not supplied, then 

102 include all columns. 

103 exclude_names : list [str], optional 

104 List of names to exclude from output (applied after ``include_names``). 

105 If not supplied then no columns are excluded. 

106 schema_only : bool, optional 

107 Only read the schema/metadata with table information. 

108 filters : list [tuple] or list [list [tuple] ] or None, optional 

109 Rows which do not match the filter predicate will be removed from 

110 scanned data. See `pyarrow.parquet.read_table()` for details. 

111 

112 Returns 

113 ------- 

114 table : `~astropy.table.Table` 

115 Table will have zero rows and only metadata information 

116 if schema_only is True. 

117 """ 

118 pa, parquet, _ = get_pyarrow() 

119 

120 if not isinstance(input, (str, os.PathLike)): 

121 # The 'read' attribute is the key component of a generic 

122 # file-like object. 

123 if not hasattr(input, "read"): 

124 raise TypeError("pyarrow can only open path-like or file-like objects.") 

125 

126 schema = parquet.read_schema(input) 

127 

128 # Pyarrow stores all metadata as byte-strings, so we convert 

129 # to UTF-8 strings here. 

130 if schema.metadata is not None: 

131 md = {k.decode("UTF-8"): v.decode("UTF-8") for k, v in schema.metadata.items()} 

132 else: 

133 md = {} 

134 

135 from astropy.table import Column, Table, meta, serialize 

136 

137 # parse metadata from table yaml 

138 meta_dict = {} 

139 if "table_meta_yaml" in md: 

140 meta_yaml = md.pop("table_meta_yaml").split("\n") 

141 meta_hdr = meta.get_header_from_yaml(meta_yaml) 

142 if "meta" in meta_hdr: 

143 meta_dict = meta_hdr["meta"] 

144 else: 

145 meta_hdr = None 

146 

147 # parse and set serialized columns 

148 full_table_columns = {name: name for name in schema.names} 

149 has_serialized_columns = False 

150 if "__serialized_columns__" in meta_dict: 

151 has_serialized_columns = True 

152 serialized_columns = meta_dict["__serialized_columns__"] 

153 for scol in serialized_columns: 

154 for name in _get_names(serialized_columns[scol]): 

155 full_table_columns[name] = scol 

156 

157 use_names = set(full_table_columns.values()) 

158 # Apply include_names before exclude_names 

159 if include_names is not None: 

160 use_names.intersection_update(include_names) 

161 if exclude_names is not None: 

162 use_names.difference_update(exclude_names) 

163 # Preserve column ordering via list, and use this dict trick 

164 # to remove duplicates and preserve ordering (for mixin columns) 

165 use_names = list( 

166 dict.fromkeys([x for x in full_table_columns.values() if x in use_names]) 

167 ) 

168 

169 # names_to_read is a list of actual serialized column names, where 

170 # e.g. the requested name 'time' becomes ['time.jd1', 'time.jd2'] 

171 names_to_read = [] 

172 for name in use_names: 

173 names = [n for n, col in full_table_columns.items() if name == col] 

174 names_to_read.extend(names) 

175 

176 if not names_to_read: 

177 raise ValueError("No include_names specified were found in the table.") 

178 

179 # We need to pop any unread serialized columns out of the meta_dict. 

180 if has_serialized_columns: 

181 for scol in list(meta_dict["__serialized_columns__"].keys()): 

182 if scol not in use_names: 

183 meta_dict["__serialized_columns__"].pop(scol) 

184 

185 # whether to return the whole table or a formatted empty table. 

186 if not schema_only: 

187 # Read the pyarrow table, specifying columns and filters. 

188 pa_table = parquet.read_table(input, columns=names_to_read, filters=filters) 

189 num_rows = pa_table.num_rows 

190 else: 

191 num_rows = 0 

192 

193 # Determine numpy/astropy types of columns from the arrow table. 

194 dtype = [] 

195 for name in names_to_read: 

196 t = schema.field(name).type 

197 

198 shape = None 

199 

200 if isinstance(t, pa.FixedSizeListType): 

201 # The FixedSizeListType has an arrow value_type and a size. 

202 value_type = t.value_type 

203 shape = (t.list_size,) 

204 elif isinstance(t, pa.ListType): 

205 # The ListType (variable length arrays) has a value type. 

206 value_type = t.value_type 

207 else: 

208 # All other arrow column types are the value_type. 

209 value_type = t 

210 

211 if value_type not in (pa.string(), pa.binary()): 

212 # Convert the pyarrow value type into a numpy dtype (which is returned 

213 # by the to_pandas_type() method). 

214 # If this is an array column, the numpy dtype needs the shape as well. 

215 if shape is None: 

216 dtype.append(value_type.to_pandas_dtype()) 

217 else: 

218 dtype.append((value_type.to_pandas_dtype(), shape)) 

219 continue 

220 

221 # Special-case for string and binary columns 

222 md_name = f"table::len::{name}" 

223 if md_name in md: 

224 # String/bytes length from header. 

225 strlen = int(md[md_name]) 

226 elif schema_only: # Find the maximum string length. 

227 # Choose an arbitrary string length since 

228 # are not reading in the table. 

229 strlen = 10 

230 warnings.warn( 

231 f"No {md_name} found in metadata. Guessing {{strlen}} for schema.", 

232 AstropyUserWarning, 

233 ) 

234 else: 

235 strlen = max(len(row.as_py()) for row in pa_table[name]) 

236 warnings.warn( 

237 f"No {md_name} found in metadata. Using longest string" 

238 f" ({strlen} characters).", 

239 AstropyUserWarning, 

240 ) 

241 strname = f"U{strlen}" if value_type == pa.string() else f"|S{strlen}" 

242 

243 # If this is an array column, the numpy dtype needs the shape as well. 

244 if shape is None: 

245 dtype.append(strname) 

246 else: 

247 dtype.append((strname, shape)) 

248 

249 if schema_only: 

250 # If we only need the schema, create an empty table with the correct dtype. 

251 data = np.zeros(0, dtype=list(zip(names_to_read, dtype))) 

252 table = Table(data=data, meta=meta_dict) 

253 else: 

254 # If we need the full table, create the table and add the columns 

255 # one at a time. This minimizes data copying. 

256 

257 table = Table(meta=meta_dict) 

258 for name, dt in zip(names_to_read, dtype): 

259 # First convert the arrow column to a numpy array. 

260 col = pa_table[name].to_numpy() 

261 

262 t = schema.field(name).type 

263 if t in (pa.string(), pa.binary()): 

264 # If it is a string/binary type, coerce it to the correct type. 

265 col = col.astype(dt) 

266 elif isinstance(t, pa.FixedSizeListType): 

267 # If it is a FixedSizeListType (array column) then it needs to 

268 # be broken into a 2D array, but only if the table has a non-zero 

269 # length. 

270 if len(col) > 0: 

271 col = np.stack(col) 

272 

273 if t.value_type in (pa.string(), pa.binary()): 

274 # If it is a string/binary type, coerce it to the 

275 # correct type. 

276 # The conversion dtype is only the first element 

277 # in the dtype tuple. 

278 col = col.astype(dt[0]) 

279 else: 

280 # This is an empty column, and needs to be created with the 

281 # correct type. 

282 col = np.zeros(0, dtype=dt) 

283 elif isinstance(t, pa.ListType): 

284 # If we have a variable length string/binary column, 

285 # we need to convert each row to the proper type. 

286 if t.value_type in (pa.string(), pa.binary()): 

287 col = np.array([row.astype(dt) for row in col], dtype=np.object_) 

288 

289 table.add_column(Column(name=name, data=col)) 

290 

291 if meta_hdr is not None: 

292 # Set description, format, unit, meta from the column 

293 # metadata that was serialized with the table. 

294 header_cols = {x["name"]: x for x in meta_hdr["datatype"]} 

295 for col in table.columns.values(): 

296 for attr in ("description", "format", "unit", "meta"): 

297 if attr in header_cols[col.name]: 

298 setattr(col, attr, header_cols[col.name][attr]) 

299 

300 # Convert all compound columns to astropy objects 

301 # (e.g. time.jd1, time.jd2 into a single time column) 

302 table = serialize._construct_mixins_from_columns(table) 

303 

304 return table

307def write_table_parquet(table, output, overwrite=False): 

308 """ 

309 Write a Table object to a Parquet file. 

310 

311 The parquet writer supports tables with regular columns, fixed-size array 

312 columns, and variable-length array columns (provided all arrays have the 

313 same type). 

314 

315 This requires `pyarrow <https://arrow.apache.org/docs/python/>`_ 

316 to be installed. 

317 

318 Parameters 

319 ---------- 

320 table : `~astropy.table.Table` 

321 Data table that is to be written to file. 

322 output : str or path-like 

323 The filename to write the table to. 

324 overwrite : bool, optional 

325 Whether to overwrite any existing file without warning. Default `False`. 

326 

327 Notes 

328 ----- 

329 Tables written with array columns (fixed-size or variable-length) cannot 

330 be read with pandas. 

331 

332 Raises 

333 ------ 

334 ValueError 

335 If one of the columns has a mixed-type variable-length array, or 

336 if it is a zero-length table and any of the columns are variable-length 

337 arrays. 

338 """ 

339 from astropy.table import meta, serialize 

340 from astropy.utils.data_info import serialize_context_as 

341 

342 pa, parquet, writer_version = get_pyarrow() 

343 

344 if not isinstance(output, (str, os.PathLike)): 

345 raise TypeError(f"`output` should be a string or path-like, not {output}") 

346 

347 # Convert all compound columns into serialized column names, where 

348 # e.g. 'time' becomes ['time.jd1', 'time.jd2']. 

349 with serialize_context_as("parquet"): 

350 encode_table = serialize.represent_mixins_as_columns(table) 

351 # We store the encoded serialization metadata as a yaml string. 

352 meta_yaml = meta.get_yaml_from_table(encode_table) 

353 meta_yaml_str = "\n".join(meta_yaml) 

354 

355 # Build the pyarrow schema by converting from the numpy dtype of each 

356 # column to an equivalent pyarrow type with from_numpy_dtype() 

357 type_list = [] 

358 for name in encode_table.dtype.names: 

359 dt = encode_table.dtype[name] 

360 if dt.type == np.object_: 

361 # If the column type is np.object_, then it should be a column 

362 # of variable-length arrays. This can be serialized with parquet 

363 # provided all of the elements have the same data-type. 

364 # Additionally, if the table has no elements, we cannot deduce 

365 # the datatype, and hence cannot serialize the table. 

366 if len(encode_table) > 0: 

367 obj_dtype = encode_table[name][0].dtype 

368 

369 # Check that the variable-length array all has the same type. 

370 for row in encode_table[name]: 

371 if row.dtype != obj_dtype: 

372 raise ValueError( 

373 f"Cannot serialize mixed-type column ({name}) with parquet." 

374 ) 

375 # Calling pa.list_() creates a ListType which is an array of variable- 

376 # length elements. 

377 arrow_type = pa.list_( 

378 value_type=pa.from_numpy_dtype(obj_dtype.type), 

379 ) 

380 else: 

381 raise ValueError( 

382 "Cannot serialize zero-length table " 

383 f"with object column ({name}) with parquet." 

384 ) 

385 elif len(dt.shape) > 0: 

386 # This column has a shape, and is an array type column. Calling 

387 # pa.list_() with a list_size creates a FixedSizeListType, which 

388 # is an array of fixed-length elements. 

389 arrow_type = pa.list_( 

390 value_type=pa.from_numpy_dtype(dt.subdtype[0].type), 

391 list_size=np.prod(dt.shape), 

392 ) 

393 else: 

394 # This is a standard column. 

395 arrow_type = pa.from_numpy_dtype(dt.type) 

396 

397 type_list.append((name, arrow_type)) 

398 

399 metadata = {} 

400 for name, col in encode_table.columns.items(): 

401 # Parquet will retain the datatypes of columns, but string and 

402 # byte column length is lost. Therefore, we special-case these 

403 # types to record the length for precise round-tripping. 

404 

405 t = col.dtype.type 

406 itemsize = col.dtype.itemsize 

407 if t is np.object_: 

408 t = encode_table[name][0].dtype.type 

409 if t == np.str_ or t == np.bytes_: 

410 # We need to scan through all of them. 

411 itemsize = -1 

412 for row in encode_table[name]: 

413 itemsize = max(itemsize, row.dtype.itemsize) 

414 

415 if t is np.str_: 

416 metadata[f"table::len::{name}"] = str(itemsize // 4) 

417 elif t is np.bytes_: 

418 metadata[f"table::len::{name}"] = str(itemsize) 

419 

420 metadata["table_meta_yaml"] = meta_yaml_str 

421 

422 # Pyarrow stores all metadata as byte strings, so we explicitly encode 

423 # our unicode strings in metadata as UTF-8 byte strings here. 

424 metadata_encode = { 

425 k.encode("UTF-8"): v.encode("UTF-8") for k, v in metadata.items() 

426 } 

427 

428 schema = pa.schema(type_list, metadata=metadata_encode) 

429 

430 if os.path.exists(output): 

431 if overwrite: 

432 # We must remove the file prior to writing below. 

433 os.remove(output) 

434 else: 

435 raise OSError(NOT_OVERWRITING_MSG.format(output)) 

436 

437 # We use version='2.0' for full support of datatypes including uint32. 

438 with parquet.ParquetWriter(output, schema, version=writer_version) as writer: 

439 # Convert each Table column to a pyarrow array 

440 arrays = [] 

441 for name in encode_table.dtype.names: 

442 dt = encode_table.dtype[name] 

443 # Parquet must be stored little-endian. When we use astype(..., copy=False) 

444 # we get a very fast conversion when the dtype is unchanged, and only 

445 # incur a cost when we need to do a byte-swap operation. 

446 dt_new = dt.newbyteorder("<") 

447 if dt.type == np.object_: 

448 # Turn the column into a list of numpy arrays. 

449 val = [row.astype(dt_new, copy=False) for row in encode_table[name]] 

450 elif len(dt.shape) > 0: 

451 if len(encode_table) > 0: 

452 val = np.split( 

453 encode_table[name].ravel().astype(dt_new.base, copy=False), 

454 len(encode_table), 

455 ) 

456 else: 

457 val = [] 

458 else: 

459 val = encode_table[name].astype(dt_new, copy=False) 

460 

461 arrays.append(pa.array(val, type=schema.field(name).type)) 

462 

463 # Create a pyarrow table from the list of arrays and the schema 

464 pa_table = pa.Table.from_arrays(arrays, schema=schema) 

465 # Write the pyarrow table to a file 

466 writer.write_table(pa_table)

469def _get_names(_dict): 

470 """Recursively find the names in a serialized column dictionary. 

471 

472 Parameters 

473 ---------- 

474 _dict : `dict` 

475 Dictionary from astropy __serialized_columns__ 

476 

477 Returns 

478 ------- 

479 all_names : `list` [`str`] 

480 All the column names mentioned in _dict and sub-dicts. 

481 """ 

482 all_names = [] 

483 for k, v in _dict.items(): 

484 if isinstance(v, dict): 

485 all_names.extend(_get_names(v)) 

486 elif k == "name": 

487 all_names.append(v) 

488 return all_names

503def get_pyarrow(): 

504 try: 

505 import pyarrow as pa 

506 from pyarrow import parquet 

507 except ImportError: 

508 raise Exception("pyarrow is required to read and write parquet files") 

509 

510 if minversion(pa, "6.0.0"):

512 else: 

513 writer_version = "2.0"

515 return pa, parquet, writer_version
