29def angular_separation(lon1, lat1, lon2, lat2): 

30 """ 

31 Angular separation between two points on a sphere. 

32 

33 Parameters 

34 ---------- 

35 lon1, lat1, lon2, lat2 : `~astropy.coordinates.Angle`, `~astropy.units.Quantity` or float 

36 Longitude and latitude of the two points. Quantities should be in 

37 angular units; floats in radians. 

38 

39 Returns 

40 ------- 

41 angular separation : `~astropy.units.Quantity` ['angle'] or float 

42 Type depends on input; ``Quantity`` in angular units, or float in 

43 radians. 

44 

45 Notes 

46 ----- 

47 The angular separation is calculated using the Vincenty formula [1]_, 

48 which is slightly more complex and computationally expensive than 

49 some alternatives, but is stable at at all distances, including the 

50 poles and antipodes. 

51 

52 .. [1] https://en.wikipedia.org/wiki/Great-circle_distance 

53 """ 

54 sdlon = np.sin(lon2 - lon1) 

55 cdlon = np.cos(lon2 - lon1) 

56 slat1 = np.sin(lat1) 

57 slat2 = np.sin(lat2) 

58 clat1 = np.cos(lat1) 

59 clat2 = np.cos(lat2) 

60 

61 num1 = clat2 * sdlon 

62 num2 = clat1 * slat2 - slat1 * clat2 * cdlon 

63 denominator = slat1 * slat2 + clat1 * clat2 * cdlon 

64 

65 return np.arctan2(np.hypot(num1, num2), denominator)

68def position_angle(lon1, lat1, lon2, lat2): 

69 """ 

70 Position Angle (East of North) between two points on a sphere. 

71 

72 Parameters 

73 ---------- 

74 lon1, lat1, lon2, lat2 : `~astropy.coordinates.Angle`, `~astropy.units.Quantity` or float 

75 Longitude and latitude of the two points. Quantities should be in 

76 angular units; floats in radians. 

77 

78 Returns 

79 ------- 

80 pa : `~astropy.coordinates.Angle` 

81 The (positive) position angle of the vector pointing from position 1 to 

82 position 2. If any of the angles are arrays, this will contain an array 

83 following the appropriate `numpy` broadcasting rules. 

84 

85 """ 

86 from .angles import Angle 

87 

88 deltalon = lon2 - lon1 

89 colat = np.cos(lat2) 

90 

91 x = np.sin(lat2) * np.cos(lat1) - colat * np.sin(lat1) * np.cos(deltalon) 

92 y = np.sin(deltalon) * colat 

93 

94 return Angle(np.arctan2(y, x), u.radian).wrap_at(360 * u.deg)

97def offset_by(lon, lat, posang, distance): 

98 """ 

99 Point with the given offset from the given point. 

100 

101 Parameters 

102 ---------- 

103 lon, lat, posang, distance : `~astropy.coordinates.Angle`, `~astropy.units.Quantity` or float 

104 Longitude and latitude of the starting point, 

105 position angle and distance to the final point. 

106 Quantities should be in angular units; floats in radians. 

107 Polar points at lat= +/-90 are treated as limit of +/-(90-epsilon) and same lon. 

108 

109 Returns 

110 ------- 

111 lon, lat : `~astropy.coordinates.Angle` 

112 The position of the final point. If any of the angles are arrays, 

113 these will contain arrays following the appropriate `numpy` broadcasting rules. 

114 0 <= lon < 2pi. 

115 """ 

116 from .angles import Angle 

117 

118 # Calculations are done using the spherical trigonometry sine and cosine rules 

119 # of the triangle A at North Pole, B at starting point, C at final point 

120 # with angles A (change in lon), B (posang), C (not used, but negative reciprocal posang) 

121 # with sides a (distance), b (final co-latitude), c (starting colatitude) 

122 # B, a, c are knowns; A and b are unknowns 

123 # https://en.wikipedia.org/wiki/Spherical_trigonometry 

124 

125 cos_a = np.cos(distance) 

126 sin_a = np.sin(distance) 

127 cos_c = np.sin(lat) 

128 sin_c = np.cos(lat) 

129 cos_B = np.cos(posang) 

130 sin_B = np.sin(posang) 

131 

132 # cosine rule: Know two sides: a,c and included angle: B; get unknown side b 

133 cos_b = cos_c * cos_a + sin_c * sin_a * cos_B 

134 # sin_b = np.sqrt(1 - cos_b**2) 

135 # sine rule and cosine rule for A (using both lets arctan2 pick quadrant). 

136 # multiplying both sin_A and cos_A by x=sin_b * sin_c prevents /0 errors 

137 # at poles. Correct for the x=0 multiplication a few lines down. 

138 # sin_A/sin_a == sin_B/sin_b # Sine rule 

139 xsin_A = sin_a * sin_B * sin_c 

140 # cos_a == cos_b * cos_c + sin_b * sin_c * cos_A # cosine rule 

141 xcos_A = cos_a - cos_b * cos_c 

142 

143 A = Angle(np.arctan2(xsin_A, xcos_A), u.radian) 

144 # Treat the poles as if they are infinitesimally far from pole but at given lon 

145 small_sin_c = sin_c < 1e-12 

146 if small_sin_c.any(): 

147 # For south pole (cos_c = -1), A = posang; for North pole, A=180 deg - posang 

148 A_pole = (90 * u.deg + cos_c * (90 * u.deg - Angle(posang, u.radian))).to(u.rad) 

149 if A.shape: 

150 # broadcast to ensure the shape is like that of A, which is also 

151 # affected by the (possible) shapes of lat, posang, and distance. 

152 small_sin_c = np.broadcast_to(small_sin_c, A.shape) 

153 A[small_sin_c] = A_pole[small_sin_c] 

154 else: 

155 A = A_pole 

156 

157 outlon = (Angle(lon, u.radian) + A).wrap_at(360.0 * u.deg).to(u.deg) 

158 outlat = Angle(np.arcsin(cos_b), u.radian).to(u.deg) 

159 

160 return outlon, outlat

163def golden_spiral_grid(size): 

164 """Generate a grid of points on the surface of the unit sphere using the 

165 Fibonacci or Golden Spiral method. 

166 

167 .. seealso:: 

168 

169 `Evenly distributing points on a sphere <https://stackoverflow.com/questions/9600801/evenly-distributing-n-points-on-a-sphere>`_ 

170 

171 Parameters 

172 ---------- 

173 size : int 

174 The number of points to generate. 

175 

176 Returns 

177 ------- 

178 rep : `~astropy.coordinates.UnitSphericalRepresentation` 

179 The grid of points. 

180 """ 

181 golden_r = (1 + 5**0.5) / 2 

182 

183 grid = np.arange(0, size, dtype=float) + 0.5 

184 lon = _TWOPI / golden_r * grid * u.rad 

185 lat = np.arcsin(1 - 2 * grid / size) * u.rad 

186 

187 return UnitSphericalRepresentation(lon, lat)

190def uniform_spherical_random_surface(size=1): 

191 """Generate a random sampling of points on the surface of the unit sphere. 

192 

193 Parameters 

194 ---------- 

195 size : int 

196 The number of points to generate. 

197 

198 Returns 

199 ------- 

200 rep : `~astropy.coordinates.UnitSphericalRepresentation` 

201 The random points. 

202 """ 

203 rng = np.random # can maybe switch to this being an input later - see #11628 

204 

205 lon = rng.uniform(0, _TWOPI, size) * u.rad 

206 lat = np.arcsin(rng.uniform(-1, 1, size=size)) * u.rad 

207 

208 return UnitSphericalRepresentation(lon, lat)

211def uniform_spherical_random_volume(size=1, max_radius=1): 

212 """Generate a random sampling of points that follow a uniform volume 

213 density distribution within a sphere. 

214 

215 Parameters 

216 ---------- 

217 size : int 

218 The number of points to generate. 

219 max_radius : number, quantity-like, optional 

220 A dimensionless or unit-ful factor to scale the random distances. 

221 

222 Returns 

223 ------- 

224 rep : `~astropy.coordinates.SphericalRepresentation` 

225 The random points. 

226 """ 

227 rng = np.random # can maybe switch to this being an input later - see #11628 

228 

229 usph = uniform_spherical_random_surface(size=size) 

230 

231 r = np.cbrt(rng.uniform(size=size)) * u.Quantity(max_radius, copy=False) 

232 return SphericalRepresentation(usph.lon, usph.lat, r)
