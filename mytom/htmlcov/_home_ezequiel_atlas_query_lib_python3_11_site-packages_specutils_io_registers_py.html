171 def _registered_readers(): 

172 # With the implementation of priorities support in the astropy registry 

173 # loaders, astropy version 4.2 and up return a tuple of ``func`` and 

174 # ``priority``, while versions < 4.2 return just the ``func`` object. 

175 # This function ignores priorities when calling extension loaders. 

176 return [((fmt, cls), func[0]) 

177 if isinstance(func, tuple) else ((fmt, cls), func) 

178 for (fmt, cls), func in io_registry._readers.items()] 

179 

180 return [fmt for (fmt, cls), func in _registered_readers() 

181 if issubclass(cls, Spectrum1D) and 

182 func.extensions is not None and
