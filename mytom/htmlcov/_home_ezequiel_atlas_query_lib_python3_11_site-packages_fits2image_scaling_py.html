55def least_squares_line_fit(sample_data, max_iterations=5, min_fit=0.5): 

56 nsamples = len(sample_data) 

57 max_masked_samples = nsamples * (1 - min_fit) 

58 

59 # Contains True if sample is included in fit False is sample is excluded

73 break 

74

76 sample_mask = np.ones(len(fit_samples), dtype=bool)

78 x = np.array(range(nfitsamples)) 

79 y = np.array(fit_samples) 

80 

81 A = np.vstack([x, np.ones(nfitsamples)]).T 

82 result = np.linalg.lstsq(A,y, rcond=None) 

83 slope, y_intercept = result[0] 

84 residuals = result[1] 

85 

86 # Need to check residual and remove outliers before refit if residual is large 

87 mean_residual = residuals / nfitsamples 

88 rms = math.sqrt(mean_residual) 

89 fitline = np.array(range(nfitsamples)) 

90 fitline = fitline * slope 

91 fitline = fitline + y_intercept 

92 over_threshold = fitline + rms 

93 under_threshold = fitline - rms 

94 

95 sample_mask[np.where(y > over_threshold)[0]] = False 

96 sample_mask[np.where(y < under_threshold)[0]] = False 

97 

98 if last_fit is None or rms < last_fit_rms: 

99 last_fit = (slope, y_intercept, i+1, nfitsamples, rms, fit_samples)

121def calc_zscale_min_max(samples, contrast= 0.1, iterations=5): 

122 nsamples = len(samples) 

123 slope, y_intecept, used_iterations, nfitsamples, rms, fit_samples = least_squares_line_fit(samples, max_iterations=iterations) 

124 

125 zmin = samples[0] 

126 zmax = samples[nsamples - 1] 

127 if contrast > 0.0: 

128 slope = slope / contrast 

129 

130 fitted_dy = slope * nsamples/2 

131 zmin = max(zmin, y_intecept - fitted_dy) 

132 zmax = min(zmax, y_intecept + fitted_dy) 

133 

134 return zmin, zmax, rms

137def linear_scale(data, zmin, zmax, max_val=255, gamma_adjust=2.5): 

138 '''Apply a linear rescale of the supplied fits images cliping between 

139 zmin and zmax and writing to the supplied outfile. 

140 ''' 

141 if zmax == zmin: 

142 # This will produce bad images, but its better than crashing with divide by zero error 

143 logging.warning("Calculated same zmax and zmin. Using zmax=zmax+1, zmin=zmin-1 instead. Image may look bad.") 

144 zmax = zmax + 1 

145 zmin = zmin - 1 

146 

147 scale = float(max_val) / (float(zmax) - float(zmin)) 

148 adjust = scale * zmin 

149 

150 data = data.astype('float') 

151 data.clip(zmin, zmax, data) 

152 data *= scale 

153 data -= adjust 

154 data.round(out=data) 

155 gamma_lookup_table = gamma_adjust_table(data.dtype, max_val=max_val, gamma_adjust=gamma_adjust) 

156 

157 #data = gamma_lookup_table[data] the below line is supposedly faster 

158 np.take(gamma_lookup_table, data.astype('int64'), out=data) 

159 

160 data = data.astype('uint8') 

161 return data

164def gamma_adjust_table(dtype, max_val=255.0, min_val=0.0, gamma_adjust=2.5): 

165 '''Creates a lookup table for a gamma adjustment 

166 ''' 

167 size = int(max_val) - int(min_val) + 1 

168 gamma_lookup_table = range(size) 

169 gamma_lookup_table = [int(size * (math.pow(float(i)/float(size), 1.0/float(gamma_adjust)))) for i in gamma_lookup_table] 

170 

171 return np.array(gamma_lookup_table, dtype=dtype)

174def auto_scale(path_to_frame, nsamples=2000, max_val=255, contrast=0.1, gamma_adjust=2.5, max_fit_iterations=1): 

175 '''Uses a zscale fit and rescales the image accordingly by calling 

176 linear scale 

177 ''' 

178 data, header = get_reduced_dimensionality_data(path_to_frame) 

179 samples = extract_samples(data, header, nsamples) 

180 median = np.median(samples)

185def get_reduced_dimensionality_data(path_to_frame): 

186 ''' 

187 Reduce the dimensionality of the data by 1. For sinistro images, this will give the first quadrant 

188 :param path_to_frame: path to fits file 

189 :return: header and modified data from astropy.io.fits 

190 ''' 

191 if type(path_to_frame) == str: 

192 path_to_frame = Path(path_to_frame) 

193 with path_to_frame.open('rb') as p: 

194 with fits.open(p) as hdul: 

195 for hdu in hdul: 

196 ''' 

197 For most images the shape of first HDU data is () and the first HDU data is the  

198 dimensions of the CCD. 

199 For sinistro, the first HDU data has shape (0,0) and subsequent HDUs are 1/4 of 

200 the chip dimensions. 

201 Therefore, just checking for a shape with 2 elements is not sufficient to identify data. 

202 We also need to check for non-zero shape elements. 

203 ''' 

204 if len(np.shape(hdu)) == 2 and np.shape(hdu)[0] > 0: 

205 return hdu.data, hdu.header 

206 raise Exception('No fits data found')

209def percentile_scale(path_to_frame, lower_percentile=5.0, upper_percentile=99.0): 

210 ''' This is not currently used, but can be compared against the auto_scale /zscaled method of image scaling 

211 It doesn't appear to work as well with nebulas/galaxies as the zscale method but works well for stars 

212 :param path_to_frame: 

213 :param lower_percentile: 

214 :param upper_percentile: 

215 :return: 

216 ''' 

217 data = fits.getdata(path_to_frame) 

218 (lower_threshold, upper_threshold) = np.percentile(data, [lower_percentile, upper_percentile]) 

219 

220 data = data.astype('float') 

221 data = ((data-lower_threshold)/(upper_threshold-lower_threshold))*255.0 

222 data[data < 0] = 0 

223 data[data > 255] = 255 

224 data = data.astype('uint8') 

225 return data

227def recalculate_median(data, percentile=99.5): 

228 data = data.astype('float') 

229 data[data < 0.] = 0. 

230 median = np.median(data) 

231 data -= median
