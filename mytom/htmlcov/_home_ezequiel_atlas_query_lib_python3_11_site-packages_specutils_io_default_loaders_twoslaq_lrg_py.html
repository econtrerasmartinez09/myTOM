27def twoslaq_lrg_fits_loader(file_obj, **kwargs): 

28 """ 

29 Load a file from the LRG subset of the 2dF-SDSS LRG/QSO survey (2SLAQ-LRG) 

30 file. 

31 

32 2SLAQ was one of a number of surveys that used the 2dF instrument on the 

33 Anglo-Australian Telescope (AAT) at Siding Spring Observatory (SSO) near 

34 Coonabarabran, Australia, and followed up the 2QZ survey. Further details 

35 can be seen at http://www.physics.usyd.edu.au/2slaq/ or at 

36 https://docs.datacentral.org.au/. 

37 

38 The LRG and QSO data appear to be in different formats, this only loads the 

39 LRG version. As there is a science and sky spectrum, the `SpectrumList` 

40 loader provides both, whereas the `Spectrum1D` loader only provides the 

41 science. 

42 

43 Parameters 

44 ---------- 

45 file_name: str 

46 The path to the FITS file 

47 Returns 

48 ------- 

49 data: SpectrumList 

50 The 2SLAQ-LRG spectrum that is represented by the data in this file. 

51 """ 

52 

53 with read_fileobj_or_hdulist(file_obj, **kwargs) as hdulist: 

54 header = hdulist[0].header 

55 spectrum = hdulist[0].data[0, 0] * Unit("count/s") 

56 sky = hdulist[0].data[1, 0] * Unit("count/s") 

57 

58 # Due to the odd structure of the file, the WCS needs to be read in 

59 # manually 

60 wcs = WCS(naxis=1) 

61 wcs.wcs.cdelt[0] = header["CD1_1"] 

62 wcs.wcs.crval[0] = header["CRVAL1"] 

63 wcs.wcs.crpix[0] = header["CRPIX1"] 

64 wcs.wcs.cunit[0] = Unit("Angstrom") 

65 

66 meta = {"header": header} 

67 

68 return SpectrumList([ 

69 Spectrum1D(flux=spectrum, wcs=wcs, meta=meta), 

70 Spectrum1D(flux=sky, wcs=wcs, meta=meta), 

71 ])
