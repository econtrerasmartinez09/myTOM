89def _load_manga_spectra(hdulist, per_unit=None): 

90 """ Return a MaNGA Spectrum1D object 

91 

92 Returns a Spectrum1D object for a MaNGA data files. Use the `per_unit` 

93 kwarg to indicate the "spaxel" or "fiber" unit for cubes and rss files, 

94 respectively. Note that the spectral axis will automatically be moved to 

95 be last during Spectrum1D initialization. 

96 

97 Parameters 

98 ---------- 

99 hdulist : fits.HDUList 

100 A MaNGA read astropy fits HDUList 

101 per_unit : astropy.units.Unit 

102 An astropy unit to divide the default flux unit by 

103 

104 Returns 

105 ------- 

106 Spectrum1D 

107 The spectrum contained in the file. 

108 """ 

109 unit = u.Unit('1e-17 erg / (Angstrom cm2 s)') 

110 if per_unit: 

111 unit = unit / per_unit 

112 

113 hdr = hdulist['PRIMARY'].header 

114 wcs = WCS(hdulist['FLUX'].header) 

115 

116 flux = hdulist['FLUX'].data * unit 

117 ivar = InverseVariance(hdulist["IVAR"].data) 

118 # SDSS masks are arrays of bit values storing multiple boolean conditions. 

119 mask = hdulist['MASK'].data != 0 

120 

121 return Spectrum1D(flux=flux, meta={'header': hdr}, wcs=wcs, 

122 uncertainty=ivar, mask=mask)
