13def get_sidereal_visibility(target, start_time, end_time, interval, airmass_limit, observation_facility=None): 

14 """ 

15 Uses astroplan to calculate the airmass for a sidereal target 

16 for each given interval between the start and end times. 

17 

18 The resulting data omits any airmass above the provided limit (or 

19 default, if one is not provided), as well as any airmass calculated 

20 during the day (defined as between astronomical twilights). 

21 

22 Important note: only works for sidereal targets! For non-sidereal visibility, see here: 

23 https://github.com/TOMToolkit/tom_nonsidereal_airmass 

24 

25 :param start_time: start of the window for which to calculate the airmass 

26 :type start_time: datetime 

27 

28 :param end_time: end of the window for which to calculate the airmass 

29 :type end_time: datetime 

30 

31 :param interval: time interval, in minutes, at which to calculate airmass within the given window 

32 :type interval: int 

33 

34 :param airmass_limit: maximum acceptable airmass for the resulting calculations 

35 :type airmass_limit: int 

36 

37 :param observation_facility: observing facility for which to calculate the airmass. None indicates all available 

38 facilities. 

39 :type observation_facility: BaseObservationFacility 

40 

41 :returns: A dictionary containing the airmass data for each site. The dict keys consist of the site name prepended 

42 with the observing facility. The values are the airmass data, structured as an array containing two arrays. The 

43 first array contains the set of datetimes used in the airmass calculations. The second array contains the 

44 corresponding set of airmasses calculated. 

45 :rtype: dict 

46 """ 

47 

48 if target.type != 'SIDEREAL': 

49 msg = '\033[1m\033[91mAirmass plotting is only supported for sidereal targets\033[0m' 

50 logger.info(msg) 

51 empty_visibility = {} 

52 return empty_visibility 

53 

54 if end_time < start_time: 

55 raise Exception('Start must be before end') 

56 

57 if airmass_limit is None: 

58 airmass_limit = 10 

59 

60 if observation_facility is None: 

61 facilities = facility.get_service_classes() 

62 else: 

63 facilities = [observation_facility] 

64 

65 body = FixedTarget(name=target.name, coord=SkyCoord(target.ra, target.dec, unit='deg')) 

66 

67 visibility = {} 

68 sun, time_range = get_astroplan_sun_and_time(start_time, end_time, interval) 

69 for observing_facility in facilities: 

70 observing_facility_class = facility.get_service_class(observing_facility) 

71 sites = observing_facility_class().get_observing_sites() 

72 for site, site_details in sites.items(): 

73 observer = Observer(longitude=site_details.get('longitude')*units.deg, 

74 latitude=site_details.get('latitude')*units.deg, 

75 elevation=site_details.get('elevation')*units.m) 

76 

77 sun_alt = observer.altaz(time_range, sun).alt 

78 obj_airmass = observer.altaz(time_range, body).secz 

79 

80 bad_indices = np.argwhere( 

81 (obj_airmass >= airmass_limit) | 

82 (obj_airmass <= 1) | 

83 (sun_alt > -18*units.deg) # between astronomical twilights, i.e. sun is up 

84 ) 

85 

86 obj_airmass = [None if i in bad_indices else float(airmass) for i, airmass in enumerate(obj_airmass)] 

87 

88 visibility[f'({observing_facility}) {site}'] = (time_range.datetime, obj_airmass) 

89 return visibility

92def get_astroplan_sun_and_time(start_time, end_time, interval): 

93 """ 

94 Uses astroplan's time_grid_from_range to generate 

95 an astropy Time object covering the time range. 

96 

97 Uses astropy's get_sun to generate sun positions over 

98 that time range. 

99 

100 If time range is small and interval is coarse, approximates 

101 the sun at a fixed position from the middle of the 

102 time range to speed up calculations. 

103 Since the sun moves ~4 minutes a day, this approximation 

104 happens when the number of days covered by the time range 

105 * 4 is less than the interval (in minutes) / 2. 

106 

107 :param start_time: start of the window for which to calculate the airmass 

108 :type start_time: datetime 

109 

110 :param end_time: end of the window for which to calculate the airmass 

111 :type end_time: datetime 

112 

113 :param interval: time interval, in minutes, at which to calculate airmass within the given window 

114 :type interval: int 

115 

116 :returns: ra/dec positions of the sun over the time range, 

117 time range between start_time and end_time at interval 

118 :rtype: astropy SkyCoord, astropy Time 

119 """ 

120 

121 start = Time(start_time) 

122 end = Time(end_time) 

123 

124 time_range = time_grid_from_range(time_range=[start, end], time_resolution=interval*units.minute) 

125 

126 number_of_days = end.mjd - start.mjd 

127 if number_of_days*4 < float(interval)/2: 

128 # Hack to speed up calculation by factor of ~3 

129 sun_coords = get_sun(time_range[int(len(time_range)/2)]) 

130 sun = FixedTarget(name='sun', coord=SkyCoord(sun_coords.ra, sun_coords.dec, unit='deg')) 

131 else: 

132 sun = get_sun(time_range) 

133 

134 return sun, time_range
