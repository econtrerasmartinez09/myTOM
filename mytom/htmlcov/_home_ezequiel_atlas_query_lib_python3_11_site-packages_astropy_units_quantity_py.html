260class Quantity(np.ndarray): 

261 """A `~astropy.units.Quantity` represents a number with some associated unit. 

262 

263 See also: https://docs.astropy.org/en/stable/units/quantity.html 

264 

265 Parameters 

266 ---------- 

267 value : number, `~numpy.ndarray`, `~astropy.units.Quantity` (sequence), or str 

268 The numerical value of this quantity in the units given by unit. If a 

269 `Quantity` or sequence of them (or any other valid object with a 

270 ``unit`` attribute), creates a new `Quantity` object, converting to 

271 `unit` units as needed. If a string, it is converted to a number or 

272 `Quantity`, depending on whether a unit is present. 

273 

274 unit : unit-like 

275 An object that represents the unit associated with the input value. 

276 Must be an `~astropy.units.UnitBase` object or a string parseable by 

277 the :mod:`~astropy.units` package. 

278 

279 dtype : ~numpy.dtype, optional 

280 The dtype of the resulting Numpy array or scalar that will 

281 hold the value. If not provided, it is determined from the input, 

282 except that any integer and (non-Quantity) object inputs are converted 

283 to float by default. 

284 If `None`, the normal `numpy.dtype` introspection is used, e.g. 

285 preventing upcasting of integers. 

286 

287 copy : bool, optional 

288 If `True` (default), then the value is copied. Otherwise, a copy will 

289 only be made if ``__array__`` returns a copy, if value is a nested 

290 sequence, or if a copy is needed to satisfy an explicitly given 

291 ``dtype``. (The `False` option is intended mostly for internal use, 

292 to speed up initialization where a copy is known to have been made. 

293 Use with care.) 

294 

295 order : {'C', 'F', 'A'}, optional 

296 Specify the order of the array. As in `~numpy.array`. This parameter 

297 is ignored if the input is a `Quantity` and ``copy=False``. 

298 

299 subok : bool, optional 

300 If `False` (default), the returned array will be forced to be a

303 `~astropy.units.Dex` for ``u.dex(u.AA)``). 

304 

305 ndmin : int, optional 

306 Specifies the minimum number of dimensions that the resulting array 

307 should have. Ones will be prepended to the shape as needed to meet 

308 this requirement. This parameter is ignored if the input is a 

309 `Quantity` and ``copy=False``. 

310 

311 Raises 

312 ------ 

313 TypeError 

314 If the value provided is not a Python numeric type. 

315 TypeError 

316 If the unit provided is not either a :class:`~astropy.units.Unit` 

317 object or a parseable string unit. 

318 

319 Notes 

320 ----- 

321 Quantities can also be created by multiplying a number or array with a 

322 :class:`~astropy.units.Unit`. See https://docs.astropy.org/en/latest/units/

349

357 unit_shape_dtype : :class:`~astropy.units.UnitBase`, str, `~astropy.units.PhysicalType`, or tuple

359 If tuple, then the first element is the unit specification 

360 and all other elements are for `numpy.ndarray` type annotations. 

361 Whether they are included depends on the python and :mod:`numpy` 

362 versions. 

363 

364 Returns 

365 ------- 

366 `typing.Annotated`, `astropy.units.Unit`, or `astropy.units.PhysicalType` 

367 Return type in this preference order:

369 * `astropy.units.Unit` or `astropy.units.PhysicalType`

612 def __array_ufunc__(self, function, method, *inputs, **kwargs): 

613 """Wrap numpy ufuncs, taking care of units. 

614 

615 Parameters 

616 ---------- 

617 function : callable 

618 ufunc to wrap. 

619 method : str 

620 Ufunc method: ``__call__``, ``at``, ``reduce``, etc. 

621 inputs : tuple 

622 Input arrays. 

623 kwargs : keyword arguments 

624 As passed on, with ``out`` containing possible quantity output. 

625 

626 Returns 

627 ------- 

628 result : `~astropy.units.Quantity` or `NotImplemented` 

629 Results of the ufunc, with the unit set properly. 

630 """ 

631 # Determine required conversion functions -- to bring the unit of the 

632 # input to that expected (e.g., radian for np.sin), or to get 

633 # consistent units between two inputs (e.g., in np.add) -- 

634 # and the unit of the result (or tuple of units for nout > 1). 

635 try: 

636 converters, unit = converters_and_unit(function, method, *inputs) 

637 

638 out = kwargs.get("out", None) 

639 # Avoid loop back by turning any Quantity output into array views. 

640 if out is not None: 

641 # If pre-allocated output is used, check it is suitable. 

642 # This also returns array view, to ensure we don't loop back. 

643 if function.nout == 1: 

644 out = out[0] 

645 out_array = check_output(out, unit, inputs, function=function) 

646 # Ensure output argument remains a tuple. 

647 kwargs["out"] = (out_array,) if function.nout == 1 else out_array 

648 

649 if method == "reduce" and "initial" in kwargs and unit is not None: 

650 # Special-case for initial argument for reductions like 

651 # np.add.reduce. This should be converted to the output unit as 

652 # well, which is typically the same as the input unit (but can 

653 # in principle be different: unitless for np.equal, radian 

654 # for np.arctan2, though those are not necessarily useful!) 

655 kwargs["initial"] = self._to_own_unit( 

656 kwargs["initial"], check_precision=False, unit=unit 

657 ) 

658 

659 # Same for inputs, but here also convert if necessary. 

660 arrays = [] 

661 for input_, converter in zip(inputs, converters): 

662 input_ = getattr(input_, "value", input_) 

663 arrays.append(converter(input_) if converter else input_) 

664 

665 # Call our superclass's __array_ufunc__

693 def _result_as_quantity(self, result, unit, out): 

694 """Turn result into a quantity with the given unit. 

695 

696 If no output is given, it will take a view of the array as a quantity, 

697 and set the unit. If output is given, those should be quantity views 

698 of the result arrays, and the function will just set the unit. 

699 

700 Parameters 

701 ---------- 

702 result : ndarray or tuple thereof 

703 Array(s) which need to be turned into quantity. 

704 unit : `~astropy.units.Unit` 

705 Unit for the quantities to be returned (or `None` if the result 

706 should not be a quantity). Should be tuple if result is a tuple. 

707 out : `~astropy.units.Quantity` or None 

708 Possible output quantity. Should be `None` or a tuple if result 

709 is a tuple. 

710 

711 Returns 

712 ------- 

713 out : `~astropy.units.Quantity` 

714 With units set. 

715 """ 

716 if isinstance(result, (tuple, list)): 

717 if out is None: 

718 out = (None,) * len(result) 

719 # Some np.linalg functions return namedtuple, which is handy to access 

720 # elements by name, but cannot be directly initialized with an iterator. 

721 result_cls = getattr(result, "_make", result.__class__) 

722 return result_cls( 

723 self._result_as_quantity(result_, unit_, out_) 

724 for (result_, unit_, out_) in zip(result, unit, out) 

725 ) 

726 

727 if out is None: 

728 # View the result array as a Quantity with the proper unit. 

729 return ( 

730 result 

731 if unit is None 

732 else self._new_view(result, unit, propagate_info=False) 

733 ) 

734 

735 elif isinstance(out, Quantity): 

736 # For given Quantity output, just set the unit. We know the unit 

737 # is not None and the output is of the correct Quantity subclass, 

738 # as it was passed through check_output. 

739 # (We cannot do this unconditionally, though, since it is possible 

740 # for out to be ndarray and the unit to be dimensionless.) 

741 out._set_unit(unit) 

742 

743 return out

763 def _new_view(self, obj=None, unit=None, propagate_info=True): 

764 """Create a Quantity view of some array-like input, and set the unit. 

765 

766 By default, return a view of ``obj`` of the same class as ``self`` and 

767 with the same unit. Subclasses can override the type of class for a 

768 given unit using ``__quantity_subclass__``, and can ensure properties 

769 other than the unit are copied using ``__array_finalize__``. 

770 

771 If the given unit defines a ``_quantity_class`` of which ``self`` 

772 is not an instance, a view using this class is taken. 

773 

774 Parameters 

775 ---------- 

776 obj : ndarray or scalar, optional 

777 The array to create a view of. If obj is a numpy or python scalar, 

778 it will be converted to an array scalar. By default, ``self`` 

779 is converted. 

780 

781 unit : unit-like, optional 

782 The unit of the resulting object. It is used to select a 

783 subclass, and explicitly assigned to the view if given. 

784 If not given, the subclass and unit will be that of ``self``. 

785 

786 propagate_info : bool, optional 

787 Whether to transfer ``info`` if present. Default: `True`, as 

788 appropriate for, e.g., unit conversions or slicing, where the 

789 nature of the object does not change. 

790 

791 Returns 

792 ------- 

793 view : `~astropy.units.Quantity` subclass 

794 

795 """ 

796 # Determine the unit and quantity subclass that we need for the view. 

797 if unit is None: 

798 unit = self.unit 

799 quantity_subclass = self.__class__ 

800 elif unit is self.unit and self.__class__ is Quantity: 

801 # The second part is because we should not presume what other 

802 # classes want to do for the same unit. E.g., Constant will 

803 # always want to fall back to Quantity, and relies on going 

804 # through `__quantity_subclass__`. 

805 quantity_subclass = Quantity 

806 else: 

807 unit = Unit(unit) 

808 quantity_subclass = getattr(unit, "_quantity_class", Quantity) 

809 if isinstance(self, quantity_subclass): 

810 quantity_subclass, subok = self.__quantity_subclass__(unit) 

811 if subok: 

812 quantity_subclass = self.__class__ 

813 

814 # We only want to propagate information from ``self`` to our new view, 

815 # so obj should be a regular array. By using ``np.array``, we also 

816 # convert python and numpy scalars, which cannot be viewed as arrays 

817 # and thus not as Quantity either, to zero-dimensional arrays. 

818 # (These are turned back into scalar in `.value`) 

819 # Note that for an ndarray input, the np.array call takes only double 

820 # ``obj.__class is np.ndarray``. So, not worth special-casing. 

821 if obj is None: 

822 obj = self.view(np.ndarray) 

823 else: 

824 obj = np.array(obj, copy=False, subok=True) 

825 

826 # Take the view, set the unit, and update possible other properties 

827 # such as ``info``, ``wrap_angle`` in `Longitude`, etc. 

828 view = obj.view(quantity_subclass) 

829 view._set_unit(unit) 

830 view.__array_finalize__(self) 

831 if propagate_info and "info" in self.__dict__: 

832 view.info = self.info 

833 return view

1628 def _to_own_unit(self, value, check_precision=True, *, unit=None): 

1629 """Convert value to one's own unit (or that given). 

1630

1632 for values of 0, infinity or nan, which are allowed to have any unit. 

1633 

1634 Parameters 

1635 ---------- 

1636 value : anything convertible to `~astropy.units.Quantity` 

1637 The value to be converted to the requested unit. 

1638 check_precision : bool 

1639 Whether to forbid conversion of float to integer if that changes 

1640 the input number. Default: `True`. 

1641 unit : `~astropy.units.Unit` or None 

1642 The unit to convert to. By default, the unit of ``self``. 

1643 

1644 Returns 

1645 ------- 

1646 value : number or `~numpy.ndarray` 

1647 In the requested units. 

1648 

1649 """ 

1650 if unit is None: 

1651 unit = self.unit 

1652 try: 

1653 _value = value.to_value(unit) 

1654 except AttributeError: 

1655 # We're not a Quantity. 

1656 # First remove two special cases (with a fast test): 

1657 # 1) Maybe masked printing? MaskedArray with quantities does not 

1658 # work very well, but no reason to break even repr and str. 

1659 # 2) np.ma.masked? useful if we're a MaskedQuantity. 

1660 if value is np.ma.masked or ( 

1661 value is np.ma.masked_print_option and self.dtype.kind == "O" 

1662 ): 

1663 return value 

1664 # Now, let's try a more general conversion. 

1665 # Plain arrays will be converted to dimensionless in the process, 

1666 # but anything with a unit attribute will use that. 

1667 try: 

1668 as_quantity = Quantity(value) 

1669 _value = as_quantity.to_value(unit) 

1670 except UnitsError: 

1671 # last chance: if this was not something with a unit 

1672 # and is all 0, inf, or nan, we treat it as arbitrary unit. 

1673 if not hasattr(value, "unit") and can_have_arbitrary_unit( 

1674 as_quantity.value 

1675 ): 

1676 _value = as_quantity.value 

1677 else: 

1678 raise 

1679 

1680 if self.dtype.kind == "i" and check_precision: 

1681 # If, e.g., we are casting float to int, we want to fail if 

1682 # precision is lost, but let things pass if it works. 

1683 _value = np.array(_value, copy=False, subok=True) 

1684 if not np.can_cast(_value.dtype, self.dtype): 

1685 self_dtype_array = np.array(_value, self.dtype, subok=True) 

1686 if not np.all((self_dtype_array == _value) | np.isnan(_value)):

1804 def __array_function__(self, function, types, args, kwargs): 

1805 """Wrap numpy functions, taking care of units. 

1806 

1807 Parameters 

1808 ---------- 

1809 function : callable 

1810 Numpy function to wrap 

1811 types : iterable of classes 

1812 Classes that provide an ``__array_function__`` override. Can 

1813 in principle be used to interact with other classes. Below, 

1814 mostly passed on to `~numpy.ndarray`, which can only interact 

1815 with subclasses. 

1816 args : tuple 

1817 Positional arguments provided in the function call. 

1818 kwargs : dict 

1819 Keyword arguments provided in the function call. 

1820 

1821 Returns 

1822 ------- 

1823 result: `~astropy.units.Quantity`, `~numpy.ndarray` 

1824 As appropriate for the function. If the function is not 

1825 supported, `NotImplemented` is returned, which will lead to 

1826 a `TypeError` unless another argument overrode the function. 

1827 

1828 Raises 

1829 ------ 

1830 ~astropy.units.UnitsError 

1831 If operands have incompatible units. 

1832 """ 

1833 # A function should be in one of the following sets or dicts: 

1834 # 1. SUBCLASS_SAFE_FUNCTIONS (set), if the numpy implementation 

1835 # supports Quantity; we pass on to ndarray.__array_function__. 

1836 # 2. FUNCTION_HELPERS (dict), if the numpy implementation is usable 

1837 # after converting quantities to arrays with suitable units, 

1838 # and possibly setting units on the result. 

1839 # 3. DISPATCHED_FUNCTIONS (dict), if the function makes sense but 

1840 # requires a Quantity-specific implementation. 

1841 # 4. UNSUPPORTED_FUNCTIONS (set), if the function does not make sense. 

1842 # For now, since we may not yet have complete coverage, if a 

1843 # function is in none of the above, we simply call the numpy 

1844 # implementation. 

1845 if function in SUBCLASS_SAFE_FUNCTIONS: 

1846 return super().__array_function__(function, types, args, kwargs) 

1847 

1848 elif function in FUNCTION_HELPERS: 

1849 function_helper = FUNCTION_HELPERS[function] 

1850 try: 

1851 args, kwargs, unit, out = function_helper(*args, **kwargs) 

1852 except NotImplementedError: 

1853 return self._not_implemented_or_raise(function, types) 

1854 

1855 result = super().__array_function__(function, types, args, kwargs) 

1856 # Fall through to return section 

1857 

1858 elif function in DISPATCHED_FUNCTIONS: 

1859 dispatched_function = DISPATCHED_FUNCTIONS[function] 

1860 try: 

1861 result, unit, out = dispatched_function(*args, **kwargs) 

1862 except NotImplementedError: 

1863 return self._not_implemented_or_raise(function, types) 

1864 

1865 # Fall through to return section 

1866 

1867 elif function in UNSUPPORTED_FUNCTIONS: 

1868 return NotImplemented 

1869 

1870 else: 

1871 warnings.warn( 

1872 f"function '{function.__name__}' is not known to astropy's Quantity." 

1873 " Will run it anyway, hoping it will treat ndarray subclasses" 

1874 " correctly. Please raise an issue at" 

1875 " https://github.com/astropy/astropy/issues.", 

1876 AstropyWarning, 

1877 ) 

1878 return super().__array_function__(function, types, args, kwargs) 

1879 

1880 # If unit is None, a plain array is expected (e.g., boolean), which 

1881 # means we're done. 

1882 # We're also done if the result was NotImplemented, which can happen 

1883 # if other inputs/outputs override __array_function__; 

1884 # hopefully, they can then deal with us. 

1885 if unit is None or result is NotImplemented: 

1886 return result 

1887 

1888 return self._result_as_quantity(result, unit, out=out)

1890 def _not_implemented_or_raise(self, function, types): 

1891 # Our function helper or dispatcher found that the function does not 

1892 # work with Quantity. In principle, there may be another class that 

1893 # knows what to do with us, for which we should return NotImplemented. 

1894 # But if there is ndarray (or a non-Quantity subclass of it) around, 

1895 # it quite likely coerces, so we should just break. 

1896 if any( 

1897 issubclass(t, np.ndarray) and not issubclass(t, Quantity) for t in types 

1898 ): 

1899 raise TypeError( 

1900 f"the Quantity implementation cannot handle {function} " 

1901 "with the given arguments." 

1902 ) from None 

1903 else: 

1904 return NotImplemented

1906 # Calculation -- override ndarray methods to take into account units. 

1907 # We use the corresponding numpy functions to evaluate the results, since 

1908 # the methods do not always allow calling with keyword arguments. 

1909 # For instance, np.array([0.,2.]).clip(a_min=0., a_max=1.) gives 

1910 # TypeError: 'a_max' is an invalid keyword argument for this function. 

1911 def _wrap_function(self, function, *args, unit=None, out=None, **kwargs): 

1912 """Wrap a numpy function that processes self, returning a Quantity. 

1913 

1914 Parameters 

1915 ----------

1917 Numpy function to wrap. 

1918 args : positional arguments 

1919 Any positional arguments to the function beyond the first argument 

1920 (which will be set to ``self``). 

1921 kwargs : keyword arguments 

1922 Keyword arguments to the function.

1925 

1926 unit : `~astropy.units.Unit` 

1927 Unit of the output result. If not given, the unit of ``self``. 

1928 out : `~astropy.units.Quantity` 

1929 A Quantity instance in which to store the output.
