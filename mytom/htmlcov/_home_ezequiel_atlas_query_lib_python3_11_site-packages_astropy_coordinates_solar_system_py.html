86class solar_system_ephemeris(ScienceState): 

87 """Default ephemerides for calculating positions of Solar-System bodies. 

88 

89 This can be one of the following: 

90 

91 - 'builtin': polynomial approximations to the orbital elements. 

92 - 'dexxx[s]', for a JPL dynamical model, where xxx is the three digit 

93 version number (e.g. de430), and the 's' is optional to specify the 

94 'small' version of a kernel. The version number must correspond to an 

95 ephemeris file available at: 

96 https://naif.jpl.nasa.gov/pub/naif/generic_kernels/spk/planets/ 

97 - 'jpl': Alias for the default JPL ephemeris (currently, 'de430'). 

98 - URL: (str) The url to a SPK ephemeris in SPICE binary (.bsp) format. 

99 - PATH: (str) File path to a SPK ephemeris in SPICE binary (.bsp) format. 

100 - `None`: Ensure an Exception is raised without an explicit ephemeris. 

101 

102 The default is 'builtin', which uses the ``epv00`` and ``plan94`` 

103 routines from the ``erfa`` implementation of the Standards Of Fundamental 

104 Astronomy library. 

105 

106 Notes 

107 ----- 

108 Any file required will be downloaded (and cached) when the state is set. 

109 The default Satellite Planet Kernel (SPK) file from NASA JPL (de430) is 

110 ~120MB, and covers years ~1550-2650 CE [1]_. The smaller de432s file is 

111 ~10MB, and covers years 1950-2050 [2]_ (and similarly for the newer de440 

112 and de440s). Older versions of the JPL ephemerides (such as the widely 

113 used de200) can be used via their URL [3]_. 

114 

115 .. [1] https://naif.jpl.nasa.gov/pub/naif/generic_kernels/spk/planets/aareadme_de430-de431.txt 

116 .. [2] https://naif.jpl.nasa.gov/pub/naif/generic_kernels/spk/planets/aareadme_de432s.txt 

117 .. [3] https://naif.jpl.nasa.gov/pub/naif/generic_kernels/spk/planets/a_old_versions/ 

118 """

126 if value is None:

132 @classmethod 

133 def get_kernel(cls, value): 

134 # ScienceState only ensures the `_value` attribute is up to date, 

135 # so we need to be sure any kernel returned is consistent. 

136 if cls._kernel is None or cls._kernel.origin != value: 

137 if cls._kernel is not None: 

138 cls._kernel.daf.file.close() 

139 cls._kernel = None 

140 kernel = _get_kernel(value) 

141 if kernel is not None: 

142 kernel.origin = value 

143 cls._kernel = kernel 

144 return cls._kernel

147 def kernel(cls): 

148 return cls.get_kernel(cls._value)

156 PLAN94_BODY_NAME_TO_PLANET_INDEX.keys()

159 return tuple(BODY_NAME_TO_KERNEL_SPEC.keys())

162def _get_kernel(value): 

163 """ 

164 Try importing jplephem, download/retrieve from cache the Satellite Planet 

165 Kernel corresponding to the given ephemeris. 

166 """ 

167 if value is None or value.lower() == "builtin": 

168 return None 

169 

170 try: 

171 from jplephem.spk import SPK 

172 except ImportError: 

173 raise ImportError( 

174 "Solar system JPL ephemeris calculations require the jplephem package " 

175 "(https://pypi.org/project/jplephem/)" 

176 ) 

177 

178 if value.lower() == "jpl": 

179 # Get the default JPL ephemeris URL 

180 value = DEFAULT_JPL_EPHEMERIS 

181 

182 if re.compile(r"de[0-9][0-9][0-9]s?").match(value.lower()): 

183 value = ( 

184 "https://naif.jpl.nasa.gov/pub/naif/generic_kernels" 

185 f"/spk/planets/{value.lower():s}.bsp" 

186 ) 

187 

188 elif os.path.isfile(value): 

189 return SPK.open(value) 

190 

191 else: 

192 try: 

193 urlparse(value) 

194 except Exception: 

195 raise ValueError( 

196 f"{value} was not one of the standard strings and " 

197 "could not be parsed as a file path or URL" 

198 ) 

199 

200 return SPK.open(download_file(value, cache=True))

203def _get_body_barycentric_posvel(body, time, ephemeris=None, get_velocity=True): 

204 """Calculate the barycentric position (and velocity) of a solar system body. 

205 

206 Parameters 

207 ---------- 

208 body : str or other 

209 The solar system body for which to calculate positions. Can also be a 

210 kernel specifier (list of 2-tuples) if the ``ephemeris`` is a JPL 

211 kernel. 

212 time : `~astropy.time.Time` 

213 Time of observation. 

214 ephemeris : str, optional 

215 Ephemeris to use. By default, use the one set with 

216 ``astropy.coordinates.solar_system_ephemeris.set`` 

217 get_velocity : bool, optional 

218 Whether or not to calculate the velocity as well as the position. 

219 

220 Returns 

221 ------- 

222 position : `~astropy.coordinates.CartesianRepresentation` or tuple 

223 Barycentric (ICRS) position or tuple of position and velocity. 

224 

225 Notes 

226 ----- 

227 Whether or not velocities are calculated makes little difference for the 

228 built-in ephemerides, but for most JPL ephemeris files, the execution time 

229 roughly doubles. 

230 """ 

231 # If the ephemeris is to be taken from solar_system_ephemeris, or the one 

232 # it already contains, use the kernel there. Otherwise, open the ephemeris, 

233 # possibly downloading it, but make sure the file is closed at the end. 

234 default_kernel = ephemeris is None or ephemeris is solar_system_ephemeris._value 

235 kernel = None 

236 try: 

237 if default_kernel: 

238 if solar_system_ephemeris.get() is None: 

239 raise ValueError(_EPHEMERIS_NOTE) 

240 kernel = solar_system_ephemeris.kernel 

241 else: 

242 kernel = _get_kernel(ephemeris) 

243 

244 jd1, jd2 = get_jd12(time, "tdb") 

245 if kernel is None: 

246 body = body.lower() 

247 earth_pv_helio, earth_pv_bary = erfa.epv00(jd1, jd2) 

248 if body == "earth": 

249 body_pv_bary = earth_pv_bary 

250 

251 elif body == "moon": 

252 # The moon98 documentation notes that it takes TT, but that TDB leads 

253 # to errors smaller than the uncertainties in the algorithm. 

254 # moon98 returns the astrometric position relative to the Earth. 

255 moon_pv_geo = erfa.moon98(jd1, jd2) 

256 body_pv_bary = erfa.pvppv(moon_pv_geo, earth_pv_bary) 

257 else: 

258 sun_pv_bary = erfa.pvmpv(earth_pv_bary, earth_pv_helio) 

259 if body == "sun": 

260 body_pv_bary = sun_pv_bary 

261 else: 

262 try: 

263 body_index = PLAN94_BODY_NAME_TO_PLANET_INDEX[body] 

264 except KeyError: 

265 raise KeyError( 

266 f"{body}'s position and velocity cannot be " 

267 f"calculated with the '{ephemeris}' ephemeris." 

268 ) 

269 body_pv_helio = erfa.plan94(jd1, jd2, body_index) 

270 body_pv_bary = erfa.pvppv(body_pv_helio, sun_pv_bary) 

271 

272 body_pos_bary = CartesianRepresentation( 

273 body_pv_bary["p"], unit=u.au, xyz_axis=-1, copy=False 

274 ) 

275 if get_velocity: 

276 body_vel_bary = CartesianRepresentation( 

277 body_pv_bary["v"], unit=u.au / u.day, xyz_axis=-1, copy=False 

278 ) 

279 

280 else: 

281 if isinstance(body, str): 

282 # Look up kernel chain for JPL ephemeris, based on name 

283 try: 

284 kernel_spec = BODY_NAME_TO_KERNEL_SPEC[body.lower()] 

285 except KeyError: 

286 raise KeyError( 

287 f"{body}'s position cannot be calculated with " 

288 f"the {ephemeris} ephemeris." 

289 ) 

290 else: 

291 # otherwise, assume the user knows what their doing and intentionally 

292 # passed in a kernel chain 

293 kernel_spec = body 

294 

295 # jplephem cannot handle multi-D arrays, so convert to 1D here. 

296 jd1_shape = getattr(jd1, "shape", ()) 

297 if len(jd1_shape) > 1: 

298 jd1, jd2 = jd1.ravel(), jd2.ravel() 

299 # Note that we use the new jd1.shape here to create a 1D result array. 

300 # It is reshaped below. 

301 body_posvel_bary = np.zeros( 

302 (2 if get_velocity else 1, 3) + getattr(jd1, "shape", ()) 

303 ) 

304 for pair in kernel_spec: 

305 spk = kernel[pair] 

306 if spk.data_type == 3: 

307 # Type 3 kernels contain both position and velocity. 

308 posvel = spk.compute(jd1, jd2) 

309 if get_velocity: 

310 body_posvel_bary += posvel.reshape(body_posvel_bary.shape) 

311 else: 

312 body_posvel_bary[0] += posvel[:4] 

313 else: 

314 # spk.generate first yields the position and then the 

315 # derivative. If no velocities are desired, body_posvel_bary 

316 # has only one element and thus the loop ends after a single 

317 # iteration, avoiding the velocity calculation. 

318 for body_p_or_v, p_or_v in zip( 

319 body_posvel_bary, spk.generate(jd1, jd2) 

320 ): 

321 body_p_or_v += p_or_v 

322 

323 body_posvel_bary.shape = body_posvel_bary.shape[:2] + jd1_shape 

324 body_pos_bary = CartesianRepresentation( 

325 body_posvel_bary[0], unit=u.km, copy=False 

326 ) 

327 if get_velocity: 

328 body_vel_bary = CartesianRepresentation( 

329 body_posvel_bary[1], unit=u.km / u.day, copy=False 

330 ) 

331 

332 return (body_pos_bary, body_vel_bary) if get_velocity else body_pos_bary 

333 

334 finally: 

335 if not default_kernel and kernel is not None: 

336 kernel.daf.file.close()

403def _get_apparent_body_position(body, time, ephemeris, obsgeoloc=None): 

404 """Calculate the apparent position of body ``body`` relative to Earth. 

405 

406 This corrects for the light-travel time to the object. 

407 

408 Parameters 

409 ---------- 

410 body : str or other 

411 The solar system body for which to calculate positions. Can also be a 

412 kernel specifier (list of 2-tuples) if the ``ephemeris`` is a JPL 

413 kernel. 

414 time : `~astropy.time.Time` 

415 Time of observation. 

416 ephemeris : str, optional 

417 Ephemeris to use. By default, use the one set with 

418 ``~astropy.coordinates.solar_system_ephemeris.set`` 

419 obsgeoloc : `~astropy.coordinates.CartesianRepresentation`, optional 

420 The GCRS position of the observer 

421 

422 Returns 

423 ------- 

424 cartesian_position : `~astropy.coordinates.CartesianRepresentation` 

425 Barycentric (ICRS) apparent position of the body in cartesian coordinates 

426 

427 Notes 

428 ----- 

429 {_EPHEMERIS_NOTE} 

430 """ 

431 if ephemeris is None: 

432 ephemeris = solar_system_ephemeris.get() 

433 

434 # Calculate position given approximate light travel time. 

435 delta_light_travel_time = 20.0 * u.s 

436 emitted_time = time 

437 light_travel_time = 0.0 * u.s 

438 earth_loc = get_body_barycentric("earth", time, ephemeris) 

439 if obsgeoloc is not None: 

440 earth_loc += obsgeoloc 

441 while np.any(np.fabs(delta_light_travel_time) > 1.0e-8 * u.s): 

442 body_loc = get_body_barycentric(body, emitted_time, ephemeris) 

443 earth_distance = (body_loc - earth_loc).norm() 

444 delta_light_travel_time = light_travel_time - earth_distance / speed_of_light 

445 light_travel_time = earth_distance / speed_of_light 

446 emitted_time = time - light_travel_time 

447 

448 return get_body_barycentric(body, emitted_time, ephemeris)

451def get_body(body, time, location=None, ephemeris=None): 

452 """ 

453 Get a `~astropy.coordinates.SkyCoord` for a solar system body as observed 

454 from a location on Earth in the `~astropy.coordinates.GCRS` reference 

455 system. 

456 

457 Parameters 

458 ---------- 

459 body : str or list of tuple 

460 The solar system body for which to calculate positions. Can also be a 

461 kernel specifier (list of 2-tuples) if the ``ephemeris`` is a JPL 

462 kernel. 

463 time : `~astropy.time.Time` 

464 Time of observation. 

465 location : `~astropy.coordinates.EarthLocation`, optional 

466 Location of observer on the Earth. If not given, will be taken from 

467 ``time`` (if not present, a geocentric observer will be assumed). 

468 ephemeris : str, optional 

469 Ephemeris to use. If not given, use the one set with 

470 ``astropy.coordinates.solar_system_ephemeris.set`` (which is 

471 set to 'builtin' by default). 

472 

473 Returns 

474 ------- 

475 skycoord : `~astropy.coordinates.SkyCoord` 

476 GCRS Coordinate for the body 

477 

478 Notes 

479 ----- 

480 The coordinate returned is the apparent position, which is the position of 

481 the body at time *t* minus the light travel time from the *body* to the 

482 observing *location*. 

483 

484 {_EPHEMERIS_NOTE} 

485 """ 

486 if location is None: 

487 location = time.location 

488 

489 if location is not None: 

490 obsgeoloc, obsgeovel = location.get_gcrs_posvel(time) 

491 else: 

492 obsgeoloc, obsgeovel = None, None 

493 

494 cartrep = _get_apparent_body_position(body, time, ephemeris, obsgeoloc) 

495 icrs = ICRS(cartrep) 

496 gcrs = icrs.transform_to( 

497 GCRS(obstime=time, obsgeoloc=obsgeoloc, obsgeovel=obsgeovel) 

498 ) 

499 

500 return SkyCoord(gcrs)
