37def cartesian_to_spherical(x, y, z): 

38 """ 

39 Converts 3D rectangular cartesian coordinates to spherical polar 

40 coordinates. 

41 

42 Note that the resulting angles are latitude/longitude or 

43 elevation/azimuthal form. I.e., the origin is along the equator 

44 rather than at the north pole. 

45 

46 .. note:: 

47 This function simply wraps functionality provided by the 

48 `~astropy.coordinates.CartesianRepresentation` and 

49 `~astropy.coordinates.SphericalRepresentation` classes. In general, 

50 for both performance and readability, we suggest using these classes 

51 directly. But for situations where a quick one-off conversion makes 

52 sense, this function is provided. 

53 

54 Parameters 

55 ---------- 

56 x : scalar, array-like, or `~astropy.units.Quantity` 

57 The first Cartesian coordinate. 

58 y : scalar, array-like, or `~astropy.units.Quantity` 

59 The second Cartesian coordinate. 

60 z : scalar, array-like, or `~astropy.units.Quantity` 

61 The third Cartesian coordinate. 

62 

63 Returns 

64 ------- 

65 r : `~astropy.units.Quantity` 

66 The radial coordinate (in the same units as the inputs). 

67 lat : `~astropy.units.Quantity` ['angle'] 

68 The latitude in radians 

69 lon : `~astropy.units.Quantity` ['angle'] 

70 The longitude in radians 

71 """ 

72 if not hasattr(x, "unit"): 

73 x = x * u.dimensionless_unscaled 

74 if not hasattr(y, "unit"): 

75 y = y * u.dimensionless_unscaled 

76 if not hasattr(z, "unit"): 

77 z = z * u.dimensionless_unscaled 

78 

79 cart = CartesianRepresentation(x, y, z) 

80 sph = cart.represent_as(SphericalRepresentation) 

81 

82 return sph.distance, sph.lat, sph.lon

85def spherical_to_cartesian(r, lat, lon): 

86 """ 

87 Converts spherical polar coordinates to rectangular cartesian 

88 coordinates. 

89 

90 Note that the input angles should be in latitude/longitude or 

91 elevation/azimuthal form. I.e., the origin is along the equator 

92 rather than at the north pole. 

93 

94 .. note:: 

95 This is a low-level function used internally in 

96 `astropy.coordinates`. It is provided for users if they really 

97 want to use it, but it is recommended that you use the 

98 `astropy.coordinates` coordinate systems. 

99 

100 Parameters 

101 ---------- 

102 r : scalar, array-like, or `~astropy.units.Quantity` 

103 The radial coordinate (in the same units as the inputs). 

104 lat : scalar, array-like, or `~astropy.units.Quantity` ['angle'] 

105 The latitude (in radians if array or scalar) 

106 lon : scalar, array-like, or `~astropy.units.Quantity` ['angle'] 

107 The longitude (in radians if array or scalar) 

108 

109 Returns 

110 ------- 

111 x : float or array 

112 The first cartesian coordinate. 

113 y : float or array 

114 The second cartesian coordinate. 

115 z : float or array 

116 The third cartesian coordinate. 

117 

118 """ 

119 if not hasattr(r, "unit"): 

120 r = r * u.dimensionless_unscaled 

121 if not hasattr(lat, "unit"): 

122 lat = lat * u.radian 

123 if not hasattr(lon, "unit"): 

124 lon = lon * u.radian 

125 

126 sph = SphericalRepresentation(distance=r, lat=lat, lon=lon) 

127 cart = sph.represent_as(CartesianRepresentation) 

128 

129 return cart.x, cart.y, cart.z

132def get_sun(time): 

133 """ 

134 Determines the location of the sun at a given time (or times, if the input 

135 is an array `~astropy.time.Time` object), in geocentric coordinates. 

136 

137 Parameters 

138 ---------- 

139 time : `~astropy.time.Time` 

140 The time(s) at which to compute the location of the sun. 

141 

142 Returns 

143 ------- 

144 newsc : `~astropy.coordinates.SkyCoord` 

145 The location of the sun as a `~astropy.coordinates.SkyCoord` in the 

146 `~astropy.coordinates.GCRS` frame. 

147 

148 Notes 

149 ----- 

150 The algorithm for determining the sun/earth relative position is based 

151 on the simplified version of VSOP2000 that is part of ERFA. Compared to 

152 JPL's ephemeris, it should be good to about 4 km (in the Sun-Earth 

153 vector) from 1900-2100 C.E., 8 km for the 1800-2200 span, and perhaps 

154 250 km over the 1000-3000. 

155 

156 """ 

157 earth_pv_helio, earth_pv_bary = erfa.epv00(*get_jd12(time, "tdb")) 

158 

159 # We have to manually do aberration because we're outputting directly into 

160 # GCRS 

161 earth_p = earth_pv_helio["p"] 

162 earth_v = earth_pv_bary["v"] 

163 

164 # convert barycentric velocity to units of c, but keep as array for passing in to erfa 

165 earth_v /= c.to_value(u.au / u.d) 

166 

167 dsun = np.sqrt(np.sum(earth_p**2, axis=-1)) 

168 invlorentz = (1 - np.sum(earth_v**2, axis=-1)) ** 0.5 

169 properdir = erfa.ab( 

170 earth_p / dsun.reshape(dsun.shape + (1,)), -earth_v, dsun, invlorentz 

171 ) 

172 

173 cartrep = CartesianRepresentation( 

174 x=-dsun * properdir[..., 0] * u.AU, 

175 y=-dsun * properdir[..., 1] * u.AU, 

176 z=-dsun * properdir[..., 2] * u.AU, 

177 ) 

178 return SkyCoord(cartrep, frame=GCRS(obstime=time))

185def get_constellation(coord, short_name=False, constellation_list="iau"): 

186 """ 

187 Determines the constellation(s) a given coordinate object contains. 

188 

189 Parameters 

190 ---------- 

191 coord : coordinate-like 

192 The object to determine the constellation of. 

193 short_name : bool 

194 If True, the returned names are the IAU-sanctioned abbreviated 

195 names. Otherwise, full names for the constellations are used. 

196 constellation_list : str 

197 The set of constellations to use. Currently only ``'iau'`` is 

198 supported, meaning the 88 "modern" constellations endorsed by the IAU. 

199 

200 Returns 

201 ------- 

202 constellation : str or string array 

203 If ``coords`` contains a scalar coordinate, returns the name of the 

204 constellation. If it is an array coordinate object, it returns an array 

205 of names. 

206 

207 Notes 

208 ----- 

209 To determine which constellation a point on the sky is in, this precesses 

210 to B1875, and then uses the Delporte boundaries of the 88 modern 

211 constellations, as tabulated by 

212 `Roman 1987 <https://cdsarc.cds.unistra.fr/viz-bin/cat/VI/42>`_. 

213 """ 

214 if constellation_list != "iau": 

215 raise ValueError("only 'iau' us currently supported for constellation_list") 

216 

217 # read the data files and cache them if they haven't been already 

218 if not _constellation_data: 

219 cdata = data.get_pkg_data_contents("data/constellation_data_roman87.dat") 

220 ctable = ascii.read(cdata, names=["ral", "rau", "decl", "name"]) 

221 cnames = data.get_pkg_data_contents( 

222 "data/constellation_names.dat", encoding="UTF8" 

223 ) 

224 cnames_short_to_long = { 

225 l[:3]: l[4:] for l in cnames.split("\n") if not l.startswith("#") 

226 } 

227 cnames_long = np.array([cnames_short_to_long[nm] for nm in ctable["name"]]) 

228 

229 _constellation_data["ctable"] = ctable 

230 _constellation_data["cnames_long"] = cnames_long 

231 else: 

232 ctable = _constellation_data["ctable"] 

233 cnames_long = _constellation_data["cnames_long"] 

234 

235 isscalar = coord.isscalar 

236 

237 # if it is geocentric, we reproduce the frame but with the 1875 equinox, 

238 # which is where the constellations are defined 

239 # this yields a "dubious year" warning because ERFA considers the year 1875 

240 # "dubious", probably because UTC isn't well-defined then and precession 

241 # models aren't precisely calibrated back to then. But it's plenty 

242 # sufficient for constellations 

243 with warnings.catch_warnings(): 

244 warnings.simplefilter("ignore", erfa.ErfaWarning) 

245 constel_coord = coord.transform_to(PrecessedGeocentric(equinox="B1875")) 

246 if isscalar: 

247 rah = constel_coord.ra.ravel().hour 

248 decd = constel_coord.dec.ravel().deg 

249 else: 

250 rah = constel_coord.ra.hour 

251 decd = constel_coord.dec.deg 

252 

253 constellidx = -np.ones(len(rah), dtype=int) 

254 

255 notided = constellidx == -1 # should be all 

256 for i, row in enumerate(ctable): 

257 msk = (row["ral"] < rah) & (rah < row["rau"]) & (decd > row["decl"]) 

258 constellidx[notided & msk] = i 

259 notided = constellidx == -1 

260 if np.sum(notided) == 0: 

261 break 

262 else: 

263 raise ValueError( 

264 f"Could not find constellation for coordinates {constel_coord[notided]}" 

265 ) 

266 

267 if short_name: 

268 names = ctable["name"][constellidx] 

269 else: 

270 names = cnames_long[constellidx] 

271 

272 if isscalar: 

273 return names[0] 

274 else: 

275 return names

278def _concatenate_components(reps_difs, names): 

279 """Helper function for the concatenate function below. Gets and 

280 concatenates all of the individual components for an iterable of 

281 representations or differentials. 

282 """ 

283 values = [] 

284 for name in names: 

285 unit0 = getattr(reps_difs[0], name).unit 

286 # Go via to_value because np.concatenate doesn't work with Quantity 

287 data_vals = [getattr(x, name).to_value(unit0) for x in reps_difs] 

288 concat_vals = np.concatenate(np.atleast_1d(*data_vals)) 

289 concat_vals = concat_vals << unit0 

290 values.append(concat_vals) 

291 

292 return values

295def concatenate_representations(reps): 

296 """ 

297 Combine multiple representation objects into a single instance by 

298 concatenating the data in each component. 

299 

300 Currently, all of the input representations have to be the same type. This 

301 properly handles differential or velocity data, but all input objects must 

302 have the same differential object type as well. 

303 

304 Parameters 

305 ---------- 

306 reps : sequence of `~astropy.coordinates.BaseRepresentation` 

307 The objects to concatenate 

308 

309 Returns 

310 ------- 

311 rep : `~astropy.coordinates.BaseRepresentation` subclass instance 

312 A single representation object with its data set to the concatenation of 

313 all the elements of the input sequence of representations. 

314 

315 """ 

316 if not isinstance(reps, (Sequence, np.ndarray)): 

317 raise TypeError("Input must be a list or iterable of representation objects.") 

318 

319 # First, validate that the representations are the same, and 

320 # concatenate all of the positional data: 

321 rep_type = type(reps[0]) 

322 if any(type(r) != rep_type for r in reps): 

323 raise TypeError("Input representations must all have the same type.") 

324 

325 # Construct the new representation with the concatenated data from the 

326 # representations passed in 

327 values = _concatenate_components(reps, rep_type.attr_classes.keys()) 

328 new_rep = rep_type(*values) 

329 

330 has_diff = any("s" in rep.differentials for rep in reps) 

331 if has_diff and any("s" not in rep.differentials for rep in reps): 

332 raise ValueError( 

333 "Input representations must either all contain " 

334 "differentials, or not contain differentials." 

335 ) 

336 

337 if has_diff: 

338 dif_type = type(reps[0].differentials["s"]) 

339 

340 if any( 

341 "s" not in r.differentials or type(r.differentials["s"]) != dif_type 

342 for r in reps 

343 ): 

344 raise TypeError( 

345 "All input representations must have the same differential type." 

346 ) 

347 

348 values = _concatenate_components( 

349 [r.differentials["s"] for r in reps], dif_type.attr_classes.keys() 

350 ) 

351 new_dif = dif_type(*values) 

352 new_rep = new_rep.with_differentials({"s": new_dif}) 

353 

354 return new_rep

357def concatenate(coords): 

358 """ 

359 Combine multiple coordinate objects into a single 

360 `~astropy.coordinates.SkyCoord`. 

361 

362 "Coordinate objects" here mean frame objects with data, 

363 `~astropy.coordinates.SkyCoord`, or representation objects. Currently, 

364 they must all be in the same frame, but in a future version this may be 

365 relaxed to allow inhomogeneous sequences of objects. 

366 

367 Parameters 

368 ---------- 

369 coords : sequence of coordinate-like 

370 The objects to concatenate 

371 

372 Returns 

373 ------- 

374 cskycoord : SkyCoord 

375 A single sky coordinate with its data set to the concatenation of all 

376 the elements in ``coords`` 

377 """ 

378 if getattr(coords, "isscalar", False) or not isiterable(coords): 

379 raise TypeError("The argument to concatenate must be iterable") 

380 

381 scs = [SkyCoord(coord, copy=False) for coord in coords] 

382 

383 # Check that all frames are equivalent 

384 for sc in scs[1:]: 

385 if not sc.is_equivalent_frame(scs[0]): 

386 raise ValueError( 

387 f"All inputs must have equivalent frames: {sc} != {scs[0]}" 

388 ) 

389 

390 # TODO: this can be changed to SkyCoord.from_representation() for a speed 

391 # boost when we switch to using classmethods 

392 return SkyCoord( 

393 concatenate_representations([c.data for c in coords]), frame=scs[0].frame 

394 )
