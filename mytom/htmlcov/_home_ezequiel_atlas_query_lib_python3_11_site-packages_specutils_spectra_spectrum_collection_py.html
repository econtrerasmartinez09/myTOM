14class SpectrumCollection(NDIOMixin): 

15 """ 

16 A class to represent a heterogeneous set of spectra that are the same length 

17 but have different spectral axes. Spectra that meet this requirement can be 

18 stored as multidimensional arrays, and thus can have operations performed 

19 on them faster than if they are treated as individual 

20 :class:`~specutils.Spectrum1D` objects. 

21 

22 For multidimensional spectra that all have the *same* spectral axis, use a 

23 :class:`~specutils.Spectrum1D` with dimension greater than 1. For a 

24 collection of spectra that have different shapes, use 

25 :class:`~specutils.SpectrumList`. For more on this topic, see 

26 :ref:`specutils-representation-overview`. 

27 

28 The attributes on this class uses the same names and conventions as 

29 :class:`~specutils.Spectrum1D`, allowing some operations to work the same. 

30 Where this does not work, the user can use standard indexing notation to 

31 access individual :class:`~specutils.Spectrum1D` objects. 

32 

33 Parameters 

34 ---------- 

35 flux : :class:`astropy.units.Quantity` 

36 The flux data. The trailing dimension should be the spectral dimension. 

37 spectral_axis : :class:`astropy.units.Quantity` 

38 The spectral axes of the spectra (e.g., wavelength). Must match the 

39 dimensionality of ``flux``. 

40 wcs : wcs object or None 

41 A wcs object (if available) for the collection of spectra. The object 

42 must follow standard indexing rules to get a sub-wcs if it is provided. 

43 uncertainty : :class:`astropy.nddata.NDUncertainty` or ndarray 

44 The uncertainties associated with each spectrum of the collection. In 

45 the case that only an n-dimensional quantity or ndaray is provided, 

46 the uncertainties are assumed to be standard deviations. Must match the 

47 dimensionality of ``flux``. 

48 mask : ndarray or None 

49 The n-dimensional mask information associated with each spectrum. If 

50 present, must match the dimensionality of ``flux``. 

51 meta : list 

52 The list of dictionaries containing meta data to be associated with 

53 each spectrum in the collection. 

54 """ 

55 def __init__(self, flux, spectral_axis=None, wcs=None, uncertainty=None, 

56 mask=None, meta=None): 

57 # Check for quantity 

58 if not isinstance(flux, u.Quantity): 

59 raise u.UnitsError("Flux must be a `Quantity`.") 

60 

61 if spectral_axis is not None: 

62 if not isinstance(spectral_axis, u.Quantity): 

63 raise u.UnitsError("Spectral axis must be a `Quantity`.") 

64 spectral_axis = SpectralCoord(spectral_axis) 

65 

66 # Ensure that the input values are the same shape 

67 if not (flux.shape == spectral_axis.shape): 

68 raise ValueError("Shape of all data elements must be the same.") 

69 

70 if uncertainty is not None and uncertainty.array.shape != flux.shape: 

71 raise ValueError("Uncertainty must be the same shape as flux and " 

72 "spectral axis.") 

73 

74 if mask is not None and mask.shape != flux.shape: 

75 raise ValueError("Mask must be the same shape as flux and " 

76 "spectral axis.") 

77 

78 # Convert uncertainties to standard deviations if not already defined 

79 # to be of some type 

80 if uncertainty is not None and not isinstance(uncertainty, NDUncertainty): 

81 # If the uncertainties are not provided a unit, raise a warning 

82 # and use the flux units 

83 if not isinstance(uncertainty, u.Quantity): 

84 warnings.warn("No unit associated with uncertainty, assuming" 

85 f"flux units of '{flux.unit}'.") 

86 uncertainty = u.Quantity(uncertainty, unit=flux.unit) 

87 

88 uncertainty = StdDevUncertainty(uncertainty) 

89 

90 self._flux = flux 

91 self._spectral_axis = spectral_axis 

92 self._wcs = wcs 

93 self._uncertainty = uncertainty 

94 self._mask = mask 

95 self._meta = meta 

96 

97 def __getitem__(self, key): 

98 flux = self.flux[key] 

99 if flux.ndim != 1: 

100 raise ValueError("Currently only 1D data structures may be " 

101 "returned from slice operations.") 

102 spectral_axis = self.spectral_axis[key] 

103 uncertainty = None if self.uncertainty is None else self.uncertainty[key] 

104 wcs = None if self.wcs is None else self.wcs[key] 

105 mask = None if self.mask is None else self.mask[key] 

106 if self.meta is None: 

107 meta = None 

108 else: 

109 try: 

110 meta = self.meta[key] 

111 except KeyError: 

112 meta = self.meta 

113 

114 return Spectrum1D(flux=flux, spectral_axis=spectral_axis, 

115 uncertainty=uncertainty, wcs=wcs, mask=mask, 

116 meta=meta)

119 def from_spectra(cls, spectra): 

120 """ 

121 Create a spectrum collection from a set of individual 

122 :class:`specutils.Spectrum1D` objects. 

123 

124 Parameters 

125 ---------- 

126 spectra : list, ndarray 

127 A list of :class:`~specutils.Spectrum1D` objects to be held in the 

128 collection. Currently the spectral_axis parameters (e.g. observer, 

129 radial_velocity) must be the same for each spectrum. 

130 """ 

131 # Enforce that the shape of each item must be the same 

132 if not all((x.shape == spectra[0].shape for x in spectra)): 

133 raise ValueError("Shape of all elements must be the same.") 

134 

135 # Compose multi-dimensional ndarrays for each property 

136 flux = u.Quantity([spec.flux for spec in spectra]) 

137 

138 # Check that the spectral parameters are the same for each input 

139 # spectral_axis and create the multi-dimensional SpectralCoord 

140 sa = [x.spectral_axis for x in spectra] 

141 if (not all(x.radial_velocity == sa[0].radial_velocity for x in sa) or 

142 not all(x.target == sa[0].target for x in sa) or 

143 not all(x.observer == sa[0].observer for x in sa) or 

144 not all(x.doppler_convention == sa[0].doppler_convention for 

145 x in sa) or 

146 not all(x.doppler_rest == sa[0].doppler_rest for x in sa)): 

147 raise ValueError("All input spectral_axis SpectralCoord " 

148 "objects must have the same parameters.") 

149 spectral_axis = SpectralCoord(sa, 

150 radial_velocity=sa[0].radial_velocity, 

151 doppler_rest=sa[0].doppler_rest, 

152 doppler_convention=sa[0].doppler_convention, 

153 observer=sa[0].observer, 

154 target=sa[0].target) 

155 

156 # Check that either all spectra have associated uncertainties, or that 

157 # none of them do. If only some do, log an error and ignore the 

158 # uncertainties. 

159 if (not all((x.uncertainty is None for x in spectra)) and 

160 any((x.uncertainty is not None for x in spectra)) and 

161 all((x.uncertainty.uncertainty_type == 

162 spectra[0].uncertainty.uncertainty_type 

163 for x in spectra))): 

164 quncs = u.Quantity([spec.uncertainty.quantity for spec in spectra]) 

165 uncertainty = spectra[0].uncertainty.__class__(quncs) 

166 else: 

167 uncertainty = None 

168 warnings.warn("Not all spectra have associated uncertainties of " 

169 "the same type, skipping uncertainties.") 

170 

171 # Check that either all spectra have associated masks, or that 

172 # none of them do. If only some do, log an error and ignore the masks. 

173 if (not all((x.mask is None for x in spectra)) and 

174 any((x.mask is not None for x in spectra))): 

175 mask = np.array([spec.mask for spec in spectra]) 

176 else: 

177 mask = None 

178 warnings.warn("Not all spectra have associated masks, " 

179 "skipping masks.") 

180 

181 # Store the wcs and meta as lists 

182 wcs = [spec.wcs for spec in spectra] 

183 meta = [spec.meta for spec in spectra] 

184 

185 return cls(flux=flux, spectral_axis=spectral_axis, 

186 uncertainty=uncertainty, wcs=wcs, mask=mask, meta=meta)
