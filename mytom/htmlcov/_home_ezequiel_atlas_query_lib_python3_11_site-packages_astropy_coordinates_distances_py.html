23class Distance(u.SpecificTypeQuantity): 

24 """ 

25 A one-dimensional distance. 

26 

27 This can be initialized by providing one of the following: 

28 

29 * Distance ``value`` (array or float) and a ``unit`` 

30 * |Quantity| object with dimensionality of length 

31 * Redshift and (optionally) a `~astropy.cosmology.Cosmology` 

32 * Distance modulus 

33 * Parallax 

34 

35 Parameters 

36 ---------- 

37 value : scalar or `~astropy.units.Quantity` ['length'] 

38 The value of this distance. 

39 unit : `~astropy.units.UnitBase` ['length'] 

40 The unit for this distance. 

41 z : float 

42 A redshift for this distance. It will be converted to a distance 

43 by computing the luminosity distance for this redshift given the 

44 cosmology specified by ``cosmology``. Must be given as a keyword 

45 argument. 

46 cosmology : `~astropy.cosmology.Cosmology` or None 

47 A cosmology that will be used to compute the distance from ``z``. 

48 If `None`, the current cosmology will be used (see 

49 `astropy.cosmology` for details). 

50 distmod : float or `~astropy.units.Quantity` 

51 The distance modulus for this distance. Note that if ``unit`` is not 

52 provided, a guess will be made at the unit between AU, pc, kpc, and Mpc. 

53 parallax : `~astropy.units.Quantity` or `~astropy.coordinates.Angle` 

54 The parallax in angular units. 

55 dtype : `~numpy.dtype`, optional 

56 See `~astropy.units.Quantity`. 

57 copy : bool, optional 

58 See `~astropy.units.Quantity`. 

59 order : {'C', 'F', 'A'}, optional 

60 See `~astropy.units.Quantity`. 

61 subok : bool, optional 

62 See `~astropy.units.Quantity`. 

63 ndmin : int, optional 

64 See `~astropy.units.Quantity`. 

65 allow_negative : bool, optional 

66 Whether to allow negative distances (which are possible in some 

67 cosmologies). Default: `False`. 

68 

69 Raises 

70 ------ 

71 `~astropy.units.UnitsError` 

72 If the ``unit`` is not a length unit. 

73 ValueError 

74 If value specified is less than 0 and ``allow_negative=False``. 

75 

76 If ``cosmology`` is provided when ``z`` is *not* given. 

77 

78 If either none or more than one of ``value``, ``z``, ``distmod``, 

79 or ``parallax`` were given. 

80 

81 

82 Examples 

83 -------- 

84 >>> from astropy import units as u 

85 >>> from astropy.cosmology import WMAP5 

86 >>> Distance(10, u.Mpc) 

87 <Distance 10. Mpc> 

88 >>> Distance(40*u.pc, unit=u.kpc) 

89 <Distance 0.04 kpc> 

90 >>> Distance(z=0.23) # doctest: +FLOAT_CMP 

91 <Distance 1184.01657566 Mpc> 

92 >>> Distance(z=0.23, cosmology=WMAP5) # doctest: +FLOAT_CMP 

93 <Distance 1147.78831918 Mpc> 

94 >>> Distance(distmod=24.47*u.mag) # doctest: +FLOAT_CMP 

95 <Distance 783.42964277 kpc> 

96 >>> Distance(parallax=21.34*u.mas) # doctest: +FLOAT_CMP 

97 <Distance 46.86035614 pc> 

98 """

103 def __new__( 

104 cls, 

105 value=None, 

106 unit=None, 

107 z=None, 

108 cosmology=None, 

109 distmod=None, 

110 parallax=None, 

111 dtype=np.inexact, 

112 copy=True, 

113 order=None, 

114 subok=False, 

115 ndmin=0, 

116 allow_negative=False, 

117 ): 

118 n_not_none = sum(x is not None for x in [value, z, distmod, parallax]) 

119 if n_not_none == 0: 

120 raise ValueError( 

121 "none of `value`, `z`, `distmod`, or `parallax` " 

122 "were given to Distance constructor" 

123 ) 

124 elif n_not_none > 1: 

125 raise ValueError( 

126 "more than one of `value`, `z`, `distmod`, or " 

127 "`parallax` were given to Distance constructor" 

128 ) 

129 

130 if value is None: 

131 # If something else but `value` was provided then a new array will 

132 # be created anyways and there is no need to copy that. 

133 copy = False 

134 

135 if z is not None: 

136 if cosmology is None: 

137 from astropy.cosmology import default_cosmology 

138 

139 cosmology = default_cosmology.get() 

140 

141 value = cosmology.luminosity_distance(z) 

142 

143 elif cosmology is not None: 

144 raise ValueError( 

145 "a `cosmology` was given but `z` was not " 

146 "provided in Distance constructor" 

147 ) 

148 

149 elif distmod is not None: 

150 value = cls._distmod_to_pc(distmod) 

151 if unit is None: 

152 # if the unit is not specified, guess based on the mean of 

153 # the log of the distance 

154 meanlogval = np.log10(value.value).mean() 

155 if meanlogval > 6: 

156 unit = u.Mpc 

157 elif meanlogval > 3: 

158 unit = u.kpc 

159 elif meanlogval < -3: # ~200 AU 

160 unit = u.AU 

161 else: 

162 unit = u.pc 

163 

164 elif parallax is not None: 

165 if unit is None: 

166 unit = u.pc 

167 value = parallax.to_value(unit, equivalencies=u.parallax()) 

168 

169 if np.any(parallax < 0): 

170 if allow_negative: 

171 warnings.warn( 

172 "negative parallaxes are converted to NaN distances even when" 

173 " `allow_negative=True`, because negative parallaxes cannot be" 

174 " transformed into distances. See the discussion in this paper:" 

175 " https://arxiv.org/abs/1507.02105", 

176 AstropyWarning, 

177 ) 

178 else: 

179 raise ValueError( 

180 "some parallaxes are negative, which are not " 

181 "interpretable as distances. See the discussion in " 

182 "this paper: https://arxiv.org/abs/1507.02105 . You " 

183 "can convert negative parallaxes to NaN distances by " 

184 "providing the `allow_negative=True` argument." 

185 ) 

186 

187 # now we have arguments like for a Quantity, so let it do the work 

188 distance = super().__new__( 

189 cls, 

190 value, 

191 unit, 

192 dtype=dtype, 

193 copy=copy, 

194 order=order, 

195 subok=subok, 

196 ndmin=ndmin, 

197 ) 

198 

199 if not allow_negative and np.any(distance.value < 0): 

200 raise ValueError( 

201 "distance must be >= 0. Use the argument " 

202 "`allow_negative=True` to allow negative values." 

203 ) 

204 

205 return distance

212 def compute_z(self, cosmology=None, **atzkw): 

213 """ 

214 The redshift for this distance assuming its physical distance is 

215 a luminosity distance. 

216 

217 Parameters 

218 ---------- 

219 cosmology : `~astropy.cosmology.Cosmology` or None 

220 The cosmology to assume for this calculation, or `None` to use the 

221 current cosmology (see `astropy.cosmology` for details). 

222 **atzkw 

223 keyword arguments for :func:`~astropy.cosmology.z_at_value` 

224 

225 Returns 

226 ------- 

227 z : `~astropy.units.Quantity` 

228 The redshift of this distance given the provided ``cosmology``. 

229 

230 Warnings 

231 -------- 

232 This method can be slow for large arrays. 

233 The redshift is determined using :func:`astropy.cosmology.z_at_value`, 

234 which handles vector inputs (e.g. an array of distances) by 

235 element-wise calling of :func:`scipy.optimize.minimize_scalar`. 

236 For faster results consider using an interpolation table; 

237 :func:`astropy.cosmology.z_at_value` provides details. 

238 

239 See Also 

240 -------- 

241 :func:`astropy.cosmology.z_at_value` : Find the redshift corresponding to a 

242 :meth:`astropy.cosmology.FLRW.luminosity_distance`. 

243 """ 

244 from astropy.cosmology import z_at_value 

245 

246 if cosmology is None: 

247 from astropy.cosmology import default_cosmology 

248 

249 cosmology = default_cosmology.get() 

250 

251 atzkw.setdefault("ztol", 1.0e-10) 

252 return z_at_value(cosmology.luminosity_distance, self, **atzkw)

254 @property 

255 def distmod(self): 

256 """The distance modulus as a `~astropy.units.Quantity`.""" 

257 val = 5.0 * np.log10(self.to_value(u.pc)) - 5.0 

258 return u.Quantity(val, u.mag, copy=False) 

259 

260 @classmethod 

261 def _distmod_to_pc(cls, dm): 

262 dm = u.Quantity(dm, u.mag) 

263 return cls(10 ** ((dm.value + 5) / 5.0), u.pc, copy=False)

268 return Angle(self.to(u.milliarcsecond, u.parallax()))
