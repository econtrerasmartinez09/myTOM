150def _replace_temperatures_with_kelvin(unit): 

151 """ 

152 If a unit contains a temperature unit besides kelvin, then replace 

153 that unit with kelvin. 

154 

155 Temperatures cannot be converted directly between K, °F, °C, and 

156 °Ra, in particular since there would be different conversions for 

157 T and ΔT. However, each of these temperatures each represents the 

158 physical type. Replacing the different temperature units with 

159 kelvin allows the physical type to be treated consistently. 

160 """ 

161 physical_type_id = unit._get_physical_type_id() 

162 

163 physical_type_id_components = [] 

164 substitution_was_made = False 

165 

166 for base, power in physical_type_id: 

167 if base in ["deg_F", "deg_C", "deg_R"]: 

168 base = "K" 

169 substitution_was_made = True 

170 physical_type_id_components.append((base, power)) 

171 

172 if substitution_was_made: 

173 return core.Unit._from_physical_type_id(tuple(physical_type_id_components)) 

174 else: 

175 return unit

178def _standardize_physical_type_names(physical_type_input): 

179 """ 

180 Convert a string or `set` of strings into a `set` containing 

181 string representations of physical types. 

182 

183 The strings provided in ``physical_type_input`` can each contain 

184 multiple physical types that are separated by a regular slash. 

185 Underscores are treated as spaces so that variable names could 

186 be identical to physical type names. 

187 """ 

188 if isinstance(physical_type_input, str): 

189 physical_type_input = {physical_type_input} 

190 

191 standardized_physical_types = set() 

192 

193 for ptype_input in physical_type_input: 

194 if not isinstance(ptype_input, str): 

195 raise ValueError(f"expecting a string, but got {ptype_input}") 

196 input_set = set(ptype_input.split("/")) 

197 processed_set = {s.strip().replace("_", " ") for s in input_set} 

198 standardized_physical_types |= processed_set 

199 

200 return standardized_physical_types

203class PhysicalType: 

204 """ 

205 Represents the physical type(s) that are dimensionally compatible 

206 with a set of units. 

207 

208 Instances of this class should be accessed through either 

209 `get_physical_type` or by using the 

210 `~astropy.units.core.UnitBase.physical_type` attribute of units. 

211 This class is not intended to be instantiated directly in user code. 

212 

213 For a list of physical types, see `astropy.units.physical`. 

214 

215 Parameters 

216 ---------- 

217 unit : `~astropy.units.Unit` 

218 The unit to be represented by the physical type. 

219 

220 physical_types : `str` or `set` of `str` 

221 A `str` representing the name of the physical type of the unit, 

222 or a `set` containing strings that represent one or more names 

223 of physical types. 

224 

225 Notes 

226 ----- 

227 A physical type will be considered equal to an equivalent 

228 `PhysicalType` instance (recommended) or a string that contains a 

229 name of the physical type. The latter method is not recommended 

230 in packages, as the names of some physical types may change in the 

231 future. 

232 

233 To maintain backwards compatibility, two physical type names may be 

234 included in one string if they are separated with a slash (e.g., 

235 ``"momentum/impulse"``). String representations of physical types 

236 may include underscores instead of spaces. 

237 

238 Examples 

239 -------- 

240 `PhysicalType` instances may be accessed via the 

241 `~astropy.units.core.UnitBase.physical_type` attribute of units. 

242 

243 >>> import astropy.units as u 

244 >>> u.meter.physical_type 

245 PhysicalType('length') 

246 

247 `PhysicalType` instances may also be accessed by calling 

248 `get_physical_type`. This function will accept a unit, a string 

249 containing the name of a physical type, or the number one. 

250 

251 >>> u.get_physical_type(u.m ** -3) 

252 PhysicalType('number density') 

253 >>> u.get_physical_type("volume") 

254 PhysicalType('volume') 

255 >>> u.get_physical_type(1) 

256 PhysicalType('dimensionless') 

257 

258 Some units are dimensionally compatible with multiple physical types. 

259 A pascal is intended to represent pressure and stress, but the unit 

260 decomposition is equivalent to that of energy density. 

261 

262 >>> pressure = u.get_physical_type("pressure") 

263 >>> pressure 

264 PhysicalType({'energy density', 'pressure', 'stress'}) 

265 >>> 'energy density' in pressure 

266 True 

267 

268 Physical types can be tested for equality against other physical 

269 type objects or against strings that may contain the name of a 

270 physical type. 

271 

272 >>> area = (u.m ** 2).physical_type 

273 >>> area == u.barn.physical_type 

274 True 

275 >>> area == "area" 

276 True 

277 

278 Multiplication, division, and exponentiation are enabled so that 

279 physical types may be used for dimensional analysis. 

280 

281 >>> length = u.pc.physical_type 

282 >>> area = (u.cm ** 2).physical_type 

283 >>> length * area 

284 PhysicalType('volume') 

285 >>> area / length 

286 PhysicalType('length') 

287 >>> length ** 3 

288 PhysicalType('volume') 

289 

290 may also be performed using a string that contains the name of a 

291 physical type. 

292 

293 >>> "length" * area 

294 PhysicalType('volume') 

295 >>> "area" / length 

296 PhysicalType('length') 

297 

298 Unknown physical types are labelled as ``"unknown"``. 

299 

300 >>> (u.s ** 13).physical_type 

301 PhysicalType('unknown') 

302 

303 Dimensional analysis may be performed for unknown physical types too. 

304 

305 >>> length_to_19th_power = (u.m ** 19).physical_type 

306 >>> length_to_20th_power = (u.m ** 20).physical_type 

307 >>> length_to_20th_power / length_to_19th_power 

308 PhysicalType('length') 

309 """

371 def _dimensionally_compatible_unit(obj): 

372 """ 

373 Return a unit that corresponds to the provided argument. 

374 

375 If a unit is passed in, return that unit. If a physical type 

376 (or a `str` with the name of a physical type) is passed in, 

377 return a unit that corresponds to that physical type. If the 

378 number equal to ``1`` is passed in, return a dimensionless unit. 

379 Otherwise, return `NotImplemented`. 

380 """ 

381 if isinstance(obj, core.UnitBase): 

382 return _replace_temperatures_with_kelvin(obj) 

383 elif isinstance(obj, PhysicalType): 

384 return obj._unit 

385 elif isinstance(obj, numbers.Real) and obj == 1: 

386 return core.dimensionless_unscaled 

387 elif isinstance(obj, str): 

388 return _physical_type_from_str(obj)._unit 

389 else: 

390 return NotImplemented

397 new_unit = getattr(self._unit, operation)(other_unit) 

398 return new_unit.physical_type

432def def_physical_type(unit, name): 

433 """ 

434 Add a mapping between a unit and the corresponding physical type(s). 

435 

436 If a physical type already exists for a unit, add new physical type 

437 names so long as those names are not already in use for other 

438 physical types. 

439 

440 Parameters 

441 ---------- 

442 unit : `~astropy.units.Unit` 

443 The unit to be represented by the physical type. 

444 

445 name : `str` or `set` of `str` 

446 A `str` representing the name of the physical type of the unit, 

447 or a `set` containing strings that represent one or more names 

448 of physical types. 

449 

450 Raises 

451 ------ 

452 ValueError 

453 If a physical type name is already in use for another unit, or 

454 if attempting to name a unit as ``"unknown"``. 

455 

456 Notes 

457 ----- 

458 For a list of physical types, see `astropy.units.physical`. 

459 """ 

460 physical_type_id = unit._get_physical_type_id() 

461 physical_type_names = _standardize_physical_type_names(name) 

462 

463 if "unknown" in physical_type_names: 

464 raise ValueError("cannot uniquely define an unknown physical type") 

465 

466 names_for_other_units = set(_unit_physical_mapping.keys()).difference( 

467 _physical_unit_mapping.get(physical_type_id, {}) 

468 ) 

469 names_already_in_use = physical_type_names & names_for_other_units 

470 if names_already_in_use: 

471 raise ValueError( 

472 "the following physical type names are already in use: " 

473 f"{names_already_in_use}." 

474 ) 

475 

476 unit_already_in_use = physical_type_id in _physical_unit_mapping 

477 if unit_already_in_use: 

478 physical_type = _physical_unit_mapping[physical_type_id] 

479 physical_type_names |= set(physical_type) 

480 physical_type.__init__(unit, physical_type_names) 

481 else: 

482 physical_type = PhysicalType(unit, physical_type_names) 

483 _physical_unit_mapping[physical_type_id] = physical_type 

484 

485 for ptype in physical_type: 

486 _unit_physical_mapping[ptype] = physical_type_id 

487 

488 for ptype_name in physical_type_names: 

489 _name_physical_mapping[ptype_name] = physical_type 

490 # attribute-accessible name 

491 attr_name = ptype_name.replace(" ", "_").replace("(", "").replace(")", "") 

492 _attrname_physical_mapping[attr_name] = physical_type

495def get_physical_type(obj): 

496 """ 

497 Return the physical type that corresponds to a unit (or another 

498 physical type representation). 

499 

500 Parameters 

501 ---------- 

502 obj : quantity-like or `~astropy.units.PhysicalType`-like 

503 An object that (implicitly or explicitly) has a corresponding 

504 physical type. This object may be a unit, a 

505 `~astropy.units.Quantity`, an object that can be converted to a 

506 `~astropy.units.Quantity` (such as a number or array), a string 

507 that contains a name of a physical type, or a 

508 `~astropy.units.PhysicalType` instance. 

509 

510 Returns 

511 ------- 

512 `~astropy.units.PhysicalType` 

513 A representation of the physical type(s) of the unit. 

514 

515 Notes 

516 ----- 

517 For a list of physical types, see `astropy.units.physical`. 

518 

519 Examples 

520 -------- 

521 The physical type may be retrieved from a unit or a 

522 `~astropy.units.Quantity`. 

523 

524 >>> import astropy.units as u 

525 >>> u.get_physical_type(u.meter ** -2) 

526 PhysicalType('column density') 

527 >>> u.get_physical_type(0.62 * u.barn * u.Mpc) 

528 PhysicalType('volume') 

529 

530 The physical type may also be retrieved by providing a `str` that 

531 contains the name of a physical type. 

532 

533 >>> u.get_physical_type("energy") 

534 PhysicalType({'energy', 'torque', 'work'}) 

535 

536 Numbers and arrays of numbers correspond to a dimensionless physical 

537 type. 

538 

539 >>> u.get_physical_type(1) 

540 PhysicalType('dimensionless') 

541 """ 

542 if isinstance(obj, PhysicalType): 

543 return obj 

544 

545 if isinstance(obj, str): 

546 return _physical_type_from_str(obj) 

547 

548 if isinstance(obj, core.UnitBase): 

549 unit = obj 

550 else: 

551 try: 

552 unit = quantity.Quantity(obj, copy=False).unit 

553 except TypeError as exc: 

554 raise TypeError(f"{obj} does not correspond to a physical type.") from exc 

555 

556 unit = _replace_temperatures_with_kelvin(unit) 

557 physical_type_id = unit._get_physical_type_id() 

558 unit_has_known_physical_type = physical_type_id in _physical_unit_mapping 

559 

560 if unit_has_known_physical_type: 

561 return _physical_unit_mapping[physical_type_id] 

562 else: 

563 return PhysicalType(unit, "unknown")
