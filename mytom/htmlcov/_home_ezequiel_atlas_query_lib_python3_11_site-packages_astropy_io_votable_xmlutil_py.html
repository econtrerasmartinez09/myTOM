26def check_id(ID, name="ID", config=None, pos=None): 

27 """ 

28 Raises a `~astropy.io.votable.exceptions.VOTableSpecError` if *ID* 

29 is not a valid XML ID_. 

30 

31 *name* is the name of the attribute being checked (used only for 

32 error messages). 

33 """ 

34 if ID is not None and not xml_check.check_id(ID): 

35 warn_or_raise(W02, W02, (name, ID), config, pos) 

36 return False 

37 return True

40def fix_id(ID, config=None, pos=None): 

41 """ 

42 Given an arbitrary string, create one that can be used as an xml id. 

43 

44 This is rather simplistic at the moment, since it just replaces 

45 non-valid characters with underscores. 

46 """ 

47 if ID is None: 

48 return None 

49 corrected = xml_check.fix_id(ID) 

50 if corrected != ID: 

51 vo_warn(W03, (ID, corrected), config, pos) 

52 return corrected

95def validate_schema(filename, version="1.1"): 

96 """ 

97 Validates the given file against the appropriate VOTable schema. 

98 

99 Parameters 

100 ---------- 

101 filename : str 

102 The path to the XML file to validate 

103 

104 version : str, optional 

105 The VOTABLE version to check, which must be a string \"1.0\", 

106 \"1.1\", \"1.2\" or \"1.3\". If it is not one of these, 

107 version \"1.1\" is assumed. 

108 

109 For version \"1.0\", it is checked against a DTD, since that 

110 version did not have an XML Schema. 

111 

112 Returns 

113 ------- 

114 returncode, stdout, stderr : int, str, str 

115 Returns the returncode from xmllint and the stdout and stderr 

116 as strings 

117 """ 

118 if version not in ("1.0", "1.1", "1.2", "1.3"): 

119 log.info(f"{filename} has version {version}, using schema 1.1") 

120 version = "1.1" 

121 

122 if version in ("1.1", "1.2", "1.3"): 

123 schema_path = data.get_pkg_data_filename(f"data/VOTable.v{version}.xsd") 

124 else: 

125 schema_path = data.get_pkg_data_filename("data/VOTable.dtd") 

126 

127 return validate.validate_schema(filename, schema_path)
