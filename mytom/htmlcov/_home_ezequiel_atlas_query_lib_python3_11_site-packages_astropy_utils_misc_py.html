94def format_exception(msg, *args, **kwargs): 

95 """Fill in information about the exception that occurred. 

96 

97 Given an exception message string, uses new-style formatting arguments 

98 ``{filename}``, ``{lineno}``, ``{func}`` and/or ``{text}`` to fill in 

99 information about the exception that occurred. For example: 

100 

101 try: 

102 1/0 

103 except: 

104 raise ZeroDivisionError( 

105 format_except('A divide by zero occurred in {filename} at ' 

106 'line {lineno} of function {func}.')) 

107 

108 Any additional positional or keyword arguments passed to this function are 

109 also used to format the message. 

110 

111 .. note:: 

112 This uses `sys.exc_info` to gather up the information needed to fill 

113 in the formatting arguments. Since `sys.exc_info` is not carried 

114 outside a handled exception, it's not wise to use this 

115 outside of an ``except`` clause - if it is, this will substitute 

116 '<unknown>' for the 4 formatting arguments. 

117 """ 

118 tb = traceback.extract_tb(sys.exc_info()[2], limit=1) 

119 if len(tb) > 0: 

120 filename, lineno, func, text = tb[0] 

121 else: 

122 filename = lineno = func = text = "<unknown>" 

123 

124 return msg.format( 

125 *args, filename=filename, lineno=lineno, func=func, text=text, **kwargs 

126 )

131 A context manager (for use with the ``with`` statement) that will seed the

142 The value to use to seed the numpy RNG 

143

145 -------- 

146 A typical use case might be::

151 randarr = random.randn(100)

176def find_api_page(obj, version=None, openinbrowser=True, timeout=None): 

177 """ 

178 Determines the URL of the API page for the specified object, and 

179 optionally open that page in a web browser. 

180 

181 .. note:: 

182 You must be connected to the internet for this to function even if 

183 ``openinbrowser`` is `False`, unless you provide a local version of 

184 the documentation to ``version`` (e.g., ``file:///path/to/docs``). 

185 

186 Parameters 

187 ---------- 

188 obj 

189 The object to open the docs for or its fully-qualified name 

190 (as a str). 

191 version : str 

192 The doc version - either a version number like '0.1', 'dev' for 

193 the development/latest docs, or a URL to point to a specific 

194 location that should be the *base* of the documentation. Defaults to 

195 latest if you are on aren't on a release, otherwise, the version you 

196 are on. 

197 openinbrowser : bool 

198 If `True`, the `webbrowser` package will be used to open the doc 

199 page in a new web browser window. 

200 timeout : number, optional 

201 The number of seconds to wait before timing-out the query to 

202 the astropy documentation. If not given, the default python 

203 stdlib timeout will be used. 

204 

205 Returns 

206 ------- 

207 url : str 

208 The loaded URL 

209 

210 Raises 

211 ------ 

212 ValueError 

213 If the documentation can't be found 

214 

215 """ 

216 import webbrowser 

217 from zlib import decompress 

218 

219 from astropy.utils.data import get_readable_fileobj 

220 

221 if ( 

222 not isinstance(obj, str) 

223 and hasattr(obj, "__module__") 

224 and hasattr(obj, "__name__") 

225 ): 

226 obj = obj.__module__ + "." + obj.__name__ 

227 elif inspect.ismodule(obj): 

228 obj = obj.__name__ 

229 

230 if version is None: 

231 from astropy import version 

232 

233 if version.release: 

234 version = "v" + version.version 

235 else: 

236 version = "dev" 

237 

238 if "://" in version: 

239 if version.endswith("index.html"): 

240 baseurl = version[:-10] 

241 elif version.endswith("/"): 

242 baseurl = version 

243 else: 

244 baseurl = version + "/" 

245 elif version == "dev" or version == "latest": 

246 baseurl = "http://devdocs.astropy.org/" 

247 else: 

248 baseurl = f"https://docs.astropy.org/en/{version}/" 

249 

250 # Custom request headers; see 

251 # https://github.com/astropy/astropy/issues/8990 

252 url = baseurl + "objects.inv" 

253 headers = {"User-Agent": f"Astropy/{version}"} 

254 with get_readable_fileobj( 

255 url, encoding="binary", remote_timeout=timeout, http_headers=headers 

256 ) as uf: 

257 oiread = uf.read() 

258 

259 # need to first read/remove the first four lines, which have info before 

260 # the compressed section with the actual object inventory 

261 idx = -1 

262 headerlines = [] 

263 for _ in range(4): 

264 oldidx = idx 

265 idx = oiread.index(b"\n", oldidx + 1) 

266 headerlines.append(oiread[(oldidx + 1) : idx].decode("utf-8")) 

267 

268 # intersphinx version line, project name, and project version 

269 ivers, proj, vers, compr = headerlines 

270 if "The remainder of this file is compressed using zlib" not in compr: 

271 raise ValueError( 

272 f"The file downloaded from {baseurl}objects.inv does not seem to be" 

273 "the usual Sphinx objects.inv format. Maybe it " 

274 "has changed?" 

275 ) 

276 

277 compressed = oiread[(idx + 1) :] 

278 

279 decompressed = decompress(compressed).decode("utf-8") 

280 

281 resurl = None 

282 

283 for l in decompressed.strip().splitlines(): 

284 ls = l.split() 

285 name = ls[0] 

286 loc = ls[3] 

287 if loc.endswith("$"): 

288 loc = loc[:-1] + name 

289 

290 if name == obj: 

291 resurl = baseurl + loc 

292 break 

293 

294 if resurl is None: 

295 raise ValueError(f"Could not find the docs for the object {obj}") 

296 elif openinbrowser: 

297 webbrowser.open(resurl) 

298 

299 return resurl

443def did_you_mean(s, candidates, n=3, cutoff=0.8, fix=None): 

444 """ 

445 When a string isn't found in a set of candidates, we can be nice 

446 to provide a list of alternatives in the exception. This 

447 convenience function helps to format that part of the exception. 

448 

449 Parameters 

450 ---------- 

451 s : str 

452 

453 candidates : sequence of str or dict of str keys 

454 

455 n : int 

456 The maximum number of results to include. See 

457 `difflib.get_close_matches`. 

458 

459 cutoff : float 

460 In the range [0, 1]. Possibilities that don't score at least 

461 that similar to word are ignored. See 

462 `difflib.get_close_matches`. 

463 

464 fix : callable 

465 A callable to modify the results after matching. It should 

466 take a single string and return a sequence of strings 

467 containing the fixed matches. 

468 

469 Returns 

470 ------- 

471 message : str 

472 Returns the string "Did you mean X, Y, or Z?", or the empty 

473 string if no alternatives were found. 

474 """ 

475 if isinstance(s, str): 

476 s = strip_accents(s) 

477 s_lower = s.lower() 

478 

479 # Create a mapping from the lower case name to all capitalization 

480 # variants of that name. 

481 candidates_lower = {} 

482 for candidate in candidates: 

483 candidate_lower = candidate.lower() 

484 candidates_lower.setdefault(candidate_lower, []) 

485 candidates_lower[candidate_lower].append(candidate) 

486 

487 # The heuristic here is to first try "singularizing" the word. If 

488 # that doesn't match anything use difflib to find close matches in 

489 # original, lower and upper case. 

490 if s_lower.endswith("s") and s_lower[:-1] in candidates_lower: 

491 matches = [s_lower[:-1]] 

492 else: 

493 matches = difflib.get_close_matches( 

494 s_lower, candidates_lower, n=n, cutoff=cutoff 

495 ) 

496 

497 if len(matches): 

498 capitalized_matches = set() 

499 for match in matches: 

500 capitalized_matches.update(candidates_lower[match]) 

501 matches = capitalized_matches 

502 

503 if fix is not None: 

504 mapped_matches = [] 

505 for match in matches: 

506 mapped_matches.extend(fix(match)) 

507 matches = mapped_matches 

508 

509 matches = list(set(matches)) 

510 matches = sorted(matches) 

511 

512 if len(matches) == 1: 

513 matches = matches[0] 

514 else: 

515 matches = ", ".join(matches[:-1]) + " or " + matches[-1] 

516 return f"Did you mean {matches}?" 

517 

518 return ""

531@deprecated("4.3", _ordered_descriptor_deprecation_message) 

532class OrderedDescriptor(metaclass=abc.ABCMeta): 

533 """ 

534 Base class for descriptors whose order in the class body should be 

535 preserved. Intended for use in concert with the 

536 `OrderedDescriptorContainer` metaclass. 

537 

538 Subclasses of `OrderedDescriptor` must define a value for a class attribute 

539 called ``_class_attribute_``. This is the name of a class attribute on the 

540 *container* class for these descriptors, which will be set to an 

541 `~collections.OrderedDict` at class creation time. This 

542 `~collections.OrderedDict` will contain a mapping of all class attributes 

543 that were assigned instances of the `OrderedDescriptor` subclass, to the 

544 instances themselves. See the documentation for 

545 `OrderedDescriptorContainer` for a concrete example. 

546 

547 Optionally, subclasses of `OrderedDescriptor` may define a value for a 

548 class attribute called ``_name_attribute_``. This should be the name of 

549 an attribute on instances of the subclass. When specified, during 

550 creation of a class containing these descriptors, the name attribute on 

551 each instance will be set to the name of the class attribute it was 

552 assigned to on the class. 

553 

554 .. note:: 

555 

556 Although this class is intended for use with *descriptors* (i.e. 

557 classes that define any of the ``__get__``, ``__set__``, or 

558 ``__delete__`` magic methods), this base class is not itself a 

559 descriptor, and technically this could be used for classes that are 

560 not descriptors too. However, use with descriptors is the original 

561 intended purpose. 

562 """ 

563 

564 # This id increments for each OrderedDescriptor instance created, so they 

565 # are always ordered in the order they were created. Class bodies are 

566 # guaranteed to be executed from top to bottom. Not sure if this is 

567 # thread-safe though. 

568 _nextid = 1 

569 

570 @property 

571 @abc.abstractmethod 

572 def _class_attribute_(self): 

573 """ 

574 Subclasses should define this attribute to the name of an attribute on 

575 classes containing this subclass. That attribute will contain the mapping 

576 of all instances of that `OrderedDescriptor` subclass defined in the class 

577 body. If the same descriptor needs to be used with different classes, 

578 each with different names of this attribute, multiple subclasses will be 

579 needed. 

580 """ 

581 

582 _name_attribute_ = None 

583 """ 

584 Subclasses may optionally define this attribute to specify the name of an 

585 attribute on instances of the class that should be filled with the 

586 instance's attribute name at class creation time. 

587 """ 

588 

589 def __init__(self, *args, **kwargs): 

590 # The _nextid attribute is shared across all subclasses so that 

591 # different subclasses of OrderedDescriptors can be sorted correctly 

592 # between themselves 

593 self.__order = OrderedDescriptor._nextid 

594 OrderedDescriptor._nextid += 1 

595 super().__init__() 

596 

597 def __lt__(self, other): 

598 """ 

599 Defined for convenient sorting of `OrderedDescriptor` instances, which 

600 are defined to sort in their creation order. 

601 """ 

602 if isinstance(self, OrderedDescriptor) and isinstance(other, OrderedDescriptor): 

603 try: 

604 return self.__order < other.__order 

605 except AttributeError: 

606 raise RuntimeError( 

607 f"Could not determine ordering for {self} and {other}; at least " 

608 "one of them is not calling super().__init__ in its " 

609 "__init__." 

610 ) 

611 else: 

612 return NotImplemented

615@deprecated("4.3", _ordered_descriptor_deprecation_message) 

616class OrderedDescriptorContainer(type): 

617 """ 

618 Classes should use this metaclass if they wish to use `OrderedDescriptor` 

619 attributes, which are class attributes that "remember" the order in which 

620 they were defined in the class body. 

621 

622 Every subclass of `OrderedDescriptor` has an attribute called 

623 ``_class_attribute_``. For example, if we have 

624 

625 .. code:: python 

626 

627 class ExampleDecorator(OrderedDescriptor): 

628 _class_attribute_ = '_examples_' 

629 

630 Then when a class with the `OrderedDescriptorContainer` metaclass is 

631 created, it will automatically be assigned a class attribute ``_examples_`` 

632 referencing an `~collections.OrderedDict` containing all instances of 

633 ``ExampleDecorator`` defined in the class body, mapped to by the names of 

634 the attributes they were assigned to. 

635 

636 When subclassing a class with this metaclass, the descriptor dict (i.e. 

637 ``_examples_`` in the above example) will *not* contain descriptors 

638 inherited from the base class. That is, this only works by default with 

639 decorators explicitly defined in the class body. However, the subclass 

640 *may* define an attribute ``_inherit_decorators_`` which lists 

641 `OrderedDescriptor` classes that *should* be added from base classes. 

642 See the examples section below for an example of this. 

643 

644 Examples 

645 -------- 

646 >>> from astropy.utils import OrderedDescriptor, OrderedDescriptorContainer 

647 >>> class TypedAttribute(OrderedDescriptor): 

648 ... \"\"\" 

649 ... Attributes that may only be assigned objects of a specific type, 

650 ... or subclasses thereof. For some reason we care about their order. 

651 ... \"\"\" 

652 ... 

653 ... _class_attribute_ = 'typed_attributes' 

654 ... _name_attribute_ = 'name' 

655 ... # A default name so that instances not attached to a class can 

656 ... # still be repr'd; useful for debugging 

657 ... name = '<unbound>' 

658 ... 

659 ... def __init__(self, type): 

660 ... # Make sure not to forget to call the super __init__ 

661 ... super().__init__() 

662 ... self.type = type 

663 ... 

664 ... def __get__(self, obj, objtype=None): 

665 ... if obj is None: 

666 ... return self 

667 ... if self.name in obj.__dict__: 

668 ... return obj.__dict__[self.name] 

669 ... else: 

670 ... raise AttributeError(self.name) 

671 ... 

672 ... def __set__(self, obj, value): 

673 ... if not isinstance(value, self.type): 

674 ... raise ValueError('{0}.{1} must be of type {2!r}'.format( 

675 ... obj.__class__.__name__, self.name, self.type)) 

676 ... obj.__dict__[self.name] = value 

677 ... 

678 ... def __delete__(self, obj): 

679 ... if self.name in obj.__dict__: 

680 ... del obj.__dict__[self.name] 

681 ... else: 

682 ... raise AttributeError(self.name) 

683 ... 

684 ... def __repr__(self): 

685 ... if isinstance(self.type, tuple) and len(self.type) > 1: 

686 ... typestr = '({0})'.format( 

687 ... ', '.join(t.__name__ for t in self.type)) 

688 ... else: 

689 ... typestr = self.type.__name__ 

690 ... return '<{0}(name={1}, type={2})>'.format( 

691 ... self.__class__.__name__, self.name, typestr) 

692 ... 

693 

694 Now let's create an example class that uses this ``TypedAttribute``:: 

695 

696 >>> class Point2D(metaclass=OrderedDescriptorContainer): 

697 ... x = TypedAttribute((float, int)) 

698 ... y = TypedAttribute((float, int)) 

699 ... 

700 ... def __init__(self, x, y): 

701 ... self.x, self.y = x, y 

702 ... 

703 >>> p1 = Point2D(1.0, 2.0) 

704 >>> p1.x 

705 1.0 

706 >>> p1.y 

707 2.0 

708 >>> p2 = Point2D('a', 'b') # doctest: +IGNORE_EXCEPTION_DETAIL 

709 Traceback (most recent call last): 

710 ... 

711 ValueError: Point2D.x must be of type (float, int>) 

712 

713 We see that ``TypedAttribute`` works more or less as advertised, but 

714 there's nothing special about that. Let's see what 

715 `OrderedDescriptorContainer` did for us:: 

716 

717 >>> Point2D.typed_attributes 

718 OrderedDict([('x', <TypedAttribute(name=x, type=(float, int))>), 

719 ('y', <TypedAttribute(name=y, type=(float, int))>)]) 

720 

721 If we create a subclass, it does *not* by default add inherited descriptors 

722 to ``typed_attributes``:: 

723 

724 >>> class Point3D(Point2D): 

725 ... z = TypedAttribute((float, int)) 

726 ... 

727 >>> Point3D.typed_attributes 

728 OrderedDict([('z', <TypedAttribute(name=z, type=(float, int))>)]) 

729 

730 However, if we specify ``_inherit_descriptors_`` from ``Point2D`` then 

731 it will do so:: 

732 

733 >>> class Point3D(Point2D): 

734 ... _inherit_descriptors_ = (TypedAttribute,) 

735 ... z = TypedAttribute((float, int)) 

736 ... 

737 >>> Point3D.typed_attributes 

738 OrderedDict([('x', <TypedAttribute(name=x, type=(float, int))>), 

739 ('y', <TypedAttribute(name=y, type=(float, int))>), 

740 ('z', <TypedAttribute(name=z, type=(float, int))>)]) 

741 

742 .. note:: 

743 

744 Hopefully it is clear from these examples that this construction 

745 also allows a class of type `OrderedDescriptorContainer` to use 

746 multiple different `OrderedDescriptor` classes simultaneously. 

747 """ 

748 

749 _inherit_descriptors_ = () 

750 

751 def __init__(cls, cls_name, bases, members): 

752 descriptors = defaultdict(list) 

753 seen = set() 

754 inherit_descriptors = () 

755 descr_bases = {} 

756 

757 for mro_cls in cls.__mro__: 

758 for name, obj in mro_cls.__dict__.items(): 

759 if name in seen: 

760 # Checks if we've already seen an attribute of the given 

761 # name (if so it will override anything of the same name in 

762 # any base class) 

763 continue 

764 

765 seen.add(name) 

766 

767 if not isinstance(obj, OrderedDescriptor) or ( 

768 inherit_descriptors and not isinstance(obj, inherit_descriptors) 

769 ): 

770 # The second condition applies when checking any 

771 # subclasses, to see if we can inherit any descriptors of 

772 # the given type from subclasses (by default inheritance is 

773 # disabled unless the class has _inherit_descriptors_ 

774 # defined) 

775 continue 

776 

777 if obj._name_attribute_ is not None: 

778 setattr(obj, obj._name_attribute_, name) 

779 

780 # Don't just use the descriptor's class directly; instead go 

781 # through its MRO and find the class on which _class_attribute_ 

782 # is defined directly. This way subclasses of some 

783 # OrderedDescriptor *may* override _class_attribute_ and have 

784 # its own _class_attribute_, but by default all subclasses of 

785 # some OrderedDescriptor are still grouped together 

786 # TODO: It might be worth clarifying this in the docs 

787 if obj.__class__ not in descr_bases: 

788 for obj_cls_base in obj.__class__.__mro__: 

789 if "_class_attribute_" in obj_cls_base.__dict__: 

790 descr_bases[obj.__class__] = obj_cls_base 

791 descriptors[obj_cls_base].append((obj, name)) 

792 break 

793 else: 

794 # Make sure to put obj first for sorting purposes 

795 obj_cls_base = descr_bases[obj.__class__] 

796 descriptors[obj_cls_base].append((obj, name)) 

797 

798 if not getattr(mro_cls, "_inherit_descriptors_", False): 

799 # If _inherit_descriptors_ is undefined then we don't inherit 

800 # any OrderedDescriptors from any of the base classes, and 

801 # there's no reason to continue through the MRO 

802 break 

803 else: 

804 inherit_descriptors = mro_cls._inherit_descriptors_ 

805 

806 for descriptor_cls, instances in descriptors.items(): 

807 instances.sort() 

808 instances = OrderedDict((key, value) for value, key in instances) 

809 setattr(cls, descriptor_cls._class_attribute_, instances) 

810 

811 super().__init__(cls_name, bases, members)
