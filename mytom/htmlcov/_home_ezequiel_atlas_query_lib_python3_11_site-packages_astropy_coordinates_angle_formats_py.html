100 # NOTE THE ORDERING OF THESE RULES IS IMPORTANT!! 

101 # Regular expression rules for simple tokens 

102 def t_UFLOAT(t): 

103 r"((\d+\.\d*)|(\.\d+))([eE][+-−]?\d+)?" 

104 # The above includes Unicode "MINUS SIGN" \u2212. It is 

105 # important to include the hyphen last, or the regex will 

106 # treat this as a range. 

107 t.value = float(t.value.replace("−", "-")) 

108 return t 

109 

110 def t_UINT(t): 

111 r"\d+" 

112 t.value = int(t.value) 

113 return t 

114 

115 def t_SIGN(t): 

116 r"[+−-]" 

117 # The above include Unicode "MINUS SIGN" \u2212. It is 

118 # important to include the hyphen last, or the regex will 

119 # treat this as a range. 

120 if t.value == "+": 

121 t.value = 1.0 

122 else: 

123 t.value = -1.0 

124 return t 

125 

126 def t_EASTWEST(t): 

127 r"[EW]$" 

128 t.value = -1.0 if t.value == "W" else 1.0 

129 return t 

130 

131 def t_NORTHSOUTH(t): 

132 r"[NS]$" 

133 # We cannot use lower-case letters otherwise we'll confuse 

134 # s[outh] with s[econd] 

135 t.value = -1.0 if t.value == "S" else 1.0 

136 return t 

137 

138 def t_SIMPLE_UNIT(t): 

139 t.value = u.Unit(t.value) 

140 return t

305 def parse(self, angle, unit, debug=False): 

306 try: 

307 found_angle, found_unit = self._thread_local._parser.parse(

309 ) 

310 except ValueError as e:

402def degrees_to_dms(d): 

403 """ 

404 Convert a floating-point degree value into a ``(degree, arcminute, 

405 arcsecond)`` tuple. 

406 """ 

407 sign = np.copysign(1.0, d) 

408 

409 (df, d) = np.modf(np.abs(d)) # (degree fraction, degree) 

410 (mf, m) = np.modf(df * 60.0) # (minute fraction, minute) 

411 s = mf * 60.0 

412 

413 return np.floor(sign * d), sign * np.floor(m), sign * s

428 """ 

429 Convert degrees, arcminute, arcsecond to a float degrees value. 

430 """ 

431 _check_minute_range(m) 

432 _check_second_range(s) 

433 

434 # determine sign 

435 sign = np.copysign(1.0, d) 

436 

437 try: 

438 d = np.floor(np.abs(d)) 

439 if s is None: 

440 m = np.abs(m) 

441 s = 0 

442 else: 

443 m = np.floor(np.abs(m)) 

444 s = np.abs(s) 

445 except ValueError as err:

450 d, 

451 m, 

452 s, 

453 ) 

454 ) from err 

455 

456 return sign * (d + m / 60.0 + s / 3600.0)

584def sexagesimal_to_string(values, precision=None, pad=False, sep=(":",), fields=3): 

585 """ 

586 Given an already separated tuple of sexagesimal values, returns 

587 a string. 

588 

589 See `hours_to_string` and `degrees_to_string` for a higher-level 

590 interface to this functionality. 

591 """ 

592 # Check to see if values[0] is negative, using np.copysign to handle -0 

593 sign = np.copysign(1.0, values[0]) 

594 # If the coordinates are negative, we need to take the absolute values. 

595 # We use np.abs because abs(-0) is -0 

596 # TODO: Is this true? (MHvK, 2018-02-01: not on my system) 

597 values = [np.abs(value) for value in values] 

598 

599 if pad: 

600 if sign == -1: 

601 pad = 3 

602 else: 

603 pad = 2 

604 else: 

605 pad = 0 

606 

607 if not isinstance(sep, tuple): 

608 sep = tuple(sep) 

609 

610 if fields < 1 or fields > 3: 

611 raise ValueError("fields must be 1, 2, or 3") 

612 

613 if not sep: # empty string, False, or None, etc. 

614 sep = ("", "", "") 

615 elif len(sep) == 1: 

616 if fields == 3: 

617 sep = sep + (sep[0], "") 

618 elif fields == 2: 

619 sep = sep + ("", "") 

620 else: 

621 sep = ("", "", "") 

622 elif len(sep) == 2: 

623 sep = sep + ("",) 

624 elif len(sep) != 3: 

625 raise ValueError( 

626 "Invalid separator specification for converting angle to string." 

627 ) 

628 

629 # Simplify the expression based on the requested precision. For 

630 # example, if the seconds will round up to 60, we should convert 

631 # it to 0 and carry upwards. If the field is hidden (by the 

632 # fields kwarg) we round up around the middle, 30.0. 

633 if precision is None: 

634 rounding_thresh = 60.0 - (10.0**-8) 

635 else: 

636 rounding_thresh = 60.0 - (10.0**-precision) 

637 

638 if fields == 3 and values[2] >= rounding_thresh: 

639 values[2] = 0.0 

640 values[1] += 1.0 

641 elif fields < 3 and values[2] >= 30.0: 

642 values[1] += 1.0 

643 

644 if fields >= 2 and values[1] >= 60.0: 

645 values[1] = 0.0 

646 values[0] += 1.0 

647 elif fields < 2 and values[1] >= 30.0: 

648 values[0] += 1.0 

649 

650 literal = [] 

651 last_value = "" 

652 literal.append("{0:0{pad}.0f}{sep[0]}") 

653 if fields >= 2: 

654 literal.append("{1:02d}{sep[1]}") 

655 if fields == 3: 

656 if precision is None: 

657 last_value = f"{abs(values[2]):.8f}" 

658 last_value = last_value.rstrip("0").rstrip(".") 

659 else: 

660 last_value = "{0:.{precision}f}".format(abs(values[2]), precision=precision) 

661 if len(last_value) == 1 or last_value[1] == ".": 

662 last_value = "0" + last_value 

663 literal.append("{last_value}{sep[2]}") 

664 literal = "".join(literal) 

665 return literal.format( 

666 np.copysign(values[0], sign), 

667 int(values[1]), 

668 values[2], 

669 sep=sep, 

670 pad=pad, 

671 last_value=last_value, 

672 )
