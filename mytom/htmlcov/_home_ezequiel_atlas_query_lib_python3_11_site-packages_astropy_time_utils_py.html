18def day_frac(val1, val2, factor=None, divisor=None): 

19 """Return the sum of ``val1`` and ``val2`` as two float64s. 

20 

21 The returned floats are an integer part and the fractional remainder, 

22 with the latter guaranteed to be within -0.5 and 0.5 (inclusive on 

23 either side, as the integer is rounded to even). 

24 

25 The arithmetic is all done with exact floating point operations so no 

26 precision is lost to rounding error. It is assumed the sum is less 

27 than about 1e16, otherwise the remainder will be greater than 1.0. 

28 

29 Parameters 

30 ---------- 

31 val1, val2 : array of float 

32 Values to be summed. 

33 factor : float, optional 

34 If given, multiply the sum by it. 

35 divisor : float, optional 

36 If given, divide the sum by it. 

37 

38 Returns 

39 ------- 

40 day, frac : float64 

41 Integer and fractional part of val1 + val2. 

42 """ 

43 # Add val1 and val2 exactly, returning the result as two float64s. 

44 # The first is the approximate sum (with some floating point error) 

45 # and the second is the error of the float64 sum. 

46 sum12, err12 = two_sum(val1, val2) 

47 

48 if factor is not None: 

49 sum12, carry = two_product(sum12, factor) 

50 carry += err12 * factor 

51 sum12, err12 = two_sum(sum12, carry) 

52 

53 if divisor is not None: 

54 q1 = sum12 / divisor 

55 p1, p2 = two_product(q1, divisor) 

56 d1, d2 = two_sum(sum12, -p1) 

57 d2 += err12 

58 d2 -= p2 

59 q2 = (d1 + d2) / divisor # 3-part float fine here; nothing can be lost 

60 sum12, err12 = two_sum(q1, q2) 

61 

62 # get integer fraction 

63 day = np.round(sum12) 

64 # Calculate remaining fraction. This can have gotten >0.5 or <-0.5, which means 

65 # we would lose one bit of precision. So, correct for that. Here, we need 

66 # particular care for the case that frac=0.5 and check>0 or frac=-0.5 and check<0, 

67 # since in that case if check is large enough, rounding was done the wrong way. 

68 frac, check = two_sum(sum12 - day, err12) 

69 excess = np.where( 

70 frac * np.sign(check) != 0.5, np.round(frac), np.round(frac + 2 * check) 

71 ) 

72 day += excess 

73 frac = sum12 - day 

74 frac += err12 

75 return day, frac

78def quantity_day_frac(val1, val2=None): 

79 """Like ``day_frac``, but for quantities with units of time. 

80 

81 The quantities are separately converted to days. Here, we need to take 

82 care with the conversion since while the routines here can do accurate 

83 multiplication, the conversion factor itself may not be accurate. For 

84 instance, if the quantity is in seconds, the conversion factor is 

85 1./86400., which is not exactly representable as a float. 

86 

87 To work around this, for conversion factors less than unity, rather than 

88 multiply by that possibly inaccurate factor, the value is divided by the 

89 conversion factor of a day to that unit (i.e., by 86400. for seconds). For 

90 conversion factors larger than 1, such as 365.25 for years, we do just 

91 multiply. With this scheme, one has precise conversion factors for all 

92 regular time units that astropy defines. Note, however, that it does not 

93 necessarily work for all custom time units, and cannot work when conversion 

94 to time is via an equivalency. For those cases, one remains limited by the 

95 fact that Quantity calculations are done in double precision, not in 

96 quadruple precision as for time. 

97 """ 

98 if val2 is not None: 

99 res11, res12 = quantity_day_frac(val1) 

100 res21, res22 = quantity_day_frac(val2) 

101 # This summation is can at most lose 1 ULP in the second number. 

102 return res11 + res21, res12 + res22 

103 

104 try: 

105 factor = val1.unit.to(u.day) 

106 except Exception: 

107 # Not a simple scaling, so cannot do the full-precision one. 

108 # But at least try normal conversion, since equivalencies may be set. 

109 return val1.to_value(u.day), 0.0 

110 

111 if factor == 1.0: 

112 return day_frac(val1.value, 0.0) 

113 

114 if factor > 1: 

115 return day_frac(val1.value, 0.0, factor=factor) 

116 else: 

117 divisor = u.day.to(val1.unit) 

118 return day_frac(val1.value, 0.0, divisor=divisor)

121def two_sum(a, b): 

122 """ 

123 Add ``a`` and ``b`` exactly, returning the result as two float64s. 

124 The first is the approximate sum (with some floating point error) 

125 and the second is the error of the float64 sum. 

126 

127 Using the procedure of Shewchuk, 1997, 

128 Discrete & Computational Geometry 18(3):305-363 

129 http://www.cs.berkeley.edu/~jrs/papers/robustr.pdf 

130 

131 Returns 

132 ------- 

133 sum, err : float64 

134 Approximate sum of a + b and the exact floating point error 

135 """ 

136 x = a + b 

137 eb = x - a # bvirtual in Shewchuk 

138 ea = x - eb # avirtual in Shewchuk 

139 eb = b - eb # broundoff in Shewchuk 

140 ea = a - ea # aroundoff in Shewchuk 

141 return x, ea + eb

144def two_product(a, b): 

145 """ 

146 Multiple ``a`` and ``b`` exactly, returning the result as two float64s. 

147 The first is the approximate product (with some floating point error) 

148 and the second is the error of the float64 product. 

149 

150 Uses the procedure of Shewchuk, 1997, 

151 Discrete & Computational Geometry 18(3):305-363 

152 http://www.cs.berkeley.edu/~jrs/papers/robustr.pdf 

153 

154 Returns 

155 ------- 

156 prod, err : float64 

157 Approximate product a * b and the exact floating point error 

158 """ 

159 x = a * b 

160 ah, al = split(a) 

161 bh, bl = split(b) 

162 y1 = ah * bh 

163 y = x - y1 

164 y2 = al * bh 

165 y -= y2 

166 y3 = ah * bl 

167 y -= y3 

168 y4 = al * bl 

169 y = y4 - y 

170 return x, y

173def split(a): 

174 """ 

175 Split float64 in two aligned parts. 

176 

177 Uses the procedure of Shewchuk, 1997, 

178 Discrete & Computational Geometry 18(3):305-363 

179 http://www.cs.berkeley.edu/~jrs/papers/robustr.pdf 

180 

181 """ 

182 c = 134217729.0 * a # 2**27+1. 

183 abig = c - a 

184 ah = c - abig 

185 al = a - ah 

186 return ah, al

192def longdouble_to_twoval(val1, val2=None): 

193 if val2 is None: 

194 val2 = val1.dtype.type(0.0) 

195 else: 

196 best_type = np.result_type(val1.dtype, val2.dtype) 

197 val1 = val1.astype(best_type, copy=False) 

198 val2 = val2.astype(best_type, copy=False) 

199 

200 # day_frac is independent of dtype, as long as the dtype 

201 # are the same and no factor or divisor is given. 

202 i, f = day_frac(val1, val2) 

203 return i.astype(float, copy=False), f.astype(float, copy=False)

206def decimal_to_twoval1(val1, val2=None): 

207 with decimal.localcontext() as ctx: 

208 ctx.prec = _enough_decimal_places 

209 d = decimal.Decimal(val1) 

210 i = round(d) 

211 f = d - i 

212 return float(i), float(f)

229def twoval_to_string1(val1, val2, fmt): 

230 if val2 == 0.0: 

231 # For some formats, only a single float is really used. 

232 # For those, let numpy take care of correct number of digits.
