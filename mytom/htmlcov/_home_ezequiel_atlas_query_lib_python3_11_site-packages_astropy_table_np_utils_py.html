21def get_col_name_map( 

22 arrays, common_names, uniq_col_name="{col_name}_{table_name}", table_names=None 

23): 

24 """ 

25 Find the column names mapping when merging the list of structured ndarrays 

26 ``arrays``. It is assumed that col names in ``common_names`` are to be 

27 merged into a single column while the rest will be uniquely represented 

28 in the output. The args ``uniq_col_name`` and ``table_names`` specify 

29 how to rename columns in case of conflicts. 

30 

31 Returns a dict mapping each output column name to the input(s). This takes the form 

32 {outname : (col_name_0, col_name_1, ...), ... }. For key columns all of input names 

33 will be present, while for the other non-key columns the value will be (col_name_0, 

34 None, ..) or (None, col_name_1, ..) etc. 

35 """ 

36 col_name_map = collections.defaultdict(lambda: [None] * len(arrays)) 

37 col_name_list = [] 

38 

39 if table_names is None: 

40 table_names = [str(ii + 1) for ii in range(len(arrays))] 

41 

42 for idx, array in enumerate(arrays): 

43 table_name = table_names[idx] 

44 for name in array.dtype.names: 

45 out_name = name 

46 

47 if name in common_names: 

48 # If name is in the list of common_names then insert into 

49 # the column name list, but just once. 

50 if name not in col_name_list: 

51 col_name_list.append(name) 

52 else: 

53 # If name is not one of the common column outputs, and it collides 

54 # with the names in one of the other arrays, then rename 

55 others = list(arrays) 

56 others.pop(idx) 

57 if any(name in other.dtype.names for other in others): 

58 out_name = uniq_col_name.format( 

59 table_name=table_name, col_name=name 

60 ) 

61 col_name_list.append(out_name) 

62 

63 col_name_map[out_name][idx] = name 

64 

65 # Check for duplicate output column names 

66 col_name_count = Counter(col_name_list) 

67 repeated_names = [name for name, count in col_name_count.items() if count > 1] 

68 if repeated_names: 

69 raise TableMergeError( 

70 "Merging column names resulted in duplicates: {}. " 

71 "Change uniq_col_name or table_names args to fix this.".format( 

72 repeated_names 

73 ) 

74 ) 

75 

76 # Convert col_name_map to a regular dict with tuple (immutable) values 

77 col_name_map = OrderedDict((name, col_name_map[name]) for name in col_name_list) 

78 

79 return col_name_map

82def get_descrs(arrays, col_name_map): 

83 """ 

84 Find the dtypes descrs resulting from merging the list of arrays' dtypes, 

85 using the column name mapping ``col_name_map``. 

86 

87 Return a list of descrs for the output. 

88 """ 

89 out_descrs = [] 

90 

91 for out_name, in_names in col_name_map.items(): 

92 # List of input arrays that contribute to this output column 

93 in_cols = [arr[name] for arr, name in zip(arrays, in_names) if name is not None] 

94 

95 # List of names of the columns that contribute to this output column. 

96 names = [name for name in in_names if name is not None] 

97 

98 # Output dtype is the superset of all dtypes in in_arrays 

99 try: 

100 dtype = common_dtype(in_cols) 

101 except TableMergeError as tme: 

102 # Beautify the error message when we are trying to merge columns with incompatible 

103 # types by including the name of the columns that originated the error. 

104 raise TableMergeError( 

105 "The '{}' columns have incompatible types: {}".format( 

106 names[0], tme._incompat_types 

107 ) 

108 ) from tme 

109 

110 # Make sure all input shapes are the same 

111 uniq_shapes = {col.shape[1:] for col in in_cols} 

112 if len(uniq_shapes) != 1: 

113 raise TableMergeError("Key columns have different shape") 

114 shape = uniq_shapes.pop() 

115 

116 if out_name is not None: 

117 out_name = str(out_name) 

118 out_descrs.append((out_name, dtype, shape)) 

119 

120 return out_descrs

123def common_dtype(cols): 

124 """ 

125 Use numpy to find the common dtype for a list of structured ndarray columns. 

126 

127 Only allow columns within the following fundamental numpy data types: 

128 np.bool_, np.object_, np.number, np.character, np.void 

129 """ 

130 np_types = (np.bool_, np.object_, np.number, np.character, np.void) 

131 uniq_types = { 

132 tuple(issubclass(col.dtype.type, np_type) for np_type in np_types) 

133 for col in cols 

134 } 

135 if len(uniq_types) > 1: 

136 # Embed into the exception the actual list of incompatible types. 

137 incompat_types = [col.dtype.name for col in cols] 

138 tme = TableMergeError(f"Columns have incompatible types {incompat_types}") 

139 tme._incompat_types = incompat_types 

140 raise tme 

141 

142 arrs = [np.empty(1, dtype=col.dtype) for col in cols] 

143 

144 # For string-type arrays need to explicitly fill in non-zero 

145 # values or the final arr_common = .. step is unpredictable. 

146 for arr in arrs: 

147 if arr.dtype.kind in ("S", "U"): 

148 arr[0] = "0" * arr.itemsize 

149 

150 arr_common = np.array([arr[0] for arr in arrs]) 

151 return arr_common.dtype.str
