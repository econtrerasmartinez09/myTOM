25def generic_fits(file_obj, **kwargs): 

26 

27 with read_fileobj_or_hdulist(file_obj, **kwargs) as hdulist: 

28 header = hdulist[0].header 

29 data3 = hdulist[0].data 

30 wcs = WCS(header) 

31 shape = data3.shape 

32 

33 # take the reference pixel if the pos= was not supplied by the reader 

34 if 'pos' in kwargs: 

35 ix = kwargs['pos'][0] 

36 iy = kwargs['pos'][1] 

37 else: 

38 ix = int(wcs.wcs.crpix[0]) 

39 iy = int(wcs.wcs.crpix[1]) 

40 

41 # grab a spectrum from the cube 

42 if len(shape) == 3: 

43 data = data3[:,iy,ix] 

44 elif len(shape) == 4: 

45 data = data3[:,:,iy,ix].squeeze() 

46 # make sure this is a 1D array 

47 # if len(data.shape) != 1: 

48 # raise Exception("not a true cube") 

49 else: 

50 raise ValueError(f"Unexpected shape {shape}.") 

51 

52 # store some meta data 

53 meta = {'header': header} 

54 meta['xpos'] = ix 

55 meta['ypos'] = iy 

56 

57 # attach units (get it from header['BUNIT'] - what about 'JY/BEAM ' 

58 # NOTE: astropy doesn't support beam, but see comments in radio_beam 

59 data = data * Unit("Jy") 

60 

61 # now figure out the frequency axis.... 

62 sp_axis = 3 

63 naxis3 = header['NAXIS%d' % sp_axis] 

64 cunit3 = wcs.wcs.cunit[sp_axis-1] 

65 crval3 = wcs.wcs.crval[sp_axis-1] 

66 cdelt3 = wcs.wcs.cdelt[sp_axis-1] 

67 crpix3 = wcs.wcs.crpix[sp_axis-1] 

68 

69 freqs = np.arange(naxis3) + 1 

70 freqs = (freqs - crpix3) * cdelt3 + crval3 

71 

72 freqs = freqs * cunit3 

73 

74 # should wcs be transformed to a 1D case ? 

75 

76 return Spectrum1D(flux=data, wcs=wcs, meta=meta, spectral_axis=freqs)
