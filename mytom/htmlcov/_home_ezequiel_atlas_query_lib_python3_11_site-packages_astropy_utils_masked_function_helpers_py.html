402@dispatched_function 

403def bincount(x, weights=None, minlength=0): 

404 """Count number of occurrences of each value in array of non-negative ints. 

405 

406 Like `numpy.bincount`, but masked entries in ``x`` will be skipped. 

407 Any masked entries in ``weights`` will lead the corresponding bin to 

408 be masked. 

409 """ 

410 from astropy.utils.masked import Masked 

411 

412 if weights is not None: 

413 weights = np.asanyarray(weights) 

414 if isinstance(x, Masked) and x.ndim <= 1: 

415 # let other dimensions lead to errors. 

416 if weights is not None and weights.ndim == x.ndim: 

417 weights = weights[~x.mask] 

418 x = x.unmasked[~x.mask] 

419 mask = None 

420 if weights is not None: 

421 weights, w_mask = Masked._get_data_and_mask(weights) 

422 if w_mask is not None: 

423 mask = np.bincount(x, w_mask.astype(int), minlength=minlength).astype(bool) 

424 result = np.bincount(x, weights, minlength=0) 

425 return result, mask, None

476@dispatched_function 

477def block(arrays): 

478 # We need to override block since the numpy implementation can take two 

479 # different paths, one for concatenation, one for creating a large empty 

480 # result array in which parts are set. Each assumes array input and 

481 # cannot be used directly. Since it would be very costly to inspect all 

482 # arrays and then turn them back into a nested list, we just copy here the 

483 # second implementation, np.core.shape_base._block_slicing, since it is 

484 # shortest and easiest. 

485 from astropy.utils.masked import Masked 

486 

487 arrays, list_ndim, result_ndim, final_size = np.core.shape_base._block_setup(arrays) 

488 shape, slices, arrays = np.core.shape_base._block_info_recursion( 

489 arrays, list_ndim, result_ndim 

490 ) 

491 dtype = np.result_type(*[arr.dtype for arr in arrays]) 

492 F_order = all(arr.flags["F_CONTIGUOUS"] for arr in arrays) 

493 C_order = all(arr.flags["C_CONTIGUOUS"] for arr in arrays) 

494 order = "F" if F_order and not C_order else "C" 

495 result = Masked(np.empty(shape=shape, dtype=dtype, order=order)) 

496 for the_slice, arr in zip(slices, arrays): 

497 result[(Ellipsis,) + the_slice] = arr 

498 return result

501@dispatched_function 

502def broadcast_arrays(*args, subok=True): 

503 """Broadcast arrays to a common shape. 

504 

505 Like `numpy.broadcast_arrays`, applied to both unmasked data and masks. 

506 Note that ``subok`` is taken to mean whether or not subclasses of 

507 the unmasked data and masks are allowed, i.e., for ``subok=False``, 

508 `~astropy.utils.masked.MaskedNDArray` instances will be returned. 

509 """ 

510 from .core import Masked 

511 

512 are_masked = [isinstance(arg, Masked) for arg in args] 

513 data = [ 

514 (arg.unmasked if is_masked else arg) for arg, is_masked in zip(args, are_masked) 

515 ] 

516 results = np.broadcast_arrays(*data, subok=subok) 

517 

518 shape = results[0].shape if isinstance(results, list) else results.shape 

519 masks = [ 

520 (np.broadcast_to(arg.mask, shape, subok=subok) if is_masked else None) 

521 for arg, is_masked in zip(args, are_masked) 

522 ] 

523 results = [ 

524 (Masked(result, mask) if mask is not None else result) 

525 for (result, mask) in zip(results, masks) 

526 ] 

527 return results if len(results) > 1 else results[0]

751@dispatched_function 

752def piecewise(x, condlist, funclist, *args, **kw): 

753 """Evaluate a piecewise-defined function. 

754 

755 Like `numpy.piecewise` but for masked input array ``x``. 

756 Any masks in ``condlist`` are ignored. 

757 

758 """ 

759 # Copied implementation from numpy.lib.function_base.piecewise, 

760 # just to ensure output is Masked. 

761 n2 = len(funclist) 

762 # undocumented: single condition is promoted to a list of one condition 

763 if np.isscalar(condlist) or ( 

764 not isinstance(condlist[0], (list, np.ndarray)) and x.ndim != 0 

765 ): # pragma: no cover 

766 condlist = [condlist] 

767 

768 condlist = np.array(condlist, dtype=bool) 

769 n = len(condlist) 

770 

771 if n == n2 - 1: # compute the "otherwise" condition. 

772 condelse = ~np.any(condlist, axis=0, keepdims=True) 

773 condlist = np.concatenate([condlist, condelse], axis=0) 

774 n += 1 

775 elif n != n2: 

776 raise ValueError( 

777 f"with {n} condition(s), either {n} or {n + 1} functions are expected" 

778 ) 

779 

780 # The one real change... 

781 y = np.zeros_like(x) 

782 where = [] 

783 what = [] 

784 for k in range(n): 

785 item = funclist[k] 

786 if not callable(item): 

787 where.append(condlist[k]) 

788 what.append(item) 

789 else: 

790 vals = x[condlist[k]] 

791 if vals.size > 0: 

792 where.append(condlist[k]) 

793 what.append(item(vals, *args, **kw)) 

794 

795 for item, value in zip(where, what): 

796 y[item] = value 

797 

798 return y

879class MaskedFormat: 

880 """Formatter for masked array scalars. 

881 

882 For use in `numpy.array2string`, wrapping the regular formatters such 

883 that if a value is masked, its formatted string is replaced. 

884 

885 Typically initialized using the ``from_data`` class method. 

886 """ 

887 

888 def __init__(self, format_function): 

889 self.format_function = format_function 

890 # Special case for structured void and subarray: we need to make all the 

891 # format functions for the items masked as well. 

892 # TODO: maybe is a separate class is more logical? 

893 ffs = getattr(format_function, "format_functions", None) 

894 if ffs: 

895 # StructuredVoidFormat: multiple format functions to be changed. 

896 self.format_function.format_functions = [MaskedFormat(ff) for ff in ffs] 

897 

898 ff = getattr(format_function, "format_function", None) 

899 if ff: 

900 # SubarrayFormat: change format function for the elements. 

901 self.format_function.format_function = MaskedFormat(ff) 

902 

903 def __call__(self, x): 

904 if x.dtype.names: 

905 # The replacement of x with a list is needed because the function 

906 # inside StructuredVoidFormat iterates over x, which works for an 

907 # np.void but not an array scalar. 

908 return self.format_function([x[field] for field in x.dtype.names]) 

909 

910 if x.shape: 

911 # For a subarray pass on the data directly, since the 

912 # items will be iterated on inside the function. 

913 return self.format_function(x) 

914 

915 # Single element: first just typeset it normally, replace with masked 

916 # string if needed. 

917 string = self.format_function(x.unmasked[()]) 

918 if x.mask: 

919 # Strikethrough would be neat, but terminal needs a different 

920 # formatting than, say, jupyter notebook. 

921 # return "\x1B[9m"+string+"\x1B[29m" 

922 # return ''.join(s+'\u0336' for s in string) 

923 n = min(3, max(1, len(string))) 

924 return " " * (len(string) - n) + "\u2014" * n 

925 else: 

926 return string
